AI এজেন্ট বানানো ও ব্যবহার

এজেন্ট মানে AI যে নিজে নিজে সিদ্ধান্ত নেয়, টুল ব্যবহার করে, এবং মাল্টিপল স্টেপে কাজ শেষ করে।

Python দিয়ে Simple Agent
python — simple_agent.py 
import subprocess
from openai import OpenAI  # OpenRouter OpenAI-compatible

# OpenRouter দিয়ে ফ্রি মডেল ব্যবহার
client = OpenAI(
    base_url="https://openrouter.ai/api/v1",
    api_key="sk-or-...",
)

# টুল ডেফিনিশন
tools = [
    {
        "type": "function",
        "function": {
            "name": "run_bash",
            "description": "Run a bash command and return output",
            "parameters": {
                "type": "object",
                "properties": {
                    "command": {"type": "string"}
                },
                "required": ["command"]
            }
        }
    }
]

def run_bash(command: str) -> str:
    result = subprocess.run(
        command, shell=True, capture_output=True, text=True
    )
    return result.stdout + result.stderr

def agent_loop(task: str):
    messages = [{"role": "user", "content": task}]

    while True:
        response = client.chat.completions.create(
            model="google/gemini-2.0-flash-exp:free",
            messages=messages,
            tools=tools
        )
        msg = response.choices[0].message

        if not msg.tool_calls:
            print(f"\n✅ Done:\n{msg.content}")
            break

        for tool_call in msg.tool_calls:
            cmd = tool_call.function.arguments
            output = run_bash(cmd)
            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": output
            })

# ব্যবহার করুন
agent_loop("Create a Python FastAPI app with /health endpoint and run it")
LangChain Agent — ফ্রি Groq দিয়ে
python — langchain_agent.py 
from langchain_groq import ChatGroq
from langchain.agents import create_react_agent, AgentExecutor
from langchain_community.tools import ShellTool
from langchain_core.prompts import PromptTemplate

# ফ্রি Groq LLM
llm = ChatGroq(
    model="llama-3.3-70b-versatile",
    temperature=0,
    groq_api_key="your-groq-key"
)

# টুলস
tools = [ShellTool()]

# ReAct প্রম্পট টেমপ্লেট
prompt = PromptTemplate.from_template("""
You are a coding agent. Use tools to complete tasks.
Available tools: {tools}
Tool names: {tool_names}
Task: {input}
{agent_scratchpad}
""")

agent = create_react_agent(llm, tools, prompt)
executor = AgentExecutor(agent=agent, tools=tools, verbose=True)

result = executor.invoke({
    "input": "List files in current dir, then create hello.py"
})

MCP সার্ভার — AI কে সুপারপাওয়ার দিন

Model Context Protocol (MCP) হলো AI-কে external tools দেওয়ার স্ট্যান্ডার্ড। ফাইল সিস্টেম, ডেটাবেস, API — সব কিছু AI সরাসরি ব্যবহার করতে পারে।

bash — MCP ইন্সটল ও কনফিগার 
# ফাইলসিস্টেম MCP সার্ভার
npx -y @modelcontextprotocol/server-filesystem ~/projects

# PostgreSQL MCP
npx -y @modelcontextprotocol/server-postgres postgresql://localhost/mydb

# Git MCP (repository operations)
npx -y @modelcontextprotocol/server-git --repository ~/my-project

# Brave Search MCP (ওয়েব সার্চ)
BRAVE_API_KEY="your-key" npx -y @modelcontextprotocol/server-brave-search
Claude Code-এ MCP কনফিগার করুন
json — ~/.claude/claude.json

    }
  }
}

AutoGen দিয়ে Multi-Agent System
python — multi_agent.py 
import autogen

# ফ্রি Groq কনফিগ
config = [{
    "model": "llama-3.3-70b-versatile",
    "base_url": "https://api.groq.com/openai/v1",
    "api_key": "your-groq-key"
}]

# Planner এজেন্ট
planner = autogen.AssistantAgent(
    name="Planner",
    system_message="তুমি একজন সফটওয়্যার আর্কিটেক্ট। task analyze করো।",
    llm_config={"config_list": config}
)

# Coder এজেন্ট
coder = autogen.AssistantAgent(
    name="Coder",
    system_message="তুমি Python বিশেষজ্ঞ। Planner-এর plan অনুযায়ী কোড লেখো।",
    llm_config={"config_list": config}
)

# Reviewer এজেন্ট
reviewer = autogen.AssistantAgent(
    name="Reviewer",
    system_message="তুমি QA engineer। কোড রিভিউ করো এবং বাগ খোঁজো।",
    llm_config={"config_list": config}
)

# User proxy (human in the loop)
user = autogen.UserProxyAgent(
    name="User",
    code_execution_config={"work_dir": "coding", "use_docker": False}
)

# Group chat তৈরি
groupchat = autogen.GroupChat(
    agents=[user, planner, coder, reviewer],
    messages=[], max_round=15
)
manager = autogen.GroupChatManager(groupchat=groupchat,
                                     llm_config={"config_list": config})

# কাজ শুরু করুন
user.initiate_chat(manager,
    message="একটি REST API বানাও যেটা ইউজার অথেন্টিকেশন করে")
Shell Script দিয়ে AI Automation
bash — ai_review.sh 
#!/bin/bash
# git commit-এর আগে AI code review

DIFF=$(git diff --staged)

if [ -z "$DIFF" ]; then
    echo "কোনো staged পরিবর্তন নেই"
    exit 0
fi

echo "🤖 AI Code Review চলছে..."

# Ollama দিয়ে লোকাল রিভিউ (ফ্রি)
REVIEW=$(ollama run qwen2.5-coder "Review this git diff for bugs, security issues, and improvements:\n\n$DIFF")

echo "📋 AI Review:"
echo "$REVIEW"
echo
read -p "Commit করবেন? (y/n): " confirm

if [ "$confirm" = "y" ]; then
    git commit
else
    echo "❌ Commit বাতিল"
fi

git hook হিসেবে সেট করুন:
