Core Task 01

N+1 Query Problem — Eager Loading দিয়ে Fix

N+1 মানে কী? ধরো তুমি ১০০টি Post দেখাচ্ছ। প্রতিটি Post-এর Author নাম দেখাতে গেলে প্রতিটির জন্য আলাদা query যায় — মানে ১ (posts) + ১০০ (users) = ১০১টি query! এটাই N+1 সমস্যা। Eager Loading দিয়ে মাত্র ২-৩টি query-তে সব data আনা যায়।
❌ BAD — N+1 Problem (101 queries)
// PostController.php
public function index()
{
  // শুধু posts load করে — 1 query
  $posts = Post::all();

  foreach ($posts as $post) {
    // প্রতিটি post-এ আলাদা query!
    echo $post->user->name;
    // এখানে N বার query হয় 😱
  }
  // মোট: 1 + N = 101 queries!
}
✅ GOOD — Eager Loading (3 queries)
// PostController.php
public function index()
{
  // with() দিয়ে একসাথে load করো
  $posts = Post::with([
    'user',          // author
    'comments.user', // nested
    'tags'           // pivot
  ])->get();
  // মোট: মাত্র 3টি query ✅

  foreach ($posts as $post) {
    // কোনো extra query নেই!
    echo $post->user->name;
  }
}
PostController.php — Pagination + Eager Loading 
public function index(Request $request)
{
  $posts = Post::with([
    'user',
    'comments' => fn($q) => $q->with('user')->latest()->limit(3),
    'tags'
  ])
  ->withCount('comments')     // extra query না করে count নাও
  ->published()               // scope
  ->latest()
  ->paginate(20);

  return PostResource::collection($posts);
}

// ─── API Resource-এ whenLoaded() ব্যবহার করো ───
// PostResource.php
public function toArray($request): array
{
  return [
    'id'             => $this->id,
    'title'          => $this->title,
    'comments_count' => $this->comments_count,
    'author'         => new UserResource($this->whenLoaded('user')),
    'comments'       => CommentResource::collection($this->whenLoaded('comments')),
    'tags'           => TagResource::collection($this->whenLoaded('tags')),
  ];
}
with() → Eager Load load() → Lazy Eager Load withCount() → Count only whenLoaded() → Conditional in Resource select() → Specific columns only

ProductController.php — Endpoint 1: Products List 
use Illuminate\Support\Facades\Cache;

public function index(Request $request)
{
  // page, filter অনুযায়ী unique cache key
  $cacheKey = 'products.list.' . md5($request->fullUrl());

  $products = Cache::remember(
    $cacheKey,
    now()->addMinutes(60),  // 60 মিনিট cache
    function() use ($request) {
      return Product::with(['category', 'images'])
        ->when($request->category_id, fn($q, $id) =>
          $q->where('category_id', $id)
        )
        ->active()
        ->paginate(20);
    }
  );

  return ProductResource::collection($products);
}

// Update/Delete হলে cache clear করো!
public function update(UpdateProductRequest $request, Product $product)
{
  $product->update($request->validated());

  // সব product cache bust করো
  Cache::tags(['products'])->flush();

  return new ProductResource($product);
}

Core Task 03

Slow Queries — Proper Database Indexing

Index কীভাবে কাজ করে? Index হলো বইয়ের পেছনের সূচি। সূচি ছাড়া পুরো বই পড়তে হয় (Full Table Scan), সূচি থাকলে সরাসরি পাতায় যাওয়া যায়। ১ মিলিয়ন row-এ index ছাড়া query ~2 সেকেন্ড, index থাকলে ~2 মিলিসেকেন্ড!
MySQL — Slow Query Identify করো 
-- Slow query log চালু করো
SET GLOBAL slow_query_log = 'ON';
SET GLOBAL long_query_time = 1; -- 1 সেকেন্ডের বেশি log হবে
SET GLOBAL slow_query_log_file = '/var/log/mysql/slow.log';

-- EXPLAIN দিয়ে query plan দেখো
EXPLAIN SELECT * FROM orders
WHERE user_id = 100 AND status = 'pending';

-- Output এ type: ALL মানে Full Table Scan = BAD!
-- type: ref বা type: const মানে index ব্যবহার হচ্ছে = GOOD!

-- Laravel-এ
DB::enableQueryLog();
// ... your queries ...
dd(DB::getQueryLog()); // সব queries দেখো
database/migrations/add_performance_indexes.php 
// php artisan make:migration add_performance_indexes

public function up(): void
{
  // ─── orders table ───
  Schema::table('orders', function(Blueprint $table) {
    $table->index('user_id');
    $table->index('status');
    $table->index(['user_id', 'status']); // Composite — সবচেয়ে কার্যকর
    $table->index('created_at');           // date range query
    $table->index(['tenant_id', 'status', 'created_at']); // 3-column
  });

  // ─── posts table ───
  Schema::table('posts', function(Blueprint $table) {
    $table->unique('slug');                    // unique index
    $table->index(['is_published', 'created_at']); // blog listing
    $table->fullText(['title', 'content']);   // FULLTEXT search
  });

  // ─── users table ───
  Schema::table('users', function(Blueprint $table) {
    $table->index('email');      // login lookup
    $table->index('tenant_id');  // multi-tenant filter
    $table->index('created_at'); // date sort
  });
}

public function down(): void
{
  Schema::table('orders', fn($t) => $t->dropIndex(['user_id', 'status']));
}
  • WHERE clause-এ যে column ব্যবহার হয় সেখানে index দাও
  • JOIN করার সময় foreign key column-এ index অবশ্যই চাই
  • ORDER BY বা GROUP BY-এর column-এও index দিলে sort fast হয়
  • Unique constraint দরকার হলে unique() ব্যবহার করো (email, slug)
  • Composite index-এ column order গুরুত্বপূর্ণ — highest selectivity আগে
⚠️
সতর্কতা: বেশি index = INSERT/UPDATE/DELETE slow হয়, কারণ প্রতিটি write-এ index update হয়। Write-heavy table-এ অপ্রয়োজনীয় index দিও না।

Sanctum কী? Laravel-এর lightweight token-based authentication system। Mobile app বা SPA-র জন্য API token দেয়। প্রতিটি request-এ HTTP header-এ Bearer Token পাঠাতে হয়। Passport-এর মতো heavy OAuth নয় — simple ও fast।

AuthController.php — Login, Register & Logout 
namespace App\Http\Controllers\Api;

class AuthController extends Controller
{
  public function register(RegisterRequest $request)
  {
    $user = User::create([
      'name'     => $request->name,
      'email'    => $request->email,
      'password' => Hash::make($request->password),
    ]);

    $token = $user->createToken('mobile-app', ['*'])->plainTextToken;

    return response()->json([
      'token' => $token,
      'user'  => new UserResource($user),
    ], 201);
  }

  public function login(LoginRequest $request)
  {
    if (! Auth::attempt($request->only('email', 'password'))) {
      throw ValidationException::withMessages([
        'email' => ['Invalid credentials.'],
      ]);
    }

    $user  = Auth::user();

    // আগের সব tokens revoke করো (optional)
    $user->tokens()->delete();

    $token = $user->createToken(
      'api-token',
      ['products:read', 'orders:write']  // abilities/scopes
    )->plainTextToken;

    return response()->json([
      'token'      => $token,
      'token_type' => 'Bearer',
      'user'       => new UserResource($user),
    ]);
  }

  public function logout(Request $request)
  {
    // শুধু current token delete করো
    $request->user()->currentAccessToken()->delete();

    return response()->json(['message' => 'Successfully logged out.']);
  }
}

Multi-tenant Middleware — Data Isolation

Multi-tenancy মানে কী? একটি application-এ অনেক company (tenant) ব্যবহার করে, কিন্তু প্রত্যেকের data সম্পূর্ণ আলাদা। যেমন Shopify — হাজারো shop, কিন্তু এক codebase। Middleware দিয়ে প্রতিটি request-এ tenant identify করে Global Scope দিয়ে data isolated রাখা হয়।
app/Http/Middleware/IdentifyTenant.php 
// php artisan make:middleware IdentifyTenant

namespace App\Http\Middleware;

class IdentifyTenant
{
  public function handle(Request $request, Closure $next): Response
  {
    // Method 1: Subdomain থেকে identify করো
    $host   = $request->getHost();          // shop1.myapp.com
    $sub    = explode('.', $host)[0];          // shop1

    // Method 2: Header থেকে (API-r জন্য)
    // $sub = $request->header('X-Tenant-ID');

    $tenant = Cache::remember(
      "tenant.{$sub}", now()->addHour(),
      fn() => Tenant::where('subdomain', $sub)->first()
    );

    if (! $tenant) {
      return response()->json(['error' => 'Tenant not found.'], 404);
    }

    if (! $tenant->is_active) {
      return response()->json(['error' => 'Account suspended.'], 403);
    }

    // App সর্বত্র access করার জন্য bind করো
    app()->instance('tenant', $tenant);
    Config::set('app.tenant_id', $tenant->id);

    return $next($request);
  }
}
app/Traits/BelongsToTenant.php — Global Scope দিয়ে Auto Data Isolation 
trait BelongsToTenant
{
  public static function bootBelongsToTenant(): void
  {
    // প্রতিটি query-তে automatically tenant_id filter যোগ হবে
    static::addGlobalScope('tenant', function(Builder $query) {
      if ($tenantId = config('app.tenant_id')) {
        $query->where($query->getModel()->getTable() . '.tenant_id', $tenantId);
      }
    });

    // Create করার সময় tenant_id auto-set হবে
    static::creating(function($model) {
      $model->tenant_id ??= config('app.tenant_id');
    });
  }

  // Scope bypass করতে হলে
  public function scopeForAllTenants($query)
  {
    return $query->withoutGlobalScope('tenant');
  }
}

// ─── Model-এ use করো ───
class Product extends Model
{
  use BelongsToTenant;
  // ব্যস! এখন Product::all() শুধু current tenant-এর data দেবে
}

class Order extends Model { use BelongsToTenant; }
class Invoice extends Model { use BelongsToTenant; }

app/Jobs/SendWelcomeEmail.php — Job তৈরি করো 
// php artisan make:job SendWelcomeEmail

class SendWelcomeEmail implements ShouldQueue
{
  use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;

  public int $tries   = 3;   // কতবার retry করবে
  public int $timeout = 60;  // timeout seconds

  public function __construct(public User $user) {}

  public function handle(): void
  {
    Mail::to($this->user)
        ->send(new WelcomeEmail($this->user));
  }

  // Exponential backoff — retry delay বাড়তে থাকবে
  public function backoff(): array
  {
    return [30, 60, 300]; // 30s, 1min, 5min
  }

  // সব retry শেষ হলে এটা call হবে
  public function failed(\Throwable $e): void
  {
    Log::error('Welcome email failed', [
      'user_id' => $this->user->id,
      'error'   => $e->getMessage(),
    ]);

    // Admin-কে notify করো
    Notification::route('mail', 'admin@app.com')
      ->notify(new JobFailedNotification($this, $e));
  }
}

// ─── Controller থেকে dispatch করো ───
public function register(RegisterRequest $request)
{
  $user = User::create($request->validated());

  SendWelcomeEmail::dispatch($user)       // queue-এ পাঠাও
    ->onQueue('high')                      // high priority queue
    ->delay(now()->addSeconds(5));          // 5 সেকেন্ড পরে process করো

  // একসাথে অনেক job dispatch (Bus::batch)
  Bus::batch([
    new ProcessImage($user),
    new SendSlackNotification($user),
    new UpdateSearchIndex($user),
  ])->dispatch();

  return new UserResource($user);
}
