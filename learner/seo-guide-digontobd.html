Dynamic Sitemap — Laravel Route

app/Http/Controllers/SitemapController.phpPHP
namespace App\Http\Controllers;

use Illuminate\Http\Response;
use App\Models\{Product, Category, BlogPost, Page};

class SitemapController extends Controller
{
    public function index(): Response
    {
        $pages = Page::where('is_published', true)->get();
        $posts = BlogPost::latest()->published()->get();

        $content = view('sitemap.index', compact('pages', 'posts'));
        return response($content, 200)
            ->header('Content-Type', 'text/xml');
    }

    public function products(): Response
    {
        // shop subdomain-এর জন্য
        $products = Product::active()->select(['slug', 'updated_at'])->get();
        $categories = Category::active()->select(['slug', 'updated_at'])->get();

        $content = view('sitemap.products', compact('products', 'categories'));
        return response($content, 200)
            ->header('Content-Type', 'text/xml');
    }
}

SEO Service তৈরি করুন

app/Services/SeoService.phpPHP
namespace App\Services;

use Artesaos\SEOTools\Facades\{SEOMeta, OpenGraph, TwitterCard, JsonLd};

class SeoService
{
    public function setProductSeo(Product $product): void
    {
        SEOMeta::setTitle($product->name . ' | Digonto Shop');
        SEOMeta::setDescription(substr(strip_tags($product->description), 0, 160));
        SEOMeta::setKeywords($product->seo_keywords ?? []);
        SEOMeta::setCanonical(route('product.show', $product->slug));

        OpenGraph::setTitle($product->name);
        OpenGraph::setDescription($product->short_description);
        OpenGraph::addProperty('type', 'product');
        OpenGraph::addImage($product->image_url);

        // Product Schema
        JsonLd::setType('Product');
        JsonLd::setTitle($product->name);
        JsonLd::addValue('offers', [
            '@type'  => 'Offer',
            'price' => $product->price,
            'priceCurrency' => 'BDT',
            'availability' => $product->in_stock
                ? 'https://schema.org/InStock'
                : 'https://schema.org/OutOfStock',
        ]);
    }

    public function setBreadcrumbSchema(array $items): void
    {
        $listItems = [];
        foreach ($items as $i => $item) {
            $listItems[] = [
                '@type'    => 'ListItem',
                'position' => $i + 1,
                'name'     => $item['name'],
                'item'     => $item['url'],
            ];
        }
        JsonLd::addValue('breadcrumb', [
            '@type'           => 'BreadcrumbList',
            'itemListElement' => $listItems,
        ]);
    }
}

05
Vue SPA থেকে Inertia.js-এ মাইগ্রেশন
SEO সমস্যার সমাধান — SSR Setup
🚨 Vue SPA + Laravel-এর SEO সমস্যা Pure Vue SPA (API-based)-এ Google bot JavaScript render করতে পারে, কিন্তু delay হয়। ফলে meta tag, content দেরিতে দেখা যায়। Inertia.js দিয়ে SSR করলে first paint-এই সব content পাওয়া যায়।

Inertia.js কি এবং কেন দরকার?

❌
Pure Vue SPA সমস্যা
  • Googlebot JS render করতে সময় নেয়
  • Meta tag dynamically set হয়, late index হয়
  • Social share-এ blank preview
  • Initial page load blank screen
  • Crawl budget নষ্ট হয়
✅
Inertia.js SSR সুবিধা
  • Server থেকে HTML render হয়ে আসে
  • Meta tag server-side set হয়
  • Social share সঠিক OG image দেখায়
  • TTFB ও LCP উন্নত হয়
  • Laravel + Vue একই codebase-এ

import { createApp, h } from 'vue'
import { createInertiaApp } from '@inertiajs/vue3'
import { resolvePageComponent } from 'laravel-vite-plugin/inertia-helpers'

createInertiaApp({
    resolve: name =>
        resolvePageComponent(
            `./Pages/${name}.vue`,
            import.meta.glob('./Pages/**/*.vue')
        ),
    setup({ el, App, props, plugin }) {
        createApp({ render: () => h(App, props) })
            .use(plugin)
            .mount(el)
    },
    // SSR title callback
    title: title => `${title} | Digonto BD`,
})
resources/js/ssr.js — SSR Entry PointJavaScript
import { createSSRApp, h } from 'vue'
import { renderToString } from '@vue/server-renderer'
import { createInertiaApp } from '@inertiajs/vue3'
import { resolvePageComponent } from 'laravel-vite-plugin/inertia-helpers'

export default function render(page) {
    return createInertiaApp({
        page,
        render: renderToString,
        resolve: name =>
            resolvePageComponent(
                `./Pages/${name}.vue`,
                import.meta.glob('./Pages/**/*.vue')
            ),
        setup({ App, props, plugin }) {
            return createSSRApp({ render: () => h(App, props) })
                .use(plugin)
        },
    })
}

Image Optimization — WebP Conversion

app/Services/ImageService.phpPHP
use Intervention\Image\Facades\Image;

class ImageService
{
    public function optimizeAndConvert(UploadedFile $file, string $path): string
    {
        $filename = pathinfo($file->getClientOriginalName(), PATHINFO_FILENAME);

        // WebP তৈরি করুন (ছোট সাইজ)
        $webp = Image::make($file)
            ->resize(1200, null, function($c) {
                $c->aspectRatio();
                $c->upsize();
            })
            ->encode('webp', 80); // 80% quality

        $webpPath = $path . '/' . $filename . '.webp';
        Storage::put($webpPath, $webp);

        // Thumbnail তৈরি করুন
        $thumb = Image::make($file)
            ->fit(400, 400)
            ->encode('webp', 70);

        $thumbPath = $path . '/thumb_' . $filename . '.webp';
        Storage::put($thumbPath, $thumb);

        return $webpPath;
    }
}

Category Page SEO

Category Controller — SEO DataPHP
public function show(Category $category, Request $request)
{
    // Filter URL গুলো noindex করুন
    if ($request->hasAny(['sort', 'color', 'brand'])) {
        SEOMeta::setRobots('noindex, follow');
        SEOMeta::setCanonical(route('category.show', $category->slug));
    }

    // Pagination canonical
    if ($request->has('page') && $request->page > 1) {
        SEOMeta::setCanonical(
            route('category.show', $category->slug) . '?page=' . $request->page
        );
    }

    // Category description + product count SEO text
    $description = $category->seo_description
        ?? "বাংলাদেশে সেরা {$category->name} পণ্য কিনুন। "
        . "{$category->products_count}টি পণ্য পাবেন সেরা দামে।";

    SEOMeta::setDescription($description);
}
