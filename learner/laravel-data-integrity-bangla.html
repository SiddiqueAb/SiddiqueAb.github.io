02

Soft Delete — ডেটা কখনো মুছবেন নাSoft Delete Strategy

Hard Delete মানে ডেটা চিরতরে চলে যায়। Soft Delete মানে ডেটা থাকে, শুধু deleted_at timestamp সেট হয়। Accounting বা কোনো গুরুত্বপূর্ণ প্রজেক্টে সবসময় Soft Delete ব্যবহার করুন।

Migration + Model Setup
// ১. Migration-এ deleted_at column যোগ করুন
Schema::create('transactions', function (Blueprint $table) {
    $table->id();
    $table->foreignId('account_id')->constrained();
    $table->decimal('amount', 15, 2);
    $table->enum('type', ['debit', 'credit']);
    $table->string('reference')->nullable();
    $table->text('note')->nullable();
    $table->string('status')->default('active'); // active|reversed|cancelled
    $table->timestamps();
    $table->softDeletes(); // ← এটি deleted_at column যোগ করে
});

// ২. Model-এ SoftDeletes Trait ব্যবহার করুন
class Transaction extends Model
{
    use SoftDeletes; // ← এটাই যথেষ্ট!

    protected $fillable = ['account_id', 'amount', 'type', 'reference', 'note', 'status'];

    // Scope: শুধু Reversed দেখাও
    public function scopeReversed($query)
    {
        return $query->where('status', 'reversed');
    }
}

// ৩. Delete করলে সত্যিকার অর্থে delete হয় না
$transaction->delete();         // deleted_at = now()  — ডেটা আছে
$transaction->forceDelete();    // সত্যিকারের DELETE — শুধু বিশেষ ক্ষেত্রে

// ৪. Soft deleted রেকর্ড দেখতে
Transaction::withTrashed()->find($id);      // সব দেখাও
Transaction::onlyTrashed()->get();         // শুধু deleted দেখাও
$transaction->restore();               // Restore করো
✅
Best Practice: Invoice, Transaction, Journal Entry, Payment — এই ধরনের সব critical table-এ SoftDeletes অবশ্যই ব্যবহার করুন। এটি ৫ মিনিটের কাজ, কিন্তু ভবিষ্যতে অনেক বড় বিপদ থেকে বাঁচায়।
03

Database Transaction — "সব হবে নয়তো কিছুই না"DB::transaction() Mastery

Accounting-এ সবচেয়ে বিপজ্জনক পরিস্থিতি: Account থেকে টাকা কাটা গেল, কিন্তু অন্য account-এ যোগ হলো না। এটি ঠেকানোর একমাত্র উপায় Database Transaction।

📐 ACID নীতি
A — Atomicity: সব কাজ হবে অথবা কিছুই হবে না
C — Consistency: ডেটা সবসময় সঠিক অবস্থায় থাকবে
I — Isolation: দুটো transaction একে অপরকে প্রভাবিত করবে না
D — Durability: একবার save হলে power cut-এও থাকবে
TransactionService.php — সঠিক পদ্ধতি
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Log;

class AccountingService
{
    /**
     * Double Entry Accounting — টাকা transfer করা
     * Debit এক account, Credit অন্য account
     */
    public function transfer(array $data): array
    {
        return DB::transaction(function () use ($data) {

            // ১. Source account লক করুন (Pessimistic Locking)
            $source = Account::lockForUpdate()->findOrFail($data['from_account']);
            $dest   = Account::lockForUpdate()->findOrFail($data['to_account']);

            // ২. Balance check করুন
            if ($source->balance < $data['amount']) {
                throw new InsufficientBalanceException('Insufficient balance');
            }

            // ৩. Reference number তৈরি করুন
            $ref = 'TXN-' . strtoupper(uniqid());

            // ৪. Debit entry
            $debitEntry = Transaction::create([
                'account_id'  => $source->id,
                'amount'      => $data['amount'],
                'type'        => 'debit',
                'reference'   => $ref,
                'note'        => $data['note'] ?? null,
                'created_by'  => auth()->id(),
            ]);

            // ৫. Credit entry
            $creditEntry = Transaction::create([
                'account_id'  => $dest->id,
                'amount'      => $data['amount'],
                'type'        => 'credit',
                'reference'   => $ref,
                'note'        => $data['note'] ?? null,
                'created_by'  => auth()->id(),
            ]);

            // ৬. Balance আপডেট করুন
            $source->decrement('balance', $data['amount']);
            $dest->increment('balance', $data['amount']);

            return [
                'reference'    => $ref,
                'debit_entry'  => $debitEntry,
                'credit_entry' => $creditEntry,
            ];

        // যেকোনো Exception হলে সব Rollback হবে স্বয়ংক্রিয়ভাবে
        });
    }
}
🚨
Race Condition ঠেকান: lockForUpdate() ব্যবহার করুন। না হলে দুটো request একসাথে আসলে balance দুইবার কাটা যাবে। এটি Accounting-এর সবচেয়ে মারাত্মক বাগ।

Model Versioning — প্রতিটি পরিবর্তনের Version রাখুনVersion History / Snapshot

Audit Log শুধু "কী পরিবর্তন হয়েছে" দেখায়। কিন্তু Versioning মানে প্রতিটি save-এর সম্পূর্ণ snapshot রাখা — যেকোনো সময় যেকোনো পুরনো version-এ ফিরে যাওয়া যাবে।

Model Version History System
// model_versions table
Schema::create('model_versions', function (Blueprint $table) {
    $table->id();
    $table->morphs('versionable');   // model_type + model_id (polymorphic)
    $table->integer('version_number');
    $table->json('snapshot');         // সম্পূর্ণ row-এর JSON copy
    $table->foreignId('created_by')->constrained('users');
    $table->string('change_reason')->nullable();
    $table->timestamp('created_at');
});

// Versionable Trait — যেকোনো Model-এ ব্যবহার করুন
trait HasVersions
{
    protected static function bootHasVersions(): void
    {
        // প্রতিটি Update-এ স্বয়ংক্রিয়ভাবে version তৈরি হবে
        static::updating(function ($model) {
            $lastVersion = ModelVersion::where([
                'versionable_type' => get_class($model),
                'versionable_id'   => $model->id,
            ])->max('version_number') ?? 0;

            ModelVersion::create([
                'versionable_type' => get_class($model),
                'versionable_id'   => $model->id,
                'version_number'   => $lastVersion + 1,
                'snapshot'         => $model->getOriginal(), // আগের মান save করো
                'created_by'       => auth()->id(),
                'change_reason'    => request()->get('change_reason'),
            ]);
        });
    }

    // যেকোনো পুরনো version-এ ফিরে যান
    public function revertToVersion(int $versionNumber): bool
    {
        $version = ModelVersion::where([
            'versionable_type'   => get_class($this),
            'versionable_id'     => $this->id,
            'version_number'     => $versionNumber,
        ])->firstOrFail();

        return $this->update($version->snapshot);
    }

    public function versions()
    {
        return $this->morphMany(ModelVersion::class, 'versionable')->orderByDesc('version_number');
    }
}

// ব্যবহার:
$invoice = Invoice::find(1);
$invoice->versions()->get();    // সব version দেখুন
$invoice->revertToVersion(3);  // Version 3-এ ফিরে যান

ReversalService.php — সম্পূর্ণ Reversal System
class ReversalService
{
    public function reverseTransaction(int $transactionId, string $reason): array
    {
        return DB::transaction(function () use ($transactionId, $reason) {

            // ১. মূল Transaction খুঁজুন এবং lock করুন
            $original = Transaction::lockForUpdate()->findOrFail($transactionId);

            // ২. ইতিমধ্যে reversed কিনা check করুন
            if ($original->status === 'reversed') {
                throw new AlreadyReversedException('এই Transaction ইতিমধ্যে বাতিল করা হয়েছে।');
            }

            // ৩. Reversal reference তৈরি করুন
            $reversalRef = 'REV-' . $original->reference;

            // ৪. Reversal Entry তৈরি করুন (বিপরীত type)
            $reversalEntry = Transaction::create([
                'account_id'        => $original->account_id,
                'amount'            => $original->amount,
                'type'              => $original->type === 'debit' ? 'credit' : 'debit', // বিপরীত!
                'reference'         => $reversalRef,
                'note'              => 'REVERSAL: ' . $reason,
                'status'            => 'reversal',
                'reversed_entry_id' => $original->id, // কোন entry-র reversal
                'created_by'        => auth()->id(),
            ]);

            // ৫. মূল entry-কে reversed mark করুন
            $original->update([
                'status'           => 'reversed',
                'reversed_by'      => auth()->id(),
                'reversed_at'      => now(),
                'reversal_reason'  => $reason,
            ]);

            // ৬. Account balance ঠিক করুন
            $account = Account::lockForUpdate()->find($original->account_id);
            if ($original->type === 'debit') {
                $account->increment('balance', $original->amount); // Debit ছিল → Balance ফেরত
            } else {
                $account->decrement('balance', $original->amount); // Credit ছিল → Balance কমাও
            }

            // ৭. Audit Log
            AuditService::log($original, 'reversed', $original->toArray(), $reason);

            return [
                'original'  => $original->fresh(),
                'reversal'  => $reversalEntry,
            ];
        });
    }
}

// Controller-এ ব্যবহার
public function reverse(Request $request, int $id, ReversalService $service)
{
    $request->validate(['reason' => 'required|string|min:10']);

    $result = $service->reverseTransaction($id, $request->reason);

    return back()->with('success', 'Transaction সফলভাবে বাতিল করা হয়েছে।');
}

Input Validation ও Data GuardBulletproof Validation Strategy

ভুল entry-র সবচেয়ে বড় কারণ হলো সঠিক validation না থাকা। Laravel-এর Form Request ব্যবহার করে সব validation একটি জায়গায় রাখুন।

TransactionRequest.php — Bulletproof Validation
class CreateTransactionRequest extends FormRequest
{
    public function authorize(): bool
    {
        // শুধু accountant বা admin করতে পারবে
        return $this->user()->hasAnyRole(['accountant', 'admin']);
    }

    public function rules(): array
    {
        return [
            'from_account' => ['required', 'exists:accounts,id',
                                  Rule::notIn([$this->to_account])], // নিজেকে নিজে transfer না
            'to_account'   => ['required', 'exists:accounts,id'],
            'amount'       => ['required', 'numeric', 'min:1', 'max:999999999'],
            'note'         => ['nullable', 'string', 'max:500'],
            'transaction_date' => ['required', 'date', 'before_or_equal:today'],
        ];
    }

    public function messages(): array
    {
        return [
            'amount.min'            => 'পরিমাণ ০ এর বেশি হতে হবে।',
            'from_account.not_in'   => 'Source ও Destination account আলাদা হতে হবে।',
            'transaction_date.before_or_equal' => 'ভবিষ্যতের তারিখ দেওয়া যাবে না।',
        ];
    }

    // Data sanitize করুন
    protected function prepareForValidation(): void
    {
        $this->merge([
            'amount' => round(abs($this->amount), 2), // Negative amount ঠেকান
        ]);
    }
}

// Custom Rule: Period Lock — পুরনো তারিখে entry ঠেকান
class PeriodNotLocked implements Rule
{
    public function passes($attribute, $value): bool
    {
        // Closed accounting period-এ entry করা যাবে না
        return !AccountingPeriod::where('status', 'closed')
            ->whereDate('end_date', '>=', $value)
            ->exists();
    }

    public function message(): string
    {
        return 'এই accounting period বন্ধ করা হয়েছে। পুরনো তারিখে entry দেওয়া যাবে না।';
    }
}

Queue ও Job-এ IdempotencySafe Background Jobs

Queue Job retry হলে একই কাজ দুইবার হয়ে যেতে পারে — যেমন দুইবার payment process বা দুইবার email। এটি ঠেকাতে Idempotency key ব্যবহার করতে হবে।

ProcessPaymentJob.php — Idempotent Job
class ProcessPaymentJob implements ShouldQueue
{
    use Dispatchable, Queueable, SerializesModels;

    public $tries = 3;       // সর্বোচ্চ ৩ বার retry
    public $backoff = [30, 60, 120]; // 30s, 60s, 120s পর retry

    public function __construct(
        private readonly int $transactionId,
        private readonly string $idempotencyKey // Unique key
    ) {}

    public function handle(): void
    {
        // ১. ইতিমধ্যে process হয়েছে কিনা check করুন
        $processed = Cache::get('job_processed:' . $this->idempotencyKey);

        if ($processed) {
            Log::info('Job already processed, skipping', ['key' => $this->idempotencyKey]);
            return; // দ্বিতীয়বার skip করুন
        }

        // ২. Atomic lock — একসাথে দুটো process ঠেকান
        $lock = Cache::lock('payment:' . $this->transactionId, 60);

        $lock->block(10, function () {
            // ৩. আসল কাজ করুন
            $transaction = Transaction::findOrFail($this->transactionId);
            $transaction->update(['status' => 'processed']);

            // ৪. সফল হলে Cache-এ mark করুন (24 ঘণ্টা)
            Cache::put('job_processed:' . $this->idempotencyKey, true, now()->addDay());
        });
    }

    public function failed(\Throwable $exception): void
    {
        // সব retry শেষে fail হলে alert পাঠান
        Log::critical('Payment Job Failed', [
            'transaction_id' => $this->transactionId,
            'error'          => $exception->getMessage(),
        ]);
        // Notification::send(admin, new PaymentJobFailedNotification(...))
    }
}

📊 Accounting Period Lock

AccountingPeriodService.php
// Month/Year শেষে period lock করুন
class AccountingPeriodService
{
    public function closePeriod(int $year, int $month): void
    {
        // শুধু Admin বা Finance Manager করতে পারবে
        abort_unless(auth()->user()->can('lock accounting period'), 403);

        DB::transaction(function () use ($year, $month) {
            // Period close করুন
            AccountingPeriod::updateOrCreate(
                ['year' => $year, 'month' => $month],
                ['status' => 'closed', 'closed_by' => auth()->id(), 'closed_at' => now()]
            );

            // Closing balance snapshot রাখুন
            Account::each(function ($account) use ($year, $month) {
                AccountBalance::create([
                    'account_id' => $account->id,
                    'year'       => $year,
                    'month'      => $month,
                    'balance'    => $account->balance,
                ]);
            });
        });
    }
}
