🏛️ OOP — Class ও Object
Dartক্লাস ও অবজেক্ট
// Class তৈরি
class Student {
  // Properties
  String name;
  int age;
  double gpa;

  // Constructor
  Student({
    required this.name,
    required this.age,
    this.gpa = 0.0,
  });

  // Method (কাজ করার ফাংশন)
  void introduce() {
    print('আমি ${this.name}, বয়স ${this.age}');
  }

  // Getter
  bool get isFirstClass => gpa >= 3.5;

  // toString override
  @override
  String toString() => 'Student($name, $age)';
}

// Object তৈরি ও ব্যবহার
void main() {
  var s1 = Student(name: 'নাফিসা', age: 19, gpa: 3.9);
  s1.introduce();
  print(s1.isFirstClass); // true
}

// Inheritance (উত্তরাধিকার)
class GraduateStudent extends Student {
  String thesis;

  GraduateStudent({
    required String name,
    required int age,
    required this.thesis,
  }) : super(name: name, age: age);
}

  // ২ সেকেন্ড অপেক্ষা (API call simulate)

  print('লোড হচ্ছে...');
  String name = await fetchUserName();
  print('ব্যবহারকারী: $name');
}

// Error Handling
Future<void> loadData() async {
  try {
    var data = await fetchUserName();
    print(data);
  } catch (e) {
    print('সমস্যা হয়েছে: $e');
  } finally {
    print('সবসময় চলবে');
  }
}

🔢 setState — সবচেয়ে সহজ উপায়
Dartকাউন্টার অ্যাপ
class CounterApp extends StatefulWidget {
  const CounterApp({super.key});
  @override
  State<CounterApp> createState() => _CounterAppState();
}

class _CounterAppState extends State<CounterApp> {
  int _count = 0;

  void _increment() {
    setState(() {      // এটা UI আপডেট করে
      _count++;
    });
  }

  void _decrement() {
    if (_count > 0) {
      setState(() => _count--);
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('কাউন্টার')),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text('$_count', style: Theme.of(context).textTheme.displayLarge),
            const SizedBox(height: 24),
            Row(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                FloatingActionButton(onPressed: _decrement, child: const Icon(Icons.remove)),
                const SizedBox(width: 24),
                FloatingActionButton(onPressed: _increment, child: const Icon(Icons.add)),
              ],
            ),
          ],
        ),
      ),
    );
  }
}
🔵 Provider — জনপ্রিয় State Management

DartProvider Pattern
// ১. Model তৈরি করুন
class CartProvider extends ChangeNotifier {
  List<Product> _items = [];

  List<Product> get items => _items;
  int get itemCount => _items.length;

  void addItem(Product product) {
    _items.add(product);
    notifyListeners();  // UI কে জানায়
  }

  void removeItem(Product product) {
    _items.remove(product);
    notifyListeners();
  }

  double get totalPrice =>
    _items.fold(0, (sum, item) => sum + item.price);
}

// ২. main() তে Provider যোগ করুন
void main() {
  runApp(
    ChangeNotifierProvider(
      create: (_) => CartProvider(),
      child: const MyApp(),
    ),
  );
}

// ৩. Widget-এ ব্যবহার করুন
class CartScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    // data পড়া ও watch করা
    final cart = context.watch<CartProvider>();

    return Scaffold(
      appBar: AppBar(
        title: Text('কার্ট (${cart.itemCount})'),
      ),
      body: ListView.builder(
        itemCount: cart.items.length,
        itemBuilder: (ctx, i) => ListTile(
          title: Text(cart.items[i].name),
          subtitle: Text('৳ ${cart.items[i].price}'),
        ),
      ),
      bottomNavigationBar: Padding(
        padding: const EdgeInsets.all(16),
        child: Text('মোট: ৳ ${cart.totalPrice}'),
      ),
    );
  }
}

import 'dart:convert';
import 'package:http/http.dart' as http;

// Model Class
class Post {
  final int id;
  final String title;
  final String body;

  const Post({required this.id, required this.title, required this.body});

  // JSON থেকে Object বানানো
  factory Post.fromJson(Map<String, dynamic> json) {
    return Post(
      id: json['id'],
      title: json['title'],
      body: json['body'],
    );
  }
}

// API Service
class ApiService {
  static const baseUrl = 'https://jsonplaceholder.typicode.com';

  // GET — ডেটা পড়া
  static Future<List<Post>> getPosts() async {
    final response = await http.get(Uri.parse('$baseUrl/posts'));

    if (response.statusCode == 200) {
      final jsonList = jsonDecode(response.body) as List;
      return jsonList.map((json) => Post.fromJson(json)).toList();
    } else {
      throw Exception('ডেটা লোড ব্যর্থ!');
    }
  }

  // POST — ডেটা পাঠানো
  static Future<void> createPost(String title, String body) async {
    final response = await http.post(
      Uri.parse('$baseUrl/posts'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({'title': title, 'body': body}),
    );
    print('Created: ${response.statusCode}');
  }
}

// Widget-এ FutureBuilder দিয়ে দেখানো
class PostListScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('পোস্ট লিস্ট')),
      body: FutureBuilder<List<Post>>(
        future: ApiService.getPosts(),
        builder: (context, snapshot) {
          // লোড হচ্ছে
          if (snapshot.connectionState == ConnectionState.waiting) {
            return const Center(child: CircularProgressIndicator());
          }
          // সমস্যা হয়েছে
          if (snapshot.hasError) {
            return Center(child: Text('সমস্যা: ${snapshot.error}'));
          }
          // ডেটা পাওয়া গেছে
          final posts = snapshot.data!;
          return ListView.builder(
            itemCount: posts.length,
            itemBuilder: (_, i) => Card(
              margin: EdgeInsets.all(8),
              child: ListTile(
                title: Text(posts[i].title),
                subtitle: Text(posts[i].body.substring(0, 50)),
              ),
            ),
          );
        },
      ),
    );
  }
}

class Todo {
  final String id;
  final String title;
  final String description;
  final bool isCompleted;
  final DateTime createdAt;

  const Todo({
    required this.id,
    required this.title,
    this.description = '',
    this.isCompleted = false,
    required this.createdAt,
  });

  // copyWith — পরিবর্তন করে নতুন object বানানো
  Todo copyWith({
    String? id,
    String? title,
    String? description,
    bool? isCompleted,
  }) {
    return Todo(
      id: id ?? this.id,
      title: title ?? this.title,
      description: description ?? this.description,
      isCompleted: isCompleted ?? this.isCompleted,
      createdAt: this.createdAt,
    );
  }
}
⚡ Todo Provider
Dartlib/providers/todo_provider.dart
import 'package:flutter/foundation.dart';
import 'package:uuid/uuid.dart';
import '../models/todo_model.dart';

class TodoProvider extends ChangeNotifier {
  final List<Todo> _todos = [];
  final _uuid = const Uuid();

  // Getters
  List<Todo> get allTodos => List.unmodifiable(_todos);
  List<Todo> get pending => _todos.where((t) => !t.isCompleted).toList();
  List<Todo> get completed => _todos.where((t) => t.isCompleted).toList();
  int get totalCount => _todos.length;

  // Todo যোগ করুন
  void addTodo(String title, {String description = ''}) {
    if (title.trim().isEmpty) return;

    _todos.insert(0, Todo(
      id: _uuid.v4(),
      title: title.trim(),
      description: description,
      createdAt: DateTime.now(),
    ));
    notifyListeners();
  }

  // Complete toggle
  void toggleComplete(String id) {
    final index = _todos.indexWhere((t) => t.id == id);
    if (index != -1) {
      _todos[index] = _todos[index].copyWith(
        isCompleted: !_todos[index].isCompleted,
      );
      notifyListeners();
    }
  }

  // Delete
  void deleteTodo(String id) {
    _todos.removeWhere((t) => t.id == id);
    notifyListeners();
  }

  // সব completed বাদ দিন
  void clearCompleted() {
    _todos.removeWhere((t) => t.isCompleted);
    notifyListeners();
  }
}
🖥️ Todo Screen — মূল UI
Dartlib/screens/todo_screen.dart
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import '../providers/todo_provider.dart';

class TodoScreen extends StatefulWidget {
  const TodoScreen({super.key});
  @override
  State<TodoScreen> createState() => _TodoScreenState();
}

class _TodoScreenState extends State<TodoScreen> {
  final _controller = TextEditingController();
  final _formKey = GlobalKey<FormState>();

  void _submit() {
    if (!_formKey.currentState!.validate()) return;

    context.read<TodoProvider>().addTodo(_controller.text);
    _controller.clear();
  }

  @override
  void dispose() {
    _controller.dispose();  // Memory leak রোধ করুন
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final todos = context.watch<TodoProvider>().allTodos;

    return Scaffold(
      backgroundColor: const Color(0xFFF5F5F5),
      appBar: AppBar(
        title: const Text('আমার কাজের তালিকা'),
        actions: [
          IconButton(
            icon: const Icon(Icons.delete_sweep),
            onPressed: () => context.read<TodoProvider>().clearCompleted(),
            tooltip: 'সম্পন্ন মুছুন',
          ),
        ],
      ),
      body: Column(
        children: [
          // Input Section
          Padding(
            padding: const EdgeInsets.all(16),
            child: Form(
              key: _formKey,
              child: Row(
                children: [
                  Expanded(
                    child: TextFormField(
                      controller: _controller,
                      decoration: const InputDecoration(
                        hintText: 'নতুন কাজ লিখুন...',
                        border: OutlineInputBorder(),
                        filled: true,
                        fillColor: Colors.white,
                      ),
                      validator: (val) =>
                        (val == null || val.isEmpty) ? 'কাজ লিখুন' : null,
                      onFieldSubmitted: (_) => _submit(),
                    ),
                  ),
                  const SizedBox(width: 8),
                  ElevatedButton(
                    onPressed: _submit,
                    style: ElevatedButton.styleFrom(
                      padding: const EdgeInsets.all(16),
                    ),
                    child: const Icon(Icons.add),
                  ),
                ],
              ),
            ),
          ),

          // List Section
          Expanded(
            child: todos.isEmpty
              ? const Center(child: Text('কোনো কাজ নেই! ✨'))
              : ListView.builder(
                  padding: const EdgeInsets.symmetric(horizontal: 16),
                  itemCount: todos.length,
                  itemBuilder: (_, i) => TodoItem(todo: todos[i]),
                ),
          ),
        ],
      ),
    );
  }
}

// আলাদা Widget — পরিষ্কার কোড
class TodoItem extends StatelessWidget {
  final Todo todo;
  const TodoItem({super.key, required this.todo});

  @override
  Widget build(BuildContext context) {
    return Card(
      margin: const EdgeInsets.only(bottom: 8),
      child: ListTile(
        leading: Checkbox(
          value: todo.isCompleted,
          onChanged: (_) => context
            .read<TodoProvider>()
            .toggleComplete(todo.id),
        ),
        title: Text(
          todo.title,
          style: TextStyle(
            decoration: todo.isCompleted
              ? TextDecoration.lineThrough
              : null,
            color: todo.isCompleted ? Colors.grey : null,
          ),
        ),
        trailing: IconButton(
          icon: const Icon(Icons.delete, color: Colors.red),
          onPressed: () => context
            .read<TodoProvider>()
            .deleteTodo(todo.id),
        ),
      ),
    );
  }
}
