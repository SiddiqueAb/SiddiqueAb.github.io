ডাউনলোড ফোল্ডারের সব ফাইল স্বয়ংক্রিয়ভাবে ধরন অনুযায়ী সাজাবে।

Python - ফাইল সর্টার 
import os
import shutil
from pathlib import Path

# ফাইলের ধরন অনুযায়ী ফোল্ডার
FOLDER_MAP = {
    ".jpg": "ছবি", ".png": "ছবি", ".gif": "ছবি",
    ".pdf": "ডকুমেন্ট", ".docx": "ডকুমেন্ট", ".txt": "ডকুমেন্ট",
    ".mp4": "ভিডিও", ".avi": "ভিডিও", ".mkv": "ভিডিও",
    ".mp3": "অডিও", ".wav": "অডিও",
    ".zip": "আর্কাইভ", ".rar": "আর্কাইভ",
    ".py": "কোড", ".js": "কোড", ".html": "কোড",
}

def sort_files(folder):
    folder_path = Path(folder)
    moved = 0
    
    for file in folder_path.iterdir():
        if file.is_file():
            ext = file.suffix.lower()
            dest_folder = FOLDER_MAP.get(ext, "অন্যান্য")
            dest_path = folder_path / dest_folder
            dest_path.mkdir(exist_ok=True)
            shutil.move(str(file), str(dest_path / file.name))
            moved += 1
    
    print(f"✅ {moved}টি ফাইল সাজানো হয়েছে!")

# চালান
sort_files("C:/Users/আপনার_নাম/Downloads")

Python - প্রোডাক্ট ডেটা স্ক্র্যাপিং 
import requests
from bs4 import BeautifulSoup
import csv

def scrape_products(url):
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)"
    }
    response = requests.get(url, headers=headers)
    soup = BeautifulSoup(response.text, "html.parser")
    
    products = []
    
    # CSS selector দিয়ে এলিমেন্ট খুঁজুন
    for item in soup.select(".product-card"):
        name = item.select_one(".product-name").text.strip()
        price = item.select_one(".price").text.strip()
        products.append({"নাম": name, "দাম": price})
    
    # CSV তে সেভ করুন
    with open("products.csv", "w", encoding="utf-8", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=["নাম", "দাম"])
        writer.writeheader()
        writer.writerows(products)
    
    print(f"✅ {len(products)}টি প্রোডাক্ট সেভ হয়েছে!")
    return products

Python - ইমেইল পাঠান 
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
from email.mime.base import MIMEBase
from email import encoders

def send_email(to_email, subject, body, attachment=None):
    sender = "আপনার_ইমেইল@gmail.com"
    password = "আপনার_app_password"
    
    # মেসেজ তৈরি করুন
    msg = MIMEMultipart()
    msg["From"] = sender
    msg["To"] = to_email
    msg["Subject"] = subject
    msg.attach(MIMEText(body, "html", "utf-8"))
    
    # অ্যাটাচমেন্ট যোগ করুন
    if attachment:
        with open(attachment, "rb") as f:
            part = MIMEBase("application", "octet-stream")
            part.set_payload(f.read())
        encoders.encode_base64(part)
        part.add_header("Content-Disposition", f"attachment; filename={attachment}")
        msg.attach(part)
    
    # SMTP দিয়ে পাঠান
    with smtplib.SMTP("smtp.gmail.com", 587) as server:
        server.starttls()
        server.login(sender, password)
        server.sendmail(sender, to_email, msg.as_string())
    
    print(f"✅ ইমেইল পাঠানো হয়েছে: {to_email}")

pyautogui দিয়ে Python স্ক্রিনে যা দেখা যায় তা কন্ট্রোল করতে পারে। মাউস ক্লিক, কীবোর্ড টাইপ, স্ক্রিনশট, স্ক্রল — সব কিছু অটোমেট করা যায়।

Python - pyautogui 
import pyautogui
import time

# সেফটি নেট — কোণায় মাউস গেলে বন্ধ হবে
pyautogui.FAILSAFE = True

# স্ক্রিনের সাইজ দেখুন
print(pyautogui.size())  # (1920, 1080)

# মাউস মুভ করুন
pyautogui.moveTo(500, 300, duration=1)

# ক্লিক করুন
pyautogui.click(500, 300)

# ডাবল ক্লিক
pyautogui.doubleClick(500, 300)

# টাইপ করুন
pyautogui.typewrite("Hello World!", interval=0.05)

# কীবোর্ড শর্টকাট
pyautogui.hotkey("ctrl", "c")  # কপি
pyautogui.hotkey("ctrl", "v")  # পেস্ট
pyautogui.hotkey("ctrl", "s")  # সেভ

# স্ক্রিনশট নিন
screenshot = pyautogui.screenshot()
screenshot.save("screenshot.png")

# স্ক্রোল করুন
pyautogui.scroll(-3)  # নিচে স্ক্রোল

# কোনো ছবি স্ক্রিনে খুঁজুন
location = pyautogui.locateOnScreen("button.png", confidence=0.9)
if location:
    pyautogui.click(location)
Python - ফর্ম Auto-fill 
import pyautogui
import time
import csv

def auto_fill_form(data_file):
    with open(data_file, encoding="utf-8") as f:
        reader = csv.DictReader(f)
        
        for row in reader:
            # নাম ফিল্ডে ক্লিক করুন
            pyautogui.click(400, 200)
            pyautogui.typewrite(row["নাম"])
            
            # পরের ফিল্ডে যান
            pyautogui.press("tab")
            pyautogui.typewrite(row["ইমেইল"])
            
            # সাবমিট করুন
            pyautogui.press("enter")
            time.sleep(2)  # লোড হওয়ার জন্য অপেক্ষা
            
    print("✅ সব ফর্ম পূরণ হয়েছে!")

📌 Selenium কী?

Selenium দিয়ে Chrome/Firefox ব্রাউজার সম্পূর্ণ নিয়ন্ত্রণ করা যায়। JavaScript দিয়ে তৈরি ওয়েবসাইট স্ক্র্যাপ করা, অটো লগইন, ফর্ম পূরণ — সব সম্ভব।

Python - Selenium বেসিক 
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from webdriver_manager.chrome import ChromeDriverManager
import time

# Chrome চালু করুন
options = webdriver.ChromeOptions()
options.add_argument("--headless")  # ব্রাউজার না দেখিয়ে চালান

driver = webdriver.Chrome()

try:
    # ওয়েবসাইটে যান
    driver.get("https://google.com")
    
    # সার্চ বক্স খুঁজুন
    search_box = driver.find_element(By.NAME, "q")
    
    # টাইপ করুন
    search_box.send_keys("Python Automation বাংলা")
    search_box.submit()
    
    # লোড হওয়ার জন্য অপেক্ষা
    wait = WebDriverWait(driver, 10)
    results = wait.until(
        EC.presence_of_all_elements_located((By.CSS_SELECTOR, "h3"))
    )
    
    # রেজাল্ট প্রিন্ট করুন
    for result in results[:5]:
        print(result.text)
    
    # স্ক্রিনশট নিন
    driver.save_screenshot("google_result.png")

finally:
    driver.quit()
Python - অটো লগইন 
def auto_login(driver, url, username, password):
    driver.get(url)
    time.sleep(2)
    
    # ইউজারনেম ফিল্ড
    user_field = driver.find_element(By.ID, "username")
    user_field.clear()
    user_field.send_keys(username)
    
    # পাসওয়ার্ড ফিল্ড
    pass_field = driver.find_element(By.ID, "password")
    pass_field.clear()
    pass_field.send_keys(password)
    
    # লগইন বাটন ক্লিক
    driver.find_element(By.XPATH, "//button[@type='submit']").click()
    time.sleep(3)
    
    print("✅ লগইন সফল!")
    return driver
By মেথডউদাহরণকখন ব্যবহার করবেন
By.IDfind_element(By.ID, "btn")ইউনিক ID থাকলে
By.NAMEfind_element(By.NAME, "q")ফর্ম ইনপুটের জন্য
By.CLASS_NAMEfind_element(By.CLASS_NAME, "btn")CSS ক্লাস দিয়ে
By.CSS_SELECTORfind_element(By.CSS_SELECTOR, ".nav a")জটিল সিলেকশন
By.XPATHfind_element(By.XPATH, "//div[@id='x']")যেকোনো অবস্থান

Task Scheduling — নির্দিষ্ট সময়ে কাজ চালানো

📌 Scheduling কেন দরকার?

প্রতিদিন রাত ১২টায় ব্যাকআপ নেওয়া, প্রতি সোমবার সকালে রিপোর্ট পাঠানো, প্রতি ঘন্টায় ডেটা আপডেট করা — এসব কাজ schedule লাইব্রেরি দিয়ে অটোমেট করা যায়।

Python - schedule লাইব্রেরি 
import schedule
import time
import datetime

def daily_backup():
    print(f"📦 ব্যাকআপ শুরু: {datetime.datetime.now()}")
    # আপনার ব্যাকআপ কোড এখানে

def weekly_report():
    print("📊 সাপ্তাহিক রিপোর্ট পাঠানো হচ্ছে...")

def hourly_check():
    print("🔍 প্রতি ঘন্টার চেক...")

# প্রতিদিন রাত ১১টায়
schedule.every().day.at("23:00").do(daily_backup)

# প্রতি সোমবার সকাল ৯টায়
schedule.every().monday.at("09:00").do(weekly_report)

# প্রতি ঘন্টায়
schedule.every().hour.do(hourly_check)

# প্রতি ৩০ মিনিটে
schedule.every(30).minutes.do(hourly_check)

# চালু রাখুন
print("⏰ শিডিউলার চালু হয়েছে...")
while True:
    schedule.run_pending()
    time.sleep(60)

API (Application Programming Interface) হলো দুটি সফটওয়্যারের মধ্যে যোগাযোগের মাধ্যম। Weather, Currency, WhatsApp, YouTube — সব কিছুর API আছে।

Python - Weather API 
import requests

def get_weather(city):
    api_key = "আপনার_API_KEY"  # openweathermap.org থেকে নিন
    url = f"https://api.openweathermap.org/data/2.5/weather"
    
    params = {
        "q": city,
        "appid": api_key,
        "units": "metric",
        "lang": "bn"
    }
    
    response = requests.get(url, params=params)
    data = response.json()
    
    print(f"🌤️ শহর: {data['name']}")
    print(f"🌡️ তাপমাত্রা: {data['main']['temp']}°C")
    print(f"💧 আর্দ্রতা: {data['main']['humidity']}%")
    print(f"🌬️ বায়ু: {data['weather'][0]['description']}")

get_weather("Dhaka")
Python - Telegram Bot 
import requests

class TelegramBot:
    def __init__(self, token):
        self.token = token
        self.base_url = f"https://api.telegram.org/bot{token}"
    
    def send_message(self, chat_id, text):
        url = f"{self.base_url}/sendMessage"
        data = {"chat_id": chat_id, "text": text}
        response = requests.post(url, data=data)
        return response.json()
    
    def send_photo(self, chat_id, photo_path):
        url = f"{self.base_url}/sendPhoto"
        with open(photo_path, "rb") as photo:
            requests.post(url, data={"chat_id": chat_id}, 
                          files={"photo": photo})

# ব্যবহার করুন

Python - জব অ্যালার্ট সিস্টেম 
import requests
from bs4 import BeautifulSoup
import json
import schedule
import time

class JobAlertSystem:
    def __init__(self):
        self.seen_jobs = set()
        self.load_seen_jobs()
    
    def load_seen_jobs(self):
        try:
            with open("seen_jobs.json") as f:
                self.seen_jobs = set(json.load(f))
        except:
            self.seen_jobs = set()
    
    def scrape_jobs(self, keyword="Python"):
        jobs = []
        # চাকরির সাইট স্ক্র্যাপ করুন
        url = f"https://bd.linkedin.com/jobs/search/?keywords={keyword}"
        response = requests.get(url)
        soup = BeautifulSoup(response.text, "html.parser")
        
        for job in soup.select(".job-card-container"):
            title = job.select_one(".job-card-list__title")
            company = job.select_one(".job-card-container__company-name")
            job_id = job.get("data-job-id")
            
            if job_id and job_id not in self.seen_jobs:
                jobs.append({
                    "id": job_id,
                    "title": title.text.strip() if title else "",
                    "company": company.text.strip() if company else ""
                })
                self.seen_jobs.add(job_id)
        
        return jobs
    
    def check_and_alert(self):
        new_jobs = self.scrape_jobs()
        if new_jobs:
            print(f"🆕 {len(new_jobs)}টি নতুন চাকরি পাওয়া গেছে!")
            for job in new_jobs:
                print(f"  • {job['title']} @ {job['company']}")
            # এখানে ইমেইল বা Telegram নোটিফিকেশন পাঠান

# চালান
alert_system = JobAlertSystem()
schedule.every().day.at("09:00").do(alert_system.check_and_alert)
while True:
    schedule.run_pending()
    time.sleep(60)
