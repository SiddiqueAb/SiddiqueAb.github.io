B01
🔴 মেনু ক্লিকে পুরানো/ভুল পেজ খুলছে (Stale Page Bug)
মেনুতে ক্লিক করলে নতুন পেজের বদলে আগের পেজের কন্টেন্ট দেখাচ্ছে। বিশেষত Back বাটন ব্যবহারের পর।
HIGH
+

🔍 কারণ কী?

  • Inertia.js ব্রাউজার ইতিহাস (history state) ক্যাশ করে রাখে
  • Vue-এর keep-alive উপাদান পুরানো কম্পোনেন্ট মেমোরিতে রেখে দেয়
  • Inertia scroll restoration সঠিকভাবে কাজ না করলে এই সমস্যা হয়

✅ সমাধান ১ — app.js/app.ts এ সেটিং করুন:

import { createApp, h } from 'vue'
import { createInertiaApp } from '@inertiajs/vue3'
import { resolvePageComponent } from 'laravel-vite-plugin/inertia-helpers'

createInertiaApp({
  resolve: name =>
    resolvePageComponent(
      `./Pages/${name}.vue`,
      import.meta.glob('./Pages/**/*.vue')
    ),
  setup({ el, App, props, plugin }) {
    return createApp({ render: () => h(App, props) })
      .use(plugin)
      .mount(el)
  },
  // ✅ এই লাইনটি যোগ করুন — scroll position রিসেট হবে
  progress: {
    color: '#00d4ff',
    showSpinner: true,
  },
})

🟠 Back বাটনে পুরানো ডেটা দেখাচ্ছে (Back Navigation Cache)
ব্রাউজারের Back বাটনে গেলে পেজ reload হয় না, পুরানো cached ডেটা দেখায়।
HIGH
+
পেজ কম্পোনেন্টে এটি যোগ করুন 
<script setup>
import { onMounted, onUnmounted } from 'vue'
import { router } from '@inertiajs/vue3'

// ✅ Back navigation detect করে fresh load করুন
onMounted(() => {
  const handler = router.on('navigate', (e) => {
    if (e.detail.page.rememberedState) {
      // Back এ এলে fresh data load করুন
      router.reload({ preserveScroll: true })
    }
  })
  onUnmounted(handler) // cleanup
})
</script>

app/Http/Middleware/HandleInertiaRequests.php 
public function share(Request $request): array
{
    return array_merge(parent::share($request), [
        // ✅ auth user সঠিকভাবে share করুন
        'auth' => [
            'user' => $request->user() ? [
                'id'    => $request->user()->id,
                'name'  => $request->user()->name,
                'email' => $request->user()->email,
                'role'  => $request->user()->role,
            ] : null,
        ],
        // ✅ Flash messages share করুন
        'flash' => [
            'success' => $request->session()->get('success'),
            'error'   => $request->session()->get('error'),
        ],
        // ✅ ziggy routes
        'ziggy' => fn() => [
            ...(new \Tightenco\Ziggy\Ziggy)->toArray(),
            'location' => $request->url(),
        ],
    ]);
}

<script setup>
import { ref, computed, onMounted, onUnmounted } from 'vue'
import { Link, usePage, router } from '@inertiajs/vue3'

const page        = usePage()
const mobileOpen  = ref(false)
const isNavigating = ref(false)

// ✅ Active route checker
const isActive = (routeName) => {
  return route().current(routeName)
}

// ✅ Navigation হলে mobile menu বন্ধ করুন
let removeListener
onMounted(() => {
  removeListener = router.on('start', () => {
    isNavigating.value = true
    mobileOpen.value   = false // ✅ মেনু বন্ধ
  })
  router.on('finish', () => {
    isNavigating.value = false
  })
})
onUnmounted(() => removeListener && removeListener())

const navItems = [
  { label: 'হোম',       route: 'home'       },
  { label: 'পণ্য',       route: 'products.index' },
  { label: 'যোগাযোগ', route: 'contact'    },
]
</script>

<template>

      <ul class="nav-links">

            :href="route(item.route)"

Inertia Global Event Listeners (app.js) 
import { router } from '@inertiajs/vue3'

// ✅ Global navigation events
router.on('start', (event) => {
  console.log('Navigation started', event.detail.visit.url)
  // Loading spinner দেখান
  document.body.classList.add('loading')
})

router.on('finish', (event) => {
  document.body.classList.remove('loading')
  // Scroll to top
  if (!event.detail.visit.preserveScroll) {
    window.scrollTo({ top: 0, behavior: 'smooth' })
  }
})

router.on('error', (event) => {
  console.error('Navigation error', event.detail.errors)
})

// ✅ Partial reload — শুধু posts ডেটা রিফ্রেশ
router.reload({ only: ['posts', 'pagination'] })

// ✅ Visit with custom options
router.visit(route('dashboard'), {
  method: 'get',
  preserveState: false,
  preserveScroll: false,
  replace: false, // history stack এ রাখুন
  only: ['user'],
})
Lazy Loading Props (Controller) 
// ✅ Lazy loading — শুধু request হলে load হবে

use Intervention\Image\ImageManager;
use Intervention\Image\Drivers\Gd\Driver;

public function store(Request $request)
{
    $manager = new ImageManager(new Driver());
    $image   = $manager->read($request->file('image'));

    // Resize + WebP convert
    $image->scale(width: 800)
           ->toWebp(quality: 80) // ✅ WebP, quality 80%
           ->save(storage_path('app/public/images/' . Str::uuid() . '.webp'));

    // thumbnail
    $image->cover(400, 300)
           ->toWebp(quality: 70)
           ->save(storage_path('app/public/thumbnails/' . ...));
}

class HomeController extends Controller
{
    public function index()
    {
        // ✅ Cache — ১ ঘণ্টা cached
        $featured = Cache::remember('home.featured', 3600, fn() =>
            Product::select('id', 'name', 'slug', 'price', 'image')
                ->where('featured', true)
                ->where('status', 'active')
                ->limit(8)
                ->get()
        );

        $categories = Cache::remember('home.categories', 7200, fn() =>
            Category::select('id', 'name', 'slug', 'icon')
                ->withCount('products')
                ->having('products_count', '>', 0)
                ->get()
        );

        // ✅ Latest products — ৩০ মিনিট cache
        $latest = Cache::remember('home.latest', 1800, fn() =>
            Product::select('id', 'name', 'slug', 'price', 'image', 'created_at')
                ->latest()->limit(12)->get()
        );

        return Inertia::render('Home', compact(
            'featured', 'categories', 'latest'
        ));
    }
}

// ✅ Product update হলে cache clear করুন
class Product extends Model
{
    protected static function booted(): void
    {
        static::saved(fn() => Cache::forget('home.featured'));
        static::deleted(fn() => Cache::flush());
    }
}

🟣 Pinia Store — Global State Management

// stores/cartStore.js
import { defineStore } from 'pinia'

export const useCartStore = defineStore('cart', {
  state: () => ({ items: [], total: 0 }),
  getters: {
    count: (state) => state.items.reduce((a, i) => a + i.qty, 0),
  },
  actions: {
    addItem(product) {
      const existing = this.items.find(i => i.id === product.id)
      if (existing) { existing.qty++ }
      else { this.items.push({ ...product, qty: 1 }) }
      this.calcTotal()
    },
    calcTotal() {
      this.total = this.items.reduce((sum, i) => sum + i.price * i.qty, 0)
    }
  }
})
