🗄 SQL — Core Tables Schema 
-- ══════════════════════════════════════════════
   USERS TABLE
══════════════════════════════════════════════
CREATE TABLE users (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  phone         VARCHAR(15) UNIQUE NOT NULL,
  email         VARCHAR(120) UNIQUE,
  name          VARCHAR(100) NOT NULL,
  nid           VARCHAR(20) UNIQUE,
  password_hash VARCHAR(255),
  is_verified   BOOLEAN DEFAULT FALSE,
  status        ENUM('active','suspended','banned') DEFAULT 'active',
  created_at    DATETIME DEFAULT NOW(),
  INDEX idx_phone (phone),
  INDEX idx_email (email)
) ENGINE=InnoDB;

-- ══════════════════════════════════════════════
   TRAINS & SCHEDULES
══════════════════════════════════════════════
CREATE TABLE trains (
  id            SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  train_number  VARCHAR(10) UNIQUE NOT NULL,
  name          VARCHAR(100) NOT NULL,
  type          ENUM('express','local','intercity','mail'),
  total_coaches TINYINT NOT NULL
);

CREATE TABLE stations (
  id    SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  code  CHAR(5) UNIQUE NOT NULL,
  name  VARCHAR(80) NOT NULL,
  district VARCHAR(50)
);

CREATE TABLE schedules (
  id              BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  train_id        SMALLINT UNSIGNED NOT NULL,
  journey_date    DATE NOT NULL,
  from_station_id SMALLINT UNSIGNED NOT NULL,
  to_station_id   SMALLINT UNSIGNED NOT NULL,
  depart_time     TIME NOT NULL,
  arrive_time     TIME NOT NULL,
  status          ENUM('active','cancelled','delayed') DEFAULT 'active',
  UNIQUE KEY uq_train_date_route (train_id, journey_date, from_station_id),
  INDEX idx_journey_route (journey_date, from_station_id, to_station_id),
  FOREIGN KEY (train_id) REFERENCES trains(id)
) ENGINE=InnoDB;

-- ══════════════════════════════════════════════
   SEAT INVENTORY — Heart of the system
══════════════════════════════════════════════
CREATE TABLE seat_inventory (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  schedule_id   BIGINT UNSIGNED NOT NULL,
  coach_number  VARCHAR(5) NOT NULL,
  seat_class    ENUM('S_CHAIR','F_CHAIR','SNIGDHA','AC_B','AC_S','SHOVAN'),
  seat_number   VARCHAR(6) NOT NULL,
  status        ENUM('available','held','booked','unavailable')
                  DEFAULT 'available',
  held_by       BIGINT UNSIGNED NULL,
  held_until    DATETIME NULL,
  version       INT UNSIGNED DEFAULT 0,  -- optimistic concurrency
  UNIQUE KEY uq_seat (schedule_id, seat_class, seat_number),
  INDEX idx_status_schedule (schedule_id, status, seat_class),
  INDEX idx_held_until (held_until),  -- expire cleanup
  FOREIGN KEY (schedule_id) REFERENCES schedules(id)
) ENGINE=InnoDB;

-- ══════════════════════════════════════════════
   BOOKINGS
══════════════════════════════════════════════
CREATE TABLE bookings (
  id               BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  booking_ref      CHAR(10) UNIQUE NOT NULL,
  idempotency_key  VARCHAR(64) UNIQUE NOT NULL,
  user_id          BIGINT UNSIGNED NOT NULL,
  schedule_id      BIGINT UNSIGNED NOT NULL,
  seat_class       VARCHAR(20),
  passenger_count  TINYINT NOT NULL DEFAULT 1,
  total_fare       DECIMAL(10,2) NOT NULL,
  status           ENUM('pending','confirmed','cancelled','refunded')
                     DEFAULT 'pending',
  created_at       DATETIME DEFAULT NOW(),
  confirmed_at     DATETIME NULL,
  expires_at       DATETIME NOT NULL,  -- payment deadline
  INDEX idx_user_status (user_id, status),
  INDEX idx_expires (expires_at, status),
  FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB;

-- Booking seats (which seats belong to which booking)
CREATE TABLE booking_seats (
  id           BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  booking_id   BIGINT UNSIGNED NOT NULL,
  seat_inv_id  BIGINT UNSIGNED NOT NULL,
  passenger_id BIGINT UNSIGNED NOT NULL,
  fare         DECIMAL(10,2) NOT NULL,
  UNIQUE(seat_inv_id),  -- একটি seat একটি bookingেই
  FOREIGN KEY (booking_id) REFERENCES bookings(id)
) ENGINE=InnoDB;

-- ══════════════════════════════════════════════

CREATE TABLE payments (

CREATE TABLE otp_logs (
  id         BIGINT AUTO_INCREMENT PRIMARY KEY,
  phone      VARCHAR(15) NOT NULL,
  purpose    ENUM('register','login','reset'),
  attempts   TINYINT DEFAULT 0,

  used       BOOLEAN DEFAULT FALSE,

-- Rate limiting: ১ ঘণ্টায় max 5 OTP per phone
SELECT COUNT(*) FROM otp_logs
WHERE phone = ?
  AND sent_at > DATE_SUB(NOW(), INTERVAL 1 HOUR);
-- ≥ 5 হলে block করো

-- Refresh token storage (Redis-preferred, but DB fallback)
CREATE TABLE refresh_tokens (
  id         BIGINT AUTO_INCREMENT PRIMARY KEY,
  user_id    BIGINT UNSIGNED NOT NULL,
  token_hash CHAR(64) UNIQUE NOT NULL,  -- SHA-256 of token
  device_id  VARCHAR(100),
  expires_at DATETIME NOT NULL,
  revoked    BOOLEAN DEFAULT FALSE,
  INDEX(token_hash), INDEX(user_id)
);
🔑 JWT Strategy

Access token: 15 মিনিট expiry, stateless JWT। Refresh token: 30 দিন, Redis-এ store করো। Token revocation-এর জন্য Redis-এ blocklist:jti key রাখো। Logout-এ JTI blocklist-এ যোগ করো।

05 🔍

Train Search & Schedule

Search Architecture — Cache-first Strategy

Train search সবচেয়ে বেশি হিট হওয়া endpoint। ঈদের আগে লক্ষ লক্ষ মানুষ একই route search করছে। প্রতিটি search-এ DB hit করা catastrophic — Cache-first strategy অপরিহার্য।

Search Cache Strategy
Search Request: "Dhaka → Chittagong, 2025-06-05" │ ▼ Redis Cache: GET search:dhk:cgp:20250605 │ ┌────┴─────────────────────────────────┐ │ HIT ✓ │ MISS ✗ ▼ ▼ Return cached JSON Query MySQL schedules (p50 < 2ms) JOIN trains, stations WHERE journey_date=? AND from=? AND to=? │ ▼ Cache result in Redis SET EX 300 (5 min TTL) │ ▼ Return to client
SQL — Optimized Train Search Query 
-- Train search with available seat count
SELECT
  sch.id             AS schedule_id,
  t.train_number,
  t.name            AS train_name,
  t.type            AS train_type,
  fs.name           AS from_station,
  ts.name           AS to_station,
  sch.depart_time,
  sch.arrive_time,
  sch.status,
  -- Seat counts per class (aggregated)
  SUM(CASE WHEN si.seat_class='SHOVAN'  AND si.status='available' THEN 1 ELSE 0 END) shovan_avail,
  SUM(CASE WHEN si.seat_class='SNIGDHA' AND si.status='available' THEN 1 ELSE 0 END) snigdha_avail,
  SUM(CASE WHEN si.seat_class='AC_B'    AND si.status='available' THEN 1 ELSE 0 END) ac_b_avail,
  MIN(fare.base_fare)                                                              AS min_fare
FROM schedules sch
JOIN trains t         ON t.id = sch.train_id
JOIN stations fs      ON fs.id = sch.from_station_id
JOIN stations ts      ON ts.id = sch.to_station_id
LEFT JOIN seat_inventory si ON si.schedule_id = sch.id
LEFT JOIN fares fare  ON fare.schedule_id = sch.id
WHERE
  sch.journey_date    = :journey_date
  AND sch.from_station_id = :from_id
  AND sch.to_station_id   = :to_id
  AND sch.status          = 'active'
GROUP BY sch.id
ORDER BY sch.depart_time ASC;
-- EXPLAIN: uses idx_journey_route index ✅
⚡ Performance Tip

Search result cache-এর TTL 5 মিনিট রাখো। কিন্তু যখন কোনো seat book হয়, cache invalidate করো। Redis Pub/Sub দিয়ে booking service → search service-কে invalidate event পাঠাও।

06 💺

Seat Availability & Real-time Locking

Two-Layer Availability Check

Seat availability দুই layer-এ check হয়: Layer 1 → Redis (fast, approximate), Layer 2 → MySQL (authoritative, with lock)। এই pattern লক্ষ লক্ষ read request DB-তে না পাঠিয়ে Redis থেকে serve করে।

Seat Availability — Redis + DB Dual Layer
Layer 1: Redis (Real-time Seat Count Cache) HSET seat_count:schedule:1001 SHOVAN 120 SNIGDHA 40 AC_B 24 → Seat map display, listing page → 99% traffic এখানেই শেষ Layer 2: MySQL (Authoritative — only on booking) SELECT id FROM seat_inventory WHERE schedule_id=1001 AND seat_class='SHOVAN' AND status='available' LIMIT 2 FOR UPDATE; → Booking-এর সময়ই শুধু DB hit → guaranteed consistency Sync Strategy: ┌─────────────────────────────────────────┐ │ Booking confirmed → HINCRBY -1 in Redis │ │ Booking cancelled → HINCRBY +1 in Redis │ │ Every 60s → Reconcile Redis vs DB count │ └─────────────────────────────────────────┘
SQL — Pessimistic Seat Lock (Booking Critical Path) 
-- ══════════════════════════════════════════
   STEP 1: Lock requested seats atomically
══════════════════════════════════════════
START TRANSACTION;

-- Specific seats চাইলে (user seat map থেকে বেছেছে)
SELECT id, seat_number, status, version
FROM seat_inventory
WHERE
  schedule_id = :schedule_id
  AND id IN (:seat_id_1, :seat_id_2)
  AND status = 'available'
ORDER BY id ASC  -- Deadlock prevent: consistent ordering
FOR UPDATE;       -- Exclusive row lock

-- ROW_COUNT() ≠ requested count → seats no longer available
-- IF found_count < requested_count THEN ROLLBACK; END IF;

-- Any-available seat চাইলে (system assigns)
SELECT id, seat_number
FROM seat_inventory
WHERE
  schedule_id = :schedule_id
  AND seat_class = :seat_class
  AND status = 'available'
LIMIT :count
FOR UPDATE SKIP LOCKED;  -- Other workers skip locked seats

-- ══════════════════════════════════════════
   STEP 2: Hold the seats (temporary, 10 min)
══════════════════════════════════════════
UPDATE seat_inventory
  SET
    status     = 'held',
    held_by    = :user_id,
    held_until = DATE_ADD(NOW(), INTERVAL 10 MINUTE),
    version    = version + 1
  WHERE id IN (:locked_seat_ids);

-- ══════════════════════════════════════════
   STEP 3: Create pending booking
══════════════════════════════════════════
INSERT INTO bookings (
  booking_ref, idempotency_key, user_id,
  schedule_id, seat_class, passenger_count,
  total_fare, status, expires_at
) VALUES (
  UUID_SHORT(), :idem_key, :user_id,
  :schedule_id, :seat_class, :count,
  :total_fare, 'pending', DATE_ADD(NOW(), INTERVAL 10 MINUTE)
);

COMMIT;

-- ══════════════════════════════════════════
   Expire held seats (scheduled job, every 1 min)
══════════════════════════════════════════
UPDATE seat_inventory
  SET
    status     = 'available',
    held_by    = NULL,
    held_until = NULL
  WHERE
    status     = 'held'
    AND held_until < NOW();
-- Index idx_held_until ensures this is fast

Optimistic Locking Alternative

Low-contention scenarios-এ (off-peak booking) optimistic locking DB pressure কমায়। Version number দিয়ে conflict detect করো, lock ছাড়াই।

SQL — Optimistic Concurrency Control 
-- ১. Version সহ read (no lock)
SELECT id, status, version
FROM seat_inventory
WHERE id = :seat_id;
-- Returns: version=5

-- ২. Update with version check
UPDATE seat_inventory
  SET
    status  = 'held',
    held_by = :user_id,
    version = version + 1
  WHERE
    id      = :seat_id
    AND version = 5     -- stale check
    AND status  = 'available';

-- ROW_COUNT() = 0? Conflict → retry with fresh data
-- ROW_COUNT() = 1? Success → proceed
✅ কখন কোনটা ব্যবহার করবো?

Pessimistic (FOR UPDATE): ঈদের peak booking — high contention নিশ্চিত।
Optimistic (version): Admin seat management, off-peak hours — contention rare।

High-Concurrency Booking Flow & Queue System

Queue-based Booking Architecture

Peak time-এ প্রতি সেকেন্ডে হাজার হাজার booking request আসলে সরাসরি DB-তে পাঠানো সিস্টেম crush করবে। Message Queue দিয়ে requests buffer করো — worker pool rate control করে process করে।

Queue-based Booking Flow
Client Request POST /bookings {schedule_id, seats, passengers, idempotency_key} │ ▼ API Gateway → Booking Service ① Validate input ② Check idempotency_key (Redis: GET idem:{key}) → Already exists? Return cached response ✅ ③ Quick Redis availability check → HGET seat_count:schedule:{id} {class} → Count ≤ 0? Immediate reject 🚫 ④ Publish to RabbitMQ booking.queue → Return 202 Accepted + booking_token │ ▼ (async) RabbitMQ: booking.requests queue [msg1] [msg2] [msg3] ... [msg50000] TTL: 10 minutes (booking window) │ ▼ Booking Worker Pool (N workers) ① Dequeue message ② START TRANSACTION ③ SELECT seats FOR UPDATE SKIP LOCKED ④ Hold seats + create booking record ⑤ COMMIT ⑥ Publish to payment.queue ⑦ Update Redis: booking_status:{token} = "SEAT_HELD" │ ▼ Client polls: GET /bookings/status/{token} → Redis: booking_status:{token} → Returns: QUEUED / SEAT_HELD / CONFIRMED / FAILED
JavaScript — Booking Service (Node.js) 
// Booking Service — POST /bookings
async function createBooking(req, res) {
  const { schedule_id, seat_ids, passengers, idempotency_key } = req.body;

  // ① Idempotency check
  const cached = await redis.get(`idem:${idempotency_key}`);
  if (cached) return res.json(JSON.parse(cached));

  // ② Quick availability check (Redis)
  const available = await redis.hget(`seat_count:schedule:${schedule_id}`, req.body.seat_class);
  if (parseInt(available) < passengers.length) {
    return res.status(409).json({ error: 'No seats available' });
  }

  // ③ Generate booking token
  const bookingToken = crypto.randomUUID();

  // ④ Set initial status
  await redis.setex(`booking_status:${bookingToken}`, 600, 'QUEUED');

  // ⑤ Publish to queue
  await rabbitMQ.publish('booking.requests', {
    token: bookingToken,
    schedule_id,
    seat_ids,
    passengers,
    user_id: req.user.id,
    idempotency_key,
    timestamp: Date.now()
  }, { expiration: '600000' }); // 10 min TTL

  // ⑥ Return 202 immediately
  const response = { status: 'QUEUED', token: bookingToken, poll_url: `/bookings/status/${bookingToken}` };
  await redis.setex(`idem:${idempotency_key}`, 3600, JSON.stringify(response));
  return res.status(202).json(response);
}

// Booking Worker
async function processBooking(msg) {
  const { token, schedule_id, seat_ids, passengers, user_id, idempotency_key } = msg;

  const conn = await db.getConnection();
  try {
    await conn.beginTransaction();

    // Lock seats
    const [seats] = await conn.query(
      `SELECT id, seat_number FROM seat_inventory
       WHERE id IN (?) AND status = 'available' ORDER BY id ASC FOR UPDATE`,
      [seat_ids]
    );

    if (seats.length < passengers.length) {
      await conn.rollback();
      await redis.setex(`booking_status:${token}`, 600, JSON.stringify({ status: 'FAILED', reason: 'seats_taken' }));
      return;
    }

    // Hold seats
    await conn.query(
      `UPDATE seat_inventory SET status='held', held_by=?, held_until=DATE_ADD(NOW(), INTERVAL 10 MINUTE)
       WHERE id IN (?)`,
      [user_id, seat_ids]
    );

    // Create booking
    const [result] = await conn.query(
      `INSERT INTO bookings (booking_ref, idempotency_key, user_id, schedule_id, total_fare, status, expires_at)
       VALUES (?, ?, ?, ?, ?, 'pending', DATE_ADD(NOW(), INTERVAL 10 MINUTE))`,
      [token.slice(0,10).toUpperCase(), idempotency_key, user_id, schedule_id, calculateFare(seats)]
    );

    await conn.commit();
    await redis.setex(`booking_status:${token}`, 600, JSON.stringify({ status: 'SEAT_HELD', booking_id: result.insertId }));
    await rabbitMQ.publish('payment.initiate', { booking_id: result.insertId, token });

  } catch (err) {
    await conn.rollback();
    await redis.setex(`booking_status:${token}`, 600, JSON.stringify({ status: 'FAILED', reason: err.message }));
  } finally {
    conn.release();
  }
}

JavaScript — Sliding Window Rate Limiter (Redis) 
// Sliding window rate limiter
async function rateLimiter(userId, action, limit, windowSec) {
  const key = `rate:${action}:${userId}`;
  const now = Date.now();
  const windowStart = now - (windowSec * 1000);

  const pipeline = redis.pipeline();
  pipeline.zremrangebyscore(key, 0, windowStart);   // old entries remove
  pipeline.zadd(key, now, `${now}-${Math.random()}`); // add current
  pipeline.zcard(key);                              // count
  pipeline.expire(key, windowSec);                  // cleanup TTL

  const results = await pipeline.exec();
  const count = results[2][1];

  if (count > limit) {
    throw new Error(`Rate limit exceeded. Max ${limit} requests per ${windowSec}s`);
  }
  return { remaining: limit - count };
}

// Usage:
await rateLimiter(userId, 'booking', 5, 60);    // 5 bookings/min
await rateLimiter(ip, 'search', 60, 60);         // 60 searches/min
await rateLimiter(phone, 'otp', 5, 3600);        // 5 OTPs/hour

Disaster Recovery Plan

🔥 DB Primary Failure — Failover Steps

১. ProxySQL/HAProxy auto-detects primary down (health check every 5s).
২. Promote Read Replica 1 to Primary (semi-sync replication ensures max 1 tx loss).
③. Update DNS/config; new writes go to promoted replica.
৪. Alert oncall; RTO target: < 30 seconds. RPO: < 5 seconds.
৫. Old primary recovered → rejoin as new replica.

✅ Eid Peak Day Checklist

D-7: Load test with 2x expected traffic. Fix bottlenecks.
D-3: Pre-warm Redis cache with schedule data. Scale up pods.
D-1: Enable read replicas. Increase connection pool. Alert thresholds set.
Day-0: War room monitoring. Extra workers on standby. Rollback plan ready.
Post: Analyze slow queries. Redis hit rate. Queue depth metrics.

  • Transaction-এ HTTP call করা: Payment gateway API কল করার সময় DB transaction open রাখা — gateway 5s timeout নিলে lock 5 সেকেন্ড held। Gateway call transaction-এর বাইরে করো।
  • Cache invalidation race: DB update → Redis delete → অন্য request stale data পড়ে cache করে। Cache-aside pattern-এ write-through use করো বা cache TTL ছোট রাখো।
  • No circuit breaker: bKash gateway down হলে সব payment request timeout পর্যন্ত ঝুলে থাকে। Circuit breaker (e.g. Hystrix/Resilience4j) দিয়ে fast-fail করো।
  • Seat count শুধু Redis-এ: Redis restart হলে সব count শূন্য। সবসময় DB কে source of truth রাখো, Redis শুধু cache।
