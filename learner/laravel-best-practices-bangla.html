🧹 পরিষ্কার কোড (Clean Code) — মূল নীতিমালা

✅ পরিষ্কার কোডের বৈশিষ্ট্য

  • ✦ সহজ ও স্পষ্ট নামকরণ

  • ✦ কম জটিলতা, বেশি পঠনযোগ্যতা
  • ✦ DRY (Don't Repeat Yourself)

  • ✦ উপযুক্ত মন্তব্য/ডকুমেন্টেশন

❌ যা এড়িয়ে চলতে হবে

  • ✦ দীর্ঘ ফাংশন (God Functions)
  • ✦ কপি-পেস্ট কোড
  • ✦ অর্থহীন ভেরিয়েবল নাম (x, y, temp)

  $age, $income

    : $income * MINOR_TAX_RATE;
}

একটি ক্লাস শুধুমাত্র একটি কাজের জন্য দায়িত্বশীল হবে।

O O — Open/Closed Principle (খোলা/বন্ধ নীতি)

কোড সম্প্রসারণের জন্য খোলা, কিন্তু পরিবর্তনের জন্য বন্ধ।

PHPPayment Processor
// Interface তৈরি করুন
interface PaymentGatewayInterface {
  public function pay(float $amount): bool;
}

// নতুন payment method যোগ করতে পুরনো কোড বদলাতে হবে না
class BkashPayment implements PaymentGatewayInterface {
  public function pay(float $amount): bool {
    return BkashAPI::charge($amount);
  }
}

class NagadPayment implements PaymentGatewayInterface {
  public function pay(float $amount): bool {
    return NagadAPI::charge($amount);
  }
}

class OrderController {
  public function __construct(
    private PaymentGatewayInterface $payment
  ) {}

  public function checkout(float $amount) {
    $this->payment->pay($amount);
  }
}

📦 API Controller — সঠিক গঠন

PHPProductController.php
namespace App\Http\Controllers\Api\V1;

class ProductController extends Controller {
  public function __construct(
    private readonly ProductService $productService
  ) {}

  // তালিকা দেখানো
  public function index(IndexProductRequest $request): JsonResponse {
    $products = $this->productService->paginate(
      $request->validated()
    );

    return ProductResource::collection($products)
      ->response()->setStatusCode(200);
  }

  // নতুন পণ্য তৈরি
  public function store(StoreProductRequest $request): JsonResponse {
    $product = $this->productService->create(
      $request->validated()
    );

    return (new ProductResource($product))
      ->response()
      ->setStatusCode(201);
  }
}

🎨 API Resource — Response গঠন

PHPProductResource.php
class ProductResource extends JsonResource {
  public function toArray(Request $request): array {
    return [
      'id'         => $this->id,
      'name'       => $this->name,
      'price'      => number_format($this->price, 2),
      'category'   => new CategoryResource($this->category),
      'created_at' => $this->created_at->toDateTimeString(),
      'links'      => [
        'self' => route('api.products.show', $this->id),
      ],
    ];
  }
}

🛡️ Form Request Validation

PHPStoreProductRequest.php
class StoreProductRequest extends FormRequest {
  public function authorize(): bool {
    return $this->user()->can('create', Product::class);
  }

  public function rules(): array {
    return [
      'name'     => ['required', 'string', 'max:255'],
      'price'    => ['required', 'numeric', 'min:0'],
      'category_id' => ['required', 'exists:categories,id'],
      'stock'    => ['required', 'integer', 'min:0'],
    ];
  }

  public function messages(): array {
    return [
      'name.required' => 'পণ্যের নাম আবশ্যক',
      'price.min'     => 'মূল্য ঋণাত্মক হতে পারবে না',
    ];
  }
}

🎯 Service Layer — Business Logic আলাদা করুন

PHPProductService.php
class ProductService {
  public function __construct(
    private readonly ProductRepository $repository
  ) {}

  public function create(array $data): Product {
    // ব্যবসায়িক নিয়ম এখানে থাকবে
    $data['slug'] = Str::slug($data['name']);
    $data['sku'] = $this->generateSku($data['name']);

    $product = $this->repository->create($data);

    // Event fire করুন
    event(new ProductCreated($product));

    return $product;
  }

  private function generateSku(string $name): string {
    return strtoupper(substr($name, 0, 3))
      . time();
  }
}

use Inertia\Inertia;

class UserController extends Controller {
  public function index(): Response {
    return Inertia::render('Users/Index', [
      // Lazy loading — প্রয়োজনে লোড হবে
      'users' => Inertia::lazy(
        fn() => UserResource::collection(
          User::with('roles')->paginate(15)
        )
      ),
      'filters' => request()->only(['search', 'role']),
    ]);
  }

  public function store(StoreUserRequest $request) {
    User::create($request->validated());

    return redirect()->route('users.index')
      ->with('success', 'ব্যবহারকারী তৈরি হয়েছে!');
  }
}

🔗 Shared Data — Global Props

PHPHandleInertiaRequests.php
class HandleInertiaRequests extends Middleware {
  public function share(Request $request): array {
    return array_merge(parent::share($request), [
      // সব page-এ auth user পাওয়া যাবে
      'auth' => [
        'user' => $request->user() ? [
          'id'    => $request->user()->id,
          'name'  => $request->user()->name,
          'email' => $request->user()->email,
        ] : null,
      ],
      // Flash message সব page-এ পাবে
      'flash' => [
        'success' => session('success'),
        'error'   => session('error'),
      ],
    ]);
  }
}

🧩 Composable — পুনর্ব্যবহারযোগ্য লজিক

JSresources/js/composables/useProducts.js
import { ref, computed, onMounted } from 'vue'
import { useForm } from '@inertiajs/vue3'

// পণ্য সংক্রান্ত সব লজিক এখানে থাকবে
export function useProducts() {
  const products = ref([])
  const isLoading = ref(false)
  const error = ref(null)

  const totalProducts = computed(
    () => products.value.length
  )

  async function fetchProducts() {
    isLoading.value = true
    try {
      const response = await fetch('/api/v1/products')
      products.value = await response.json()
    } catch (err) {
      error.value = 'পণ্য লোড করতে সমস্যা হয়েছে'
    } finally {
      isLoading.value = false
    }
  }

  onMounted(fetchProducts)

  return { products, isLoading, error, totalProducts }
}

// ২. পড়া (না থাকলে null)
$count = Cache::get('মোট_পণ্য');

// ৩. remember — না থাকলে DB থেকে আনো এবং সংরক্ষণ করো
$products = Cache::remember(
  'সক্রিয়_পণ্য',
  now()->addHour(),
  fn() => Product::where('active', true)->get()
);

// ৪. rememberForever — মেয়াদ নেই
$settings = Cache::rememberForever(
  'সিস্টেম_সেটিংস',
  fn() => Setting::all()->pluck('value', 'key')
);

// ৫. মুছে ফেলা
Cache::forget('সক্রিয়_পণ্য');

// ৬. আছে কিনা যাচাই
if (Cache::has('মোট_পণ্য')) {
  // cache আছে
}

// ৭. Tag দিয়ে গ্রুপ করা (Redis/Memcached)
Cache::tags(['পণ্য', 'ক্যাটাগরি'])
  ->put('featured', $products, 3600);

// ৮. পুরো tag flush করা
Cache::tags('পণ্য')->flush();

// ৯. পুরো cache clear
Cache::flush();
// অথবা Artisan দিয়ে:
// php artisan cache:clear

🔄 Cache Invalidation — Model Observer দিয়ে

PHPProductObserver.php
class ProductObserver {
  // পণ্য তৈরি হলে cache মুছো
  public function created(Product $product): void {
    $this->clearProductCache();
  }

  // পণ্য আপডেট হলে cache মুছো
  public function updated(Product $product): void {
    Cache::forget("পণ্য_" . $product->id);
    $this->clearProductCache();
  }

  private function clearProductCache(): void {
    Cache::forget('পণ্য_তালিকা');
    Cache::tags('পণ্য')->flush();
  }
}

// AppServiceProvider.php তে register করুন

🚀 Bootstrap Cache — Config, Route, View Cache

📌 Bootstrap Cache কি? Laravel প্রতিটি request-এ configuration, routes, services লোড করে। Bootstrap Cache এই প্রক্রিয়াকে একবার করে ফাইলে সংরক্ষণ করে, পরবর্তী request-এ দ্রুত লোড হয়।

⚙️ Config Cache

BASHTerminal
# সব config ফাইল একটি ফাইলে combine করে
php artisan config:cache

# ফলাফল: bootstrap/cache/config.php তৈরি হয়
# এতে config/ ফোল্ডারের সব ফাইল merge থাকে

# Config cache মুছতে
php artisan config:clear

# Cache আছে কিনা দেখতে
ls bootstrap/cache/

কিভাবে কাজ করে?

config:cache → সব config ফাইল → একটি PHP array → bootstrap/cache/config.php
পরের request → এই একটি ফাইল লোড → অনেক দ্রুত!

🔐 Sanctum API Authentication

PHPAuthController.php
class AuthController extends Controller {
  public function login(LoginRequest $request): JsonResponse {
    if (! Auth::attempt($request->only('email', 'password'))) {
      return response()->json([
        'message' => 'ইমেইল বা পাসওয়ার্ড ভুল',
      ], 401);
    }

    $user = $request->user();

    // পুরনো token মুছো (Single device login)
    $user->tokens()->delete();

    // নতুন token তৈরি করো
    $token = $user->createToken(
      'auth_token',
      ['*'], // abilities
      now()->addDays(30) // expiry
    );

    return response()->json([
      'user'  => new UserResource($user),
      'token' => $token->plainTextToken,
    ]);
  }

  public function logout(Request $request): JsonResponse {
    $request->user()->currentAccessToken()->delete();
    return response()->json([
      'message' => 'সফলভাবে লগআউট হয়েছেন'
    ]);
  }
}

⚡ Rate Limiting

PHPAppServiceProvider.php
use Illuminate\Cache\RateLimiting\Limit;
use Illuminate\Support\Facades\RateLimiter;

// API Rate Limiting
RateLimiter::for('api', function(Request $request) {
  return Limit::perMinute(60)
    ->by($request->user()?->id ?: $request->ip());
});

// Login এর জন্য আলাদা limit
RateLimiter::for('login', function(Request $request) {
  return [
    Limit::perMinute(5)->by($request->ip()),
    // প্রতি email এ 5 বার
    Limit::perMinute(5)
      ->by($request->input('email')),
  ];
});
