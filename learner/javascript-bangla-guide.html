Optional Chaining (?.) ও Nullish Coalescing (??)
undefined/null error থেকে মুক্তি
ES2020প্রতিদিন লাগে
optional-chaining.js
const user = {
  name: "তানিয়া",
  address: {
    city: "ঢাকা"
  }
};

// ❌ পুরনো — crash করবে যদি address না থাকে
const zip1 = user.address.zip; // undefined (ঠিক আছে এক্ষেত্রে)
const ph1  = user.phone.home;  // 💥 TypeError: Cannot read 'home'

// ✅ Optional Chaining (?.) — crash করবে না, undefined দেবে
const ph2  = user?.phone?.home;      // undefined ✅
const city = user?.address?.city;   // "ঢাকা"

// ✅ Nullish Coalescing (??) — null/undefined হলে default দেও
const phone = user?.phone ?? "নম্বর নেই";  // "নম্বর নেই"
const name  = user?.name  ?? "অজ্ঞাত";      // "তানিয়া"

// ?? vs || পার্থক্য (গুরুত্বপূর্ণ!)
const count = 0;
console.log(count || "default");  // "default" ← 0 falsy তাই!
console.log(count ?? "default");  // 0 ← শুধু null/undefined-এ default দেয়
✅?? শুধু null ও undefined-এর ক্ষেত্রে default দেয়। কিন্তু || সব falsy মান (0, "", false)-এর ক্ষেত্রে default দেয়। টাকার পরিমাণ বা সংখ্যা নিয়ে কাজ করলে সবসময় ?? ব্যবহার করুন।

Arrow Function ( => )
ES6-এর সবচেয়ে জনপ্রিয় সংযোজন
ES6প্রতিদিন ব্যবহার

Arrow function হলো function লেখার সংক্ষিপ্ত রূপ। কিন্তু শুধু ছোট নয় — this binding-ও আলাদা।

arrow-functions.js
// ১. সাধারণ function
function add(a, b) {
  return a + b;
}

// ২. Arrow function (একই কাজ)
const add = (a, b) => {
  return a + b;
};

// ৩. Single expression — return ও {} বাদ দেওয়া যায়
const add = (a, b) => a + b;

// ৪. একটাই parameter — () বাদ দেওয়া যায়
const double = n => n * 2;

// ৫. Parameter নেই
const greet = () => "আস-সালামু আলাইকুম!";

// ৬. Object return করতে () দিতে হবে
const getUser = (name) => ({ name, role: "admin" });

// Default Parameter — ES6
const greetUser = (name = "বন্ধু") => `স্বাগতম, ${name}!`;
greetUser();           // "স্বাগতম, বন্ধু!"
greetUser("মিম");    // "স্বাগতম, মিম!"
⚠️Arrow function-এর নিজের this নেই! Parent scope-এর this inherit করে। তাই Object method ও Event Listener-এ সাবধানে ব্যবহার করুন।

closure.js
// Higher Order Function — function নিতে বা return করতে পারে
function applyDiscount(discountFn) {
  return (price) => discountFn(price);
}
const tenPercent = price => price * 0.9;
const applyTen   = applyDiscount(tenPercent);
console.log(applyTen(1000)); // 900

// Closure — inner function বাইরের variable মনে রাখে
function counter() {
  let count = 0;         // এই variable বাইরে দেখা যায় না
  return {
    increment: () => ++count,
    decrement: () => --count,
    value:     () => count,
  };
}
const c = counter();
c.increment(); c.increment();
console.log(c.value()); // 2

// ━━ ARRAY DESTRUCTURING ━━
const colors = ["লাল", "সবুজ", "নীল"];

// ❌ পুরনো পদ্ধতি
const first  = colors[0];
const second = colors[1];

// ✅ Destructuring
const [first, second, third] = colors;

// Skip করা যায়
const [, , blue] = colors;  // শুধু "নীল"

// Default value
const [a, b, c, d = "বেগুনি"] = colors;

// Variable swap (সহজে!)
let x = 1, y = 2;
[x, y] = [y, x]; // x=2, y=1

// ━━ OBJECT DESTRUCTURING ━━
const student = {
  name: "নাফিসা", age: 20,
  dept: "CSE", gpa: 3.8
};

// ✅ Object Destructuring
const { name, age, dept } = student;

// Alias (নতুন নামে নেওয়া)
const { name: studentName, gpa: cgpa } = student;

// Default value
const { name: n, roll = "N/A" } = student;

// Nested Destructuring
const { address: { city, zip } } = {
  address: { city: "ঢাকা", zip: "1207" }
};

// Function parameter-এ destructuring (React-এ props এভাবেই নেওয়া হয়)
const UserCard = ({ name, age, dept = "Unknown" }) => {
  console.log(`${name}, ${age}, ${dept}`);
};

Promise — ভবিষ্যতের মান
Callback Hell থেকে মুক্তি দিয়েছে Promise
ES6API call-এ অবশ্যই

JavaScript single-threaded হওয়ায় server থেকে data আনা, ফাইল পড়া — এসব কাজ Async। Promise হলো "আমি কাজটা করব, শেষ হলে result দেব" — এই চুক্তি।

✅ pending → fulfilled
কাজ সফল হলে .then() চলে। result পাওয়া যায়।
❌ pending → rejected
কাজ ব্যর্থ হলে .catch() চলে। error পাওয়া যায়।
async-await.js
// ━━ Promise Chain ━━
fetch("https://api.example.com/users")
  .then(response => response.json())
  .then(data => console.log(data))
  .catch(err => console.error("Error:", err))
  .finally(() => console.log("শেষ হয়েছে"));

// ━━ Async/Await — অনেক সহজ (সবচেয়ে আধুনিক পদ্ধতি) ━━
const fetchUsers = async () => {
  try {
    const response = await fetch("https://api.example.com/users");

    if (!response.ok) {
      throw new Error(`HTTP error: ${response.status}`);
    }

    const data = await response.json();
    console.log(data);
    return data;

  } catch (error) {
    console.error("কিছু একটা ভুল হয়েছে:", error.message);
  }
};

// ━━ Parallel — একসাথে অনেক request ━━
const loadAll = async () => {
  const [users, posts, comments] = await Promise.all([
    fetch("/api/users").then(r => r.json()),
    fetch("/api/posts").then(r => r.json()),
    fetch("/api/comments").then(r => r.json()),
  ]);
  // তিনটা একসাথে চলে — দ্রুত!
};
💡Promise.all() সব শেষ হওয়া পর্যন্ত অপেক্ষা করে। একটাও fail করলে সবই fail। আলাদা আলাদা result চাইলে Promise.allSettled() ব্যবহার করুন।

map, filter, reduce — Functional Array Methods
Original array পরিবর্তন না করে নতুন array তৈরি
সবচেয়ে বেশি ব্যবহারReact JSX-এ অবশ্যই
array-methods.js
const products = [
  { name: "শার্ট",   price: 500,  inStock: true  },
  { name: "প্যান্ট", price: 1200, inStock: false },
  { name: "জুতা",   price: 2500, inStock: true  },
  { name: "ব্যাগ",   price: 800,  inStock: true  },
];

// ━━ map — প্রতিটি element রূপান্তর করে নতুন array ━━
const names  = products.map(p => p.name);
// ["শার্ট", "প্যান্ট", "জুতা", "ব্যাগ"]

const withVat = products.map(p => ({
  ...p, price: p.price * 1.15
})); // ভ্যাটসহ নতুন array

// ━━ filter — শর্ত মেনে চলে এমন element রাখে ━━
const available = products.filter(p => p.inStock);
const affordable = products.filter(p => p.price < 1000);

// ━━ reduce — সব element একটা মানে নিয়ে আসে ━━
const totalPrice = products.reduce(
  (sum, p) => sum + p.price, 0
); // 5000

// ━━ find — প্রথম match করা element ━━
const shoe = products.find(p => p.name === "জুতা");

// ━━ some / every ━━
const anyAvailable  = products.some(p => p.inStock);   // true
const allAvailable  = products.every(p => p.inStock);  // false

// ━━ Chaining — একসাথে অনেক method ━━
const result = products
  .filter(p => p.inStock)           // available পণ্য
  .map(p => ({ ...p, price: p.price * 0.9 }))  // 10% ছাড়
  .sort((a, b) => a.price - b.price);  // দাম অনুযায়ী sort

// ━━ CLASS ━━
class Animal {
  // Private field (ES2022) — বাইরে access নেই
  #name;

  constructor(name, sound) {
    this.#name = name;
    this.sound = sound;
  }

  speak() {
    return `${this.#name} বলে: ${this.sound}!`;
  }

  get name() { return this.#name; }

  static create(name, sound) {
    return new Animal(name, sound);
  }
}

// Inheritance
class Dog extends Animal {
  constructor(name) {
    super(name, "ঘেউ ঘেউ"); // Parent constructor call
    this.type = "dog";
  }

  fetch() { return `${this.name} বল এনে দিল!`; }
}

// ━━ 1. DEBOUNCE — Search input-এ অপরিহার্য ━━
const debounce = (fn, delay) => {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
};

const searchInput = document.querySelector("#search");
searchInput.addEventListener("input", debounce((e) => {
  fetchSearchResults(e.target.value); // ৩০০ms পরে call হবে
}, 300));

// ━━ 2. LOCAL STORAGE — Data সংরক্ষণ ━━

  set: (key, val) => localStorage.setItem(key, JSON.stringify(val)),
  get: (key) => {
    const item = localStorage.getItem(key);
    return item ? JSON.parse(item) : null;
  },
  remove: (key) => localStorage.removeItem(key),
};
