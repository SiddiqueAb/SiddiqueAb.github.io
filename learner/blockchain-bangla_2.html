01 🧱

Blockchain কী?

Blockchain হলো একটি বিকেন্দ্রীভূত (Decentralized) ডিজিটাল লেজার যেখানে তথ্য একের পর এক "ব্লক"-এ সংরক্ষিত থাকে এবং এই ব্লকগুলো একটি চেইনের মতো যুক্ত থাকে। একবার তথ্য যোগ হলে তা পরিবর্তন করা প্রায় অসম্ভব।

🏛️ কেন্দ্রীয় ব্যাংক (Traditional)

একটি কেন্দ্রীয় কর্তৃপক্ষ সব তথ্য নিয়ন্ত্রণ করে। সে ভুল করলে বা দুর্নীতি করলে সব তথ্য ঝুঁকিতে পড়ে।

⛓️ Blockchain (Decentralized)

হাজার হাজার কম্পিউটারে একই তথ্য থাকে। কেউ একা পরিবর্তন করতে পারে না। সবাই সমান ক্ষমতা রাখে।

সহজ উদাহরণ: কল্পনা করুন একটি ক্লাসের সবার কাছে একই খাতার কপি আছে। যদি কেউ নিজের খাতায় কিছু পরিবর্তন করে, বাকি সবার খাতায় সেটা মিলবে না — তাই ধরা পড়ে যাবে। Blockchain ঠিক এভাবেই কাজ করে।

Blockchain-এর মূল বৈশিষ্ট্য

🔗 Immutability (অপরিবর্তনযোগ্যতা)

একবার লেখা তথ্য মুছে বা পরিবর্তন করা যায় না।

🌐 Decentralization (বিকেন্দ্রীকরণ)

কোনো একক কর্তৃপক্ষ নেই, হাজার হাজার node আছে।

🔍 Transparency (স্বচ্ছতা)

সব লেনদেন সবাই দেখতে পারে (Public Blockchain)।

🔐 Security (নিরাপত্তা)

Cryptography ব্যবহার করে তথ্য সুরক্ষিত থাকে।

Blockchain কোথায় ব্যবহার হয়?

  • 💰 Cryptocurrency: Bitcoin, Ethereum, BNB
  • 🎨 NFT (Digital Art): ডিজিটাল শিল্পকর্মের মালিকানা
  • 🏥 Healthcare: রোগীর মেডিকেল রেকর্ড
  • 🚢 Supply Chain: পণ্য ট্র্যাকিং
  • 🗳️ Voting System: নিরাপদ ভোটদান
  • 🏦 DeFi: বিকেন্দ্রীভূত ব্যাংকিং
  • 📜 Smart Contract: স্বয়ংক্রিয় চুক্তি

Blockchain তৈরি — Node.js

এখন আমরা Node.js দিয়ে একটি সম্পূর্ণ কার্যকরী Blockchain তৈরি করব।

Step 1: Block ক্লাস তৈরি

block.js 
const SHA256 = require('crypto-js/sha256');

// প্রতিটি Block-এর গঠন
class Block {
  constructor(index, timestamp, data, previousHash = '') {
    this.index = index;           // Block নম্বর
    this.timestamp = timestamp;   // সময়
    this.data = data;             // তথ্য/লেনদেন
    this.previousHash = previousHash; // আগের Block-এর Hash
    this.nonce = 0;              // Mining-এর জন্য
    this.hash = this.calculateHash();
  }

  // Hash গণনা করা
  calculateHash() {
    return SHA256(
      this.index +
      this.previousHash +
      this.timestamp +
      JSON.stringify(this.data) +
      this.nonce
    ).toString();
  }

  // Mining — নির্দিষ্ট কঠিনতায় Hash খোঁজা
  mineBlock(difficulty) {
    while (this.hash.substring(0, difficulty) !==
           Array(difficulty + 1).join('0')) {
      this.nonce++;
      this.hash = this.calculateHash();
    }
    console.log(`Block mined! Hash: ${this.hash}`);
  }
}

module.exports = Block;

Step 2: Blockchain ক্লাস তৈরি

blockchain.js 
const Block = require('./block');

class Blockchain {
  constructor() {
    this.chain = [this.createGenesisBlock()]; // প্রথম block
    this.difficulty = 4; // Mining কঠিনতা
    this.pendingTransactions = []; // অপেক্ষমাণ লেনদেন
    this.miningReward = 100; // Mining পুরস্কার
  }

  // Genesis Block (প্রথম Block) তৈরি
  createGenesisBlock() {
    return new Block(0, '01/01/2024', 'Genesis Block', '0');
  }

  // সর্বশেষ Block পাওয়া
  getLatestBlock() {
    return this.chain[this.chain.length - 1];
  }

  // নতুন Block যোগ করা (Mining সহ)
  minePendingTransactions(miningRewardAddress) {
    const block = new Block(
      this.chain.length,
      Date.now(),
      this.pendingTransactions,
      this.getLatestBlock().hash
    );
    block.mineBlock(this.difficulty);
    this.chain.push(block);
    this.pendingTransactions = [
      { from: null, to: miningRewardAddress, amount: this.miningReward }
    ];
  }

  // লেনদেন যোগ করা
  createTransaction(transaction) {
    this.pendingTransactions.push(transaction);
  }

  // ব্যালেন্স দেখা
  getBalance(address) {
    let balance = 0;
    for (const block of this.chain) {
      for (const trans of block.data) {
        if (trans.from === address) balance -= trans.amount;
        if (trans.to === address) balance += trans.amount;
      }
    }
    return balance;
  }

  // Blockchain বৈধতা যাচাই
  isChainValid() {
    for (let i = 1; i < this.chain.length; i++) {
      const current = this.chain[i];
      const previous = this.chain[i - 1];

      if (current.hash !== current.calculateHash()) {
        console.log('❌ Hash মিলছে না!');
        return false;
      }
      if (current.previousHash !== previous.hash) {
        console.log('❌ Chain ভাঙা!');
        return false;
      }
    }
    return true;
  }
}

module.exports = Blockchain;

contracts/BanglaToken.sol 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

// বাংলা টোকেন — একটি সহজ Cryptocurrency
contract BanglaToken {
    string public name = "Bangla Token";
    string public symbol = "BDT";
    uint256 public totalSupply;
    address public owner;

    // প্রতিটি wallet-এর ব্যালেন্স
    mapping(address => uint256) public balances;

    // অনুমতি ম্যাপিং
    mapping(address => mapping(address => uint256)) public allowances;

    // ঘটনা (Event) — লেনদেনের রেকর্ড
    event Transfer(address indexed from, address indexed to, uint256 amount);
    event Approval(address indexed owner, address indexed spender, uint256 amount);

    // Contract শুরু হওয়ার সময়
    constructor(uint256 _initialSupply) {
        owner = msg.sender;
        totalSupply = _initialSupply * 10 ** 18; // 18 decimal
        balances[msg.sender] = totalSupply;
    }

    // টোকেন পাঠানো
    function transfer(address _to, uint256 _amount) public returns (bool) {
        require(balances[msg.sender] >= _amount, "ব্যালেন্স কম!");
        require(_to != address(0), "ঠিকানা ভুল!");

        balances[msg.sender] -= _amount;
        balances[_to] += _amount;

        emit Transfer(msg.sender, _to, _amount);
        return true;
    }

    // ব্যালেন্স দেখা
    function balanceOf(address _account) public view returns (uint256) {
        return balances[_account];
    }

    // নতুন টোকেন তৈরি (শুধু owner পারবে)
    function mint(address _to, uint256 _amount) public {
        require(msg.sender == owner, "শুধু মালিক পারবে!");
        totalSupply += _amount;
        balances[_to] += _amount;
        emit Transfer(address(0), _to, _amount);
    }
}

db.js — LevelDB দিয়ে Block সংরক্ষণ 
const { Level } = require('level');

class BlockchainDB {
  constructor() {
    this.db = new Level('./blockchain-data', {
      valueEncoding: 'json'
    });
  }

  // Block সংরক্ষণ
  async saveBlock(block) {
    await this.db.put(`block:${block.index}`, block);
    await this.db.put('latest', block.index); // সর্বশেষ block নম্বর
    console.log(`✅ Block ${block.index} সংরক্ষিত হয়েছে`);
  }

  // Block পড়া
  async getBlock(index) {
    try {
      return await this.db.get(`block:${index}`);
    } catch (err) {
      return null;
    }
  }

  // সব Block লোড করা
  async getAllBlocks() {
    const blocks = [];
    const latest = await this.db.get('latest');
    for (let i = 0; i <= latest; i++) {
      blocks.push(await this.getBlock(i));
    }
    return blocks;
  }
}

module.exports = BlockchainDB;

SecureContract.sol — নিরাপদ Contract লেখার নিয়ম 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract SecureVault is ReentrancyGuard, Ownable {
    mapping(address => uint256) private deposits;

    // ✅ Checks-Effects-Interactions Pattern মানা
    function withdraw(uint256 amount) public nonReentrant {
        // 1. Check — শর্ত যাচাই
        require(deposits[msg.sender] >= amount, "যথেষ্ট নেই");

        // 2. Effects — state পরিবর্তন
        deposits[msg.sender] -= amount;

        // 3. Interactions — বাইরে পাঠানো
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "পাঠানো ব্যর্থ");
    }

    // ✅ Integer overflow নেই (Solidity 0.8+ এ built-in)
    function deposit() public payable {
        deposits[msg.sender] += msg.value;
    }
}

p2p-server.js — P2P নেটওয়ার্ক 
const WebSocket = require('ws');

const P2P_PORT = process.env.P2P_PORT || 6001;
const peers = process.env.PEERS ? process.env.PEERS.split(',') : [];

class P2PServer {
  constructor(blockchain) {
    this.blockchain = blockchain;
    this.sockets = [];
  }

  // Server শুরু করা
  listen() {
    const server = new WebSocket.Server({ port: P2P_PORT });
    server.on('connection', socket => this.connectSocket(socket));
    this.connectToPeers();
    console.log(`🌐 P2P Server চলছে port: ${P2P_PORT}`);
  }

  // Peer-এর সাথে যোগ দেওয়া
  connectToPeers() {
    peers.forEach(peer => {
      const socket = new WebSocket(peer);
      socket.on('open', () => this.connectSocket(socket));
      console.log(`🔗 Peer-এর সাথে যুক্ত হচ্ছে: ${peer}`);
    });
  }

  connectSocket(socket) {
    this.sockets.push(socket);
    console.log('✅ নতুন Peer যুক্ত হয়েছে!');
    socket.on('message', message => {
      const data = JSON.parse(message);
      this.handleMessage(data);
    });
    this.sendChain(socket); // নিজের chain শেয়ার করা
  }

  // নতুন Block ব্রডকাস্ট করা
  broadcastNewBlock() {
    this.sockets.forEach(socket => this.sendChain(socket));
  }

  sendChain(socket) {
    socket.send(JSON.stringify(this.blockchain.chain));
  }

  handleMessage(chain) {
    // দীর্ঘতম valid chain গ্রহণ করা
    if (chain.length > this.blockchain.chain.length) {
      this.blockchain.chain = chain;
      console.log('🔄 Chain আপডেট হয়েছে!');
    }
  }
}

module.exports = P2PServer;

import { useState } from 'react';
import { ethers } from 'ethers';

function App() {
  const [wallet, setWallet] = useState('');
  const [balance, setBalance] = useState('');

  // MetaMask-এর সাথে সংযোগ
  async function connectWallet() {
    if (window.ethereum) {
      const provider = new ethers.BrowserProvider(window.ethereum);
      const accounts = await provider.send('eth_requestAccounts', []);
      setWallet(accounts[0]);

      const bal = await provider.getBalance(accounts[0]);
      setBalance(ethers.formatEther(bal) + ' ETH');
    } else {
      alert('MetaMask ইনস্টল করুন!');
    }
  }

  // ETH পাঠানো
  async function sendETH(toAddress, amount) {
    const provider = new ethers.BrowserProvider(window.ethereum);
    const signer = await provider.getSigner();

    const tx = await signer.sendTransaction({
      to: toAddress,
      value: ethers.parseEther(amount)
    });

    console.log(`⏳ TX Hash: ${tx.hash}`);
    await tx.wait(); // নিশ্চিত হওয়ার অপেক্ষা
    console.log('✅ লেনদেন সম্পন্ন!');
  }

  return (
    <div>
      <h1>আমার DApp</h1>
      <button onClick={connectWallet}>Wallet সংযুক্ত করুন</button>
      {wallet && <p>ঠিকানা: {wallet}</p>}
      {balance && <p>ব্যালেন্স: {balance}</p>}
    </div>
  );
}

export default App;
