Late Static Binding (LSB) — static:: vs self::

self:: সবসময় যে class-এ লেখা আছে সেটা refer করে। কিন্তু static:: runtime-এ actual class টা refer করে। এটা না বুঝলে inheritance-এ বড় bug হয়।

late_static_binding.php
<?php

class Base {
    public static function create(): static {
        return new static(); // static:: = runtime class
    }

    public static function className(): string {
        return static::class; // runtime class name
    }

    // self:: problem দেখো:
    public static function badCreate(): self {
        return new self(); // সবসময় Base বানাবে!
    }
}

class Child extends Base {
    public string $name = 'Child Object';
}

$obj = Child::create();    // Child instance ✓
$bad = Child::badCreate(); // Base instance ✗ (bug!)

var_dump($obj instanceof Child); // true
var_dump($bad instanceof Child); // false — bug!

// Practical: Singleton Pattern with LSB
class Singleton {
    private static array $instances = [];

    final private function __construct() {}

    public static function getInstance(): static {
        $class = static::class;
        if (!isset(self::$instances[$class])) {
            self::$instances[$class] = new static();
        }
        return self::$instances[$class];
    }
}

class Database extends Singleton {}
class Cache    extends Singleton {}

$db1 = Database::getInstance();
$db2 = Database::getInstance();
echo $db1 === $db2 ? "Same instance ✓" : "Different!";

magic_methods_advanced.php
<?php

// ১. Dynamic Property Access + Validation
class StrictModel {
    private array $data = [];
    private array $rules = [
        'age'   => 'int',
        'email' => 'email',
        'name'  => 'string',
    ];

    public function __set(string $name, $value): void {
        if (!array_key_exists($name, $this->rules)) {
            throw new \InvalidArgumentException("Unknown field: $name");
        }
        $this->validate($name, $value);
        $this->data[$name] = $value;
    }

    public function __get(string $name): mixed {
        return $this->data[$name] ?? null;
    }

    public function __isset(string $name): bool {
        return isset($this->data[$name]);
    }

    private function validate(string $field, $value): void {
        match($this->rules[$field]) {
            'int'    => is_int($value) || throw new \TypeError("$field must be int"),
            'email'  => filter_var($value, FILTER_VALIDATE_EMAIL) || throw new \ValueError("Invalid email"),
            'string' => is_string($value) || throw new \TypeError("$field must be string"),
        };
    }
}

$user = new StrictModel();
$user->name  = "রাহুল";
$user->email = "rahul@example.com";
$user->age   = 25;

// ২. __invoke — Object কে Function হিসেবে ব্যবহার
class Multiplier {
    public function __construct(
        private readonly int $factor
    ) {}

    public function __invoke(int $value): int {
        return $value * $this->factor;
    }
}

echo $triple(10); // 30 — object কে function হিসেবে call
echo array_map($triple, [1, 2, 3, 4]); // [3, 6, 9, 12]

Repository Pattern + Observer Pattern (Real-world)

Repository Pattern দিয়ে database logic এবং business logic আলাদা রাখো। Observer দিয়ে loose coupling তৈরি করো।

repository_observer_pattern.php
<?php

// ═══ Repository Pattern ═══

interface UserRepositoryInterface {
    public function findById(int $id): ?UserEntity;
    public function findByEmail(string $email): ?UserEntity;
    public function save(UserEntity $user): UserEntity;
    public function delete(int $id): void;
    public function findAll(UserFilter $filter): array;
}

class MySQLUserRepository implements UserRepositoryInterface {
    public function __construct(
        private readonly \PDO $pdo
    ) {}

    public function findById(int $id): ?UserEntity {
        $stmt = $this->pdo->prepare(
            'SELECT * FROM users WHERE id = :id LIMIT 1'
        );
        $stmt->execute(['id' => $id]);
        $row = $stmt->fetch(\PDO::FETCH_ASSOC);
        return $row ? UserEntity::fromArray($row) : null;
    }

    public function save(UserEntity $user): UserEntity {
        if ($user->id === null) {
            return $this->insert($user);
        }
        return $this->update($user);
    }

    private function insert(UserEntity $user): UserEntity {
        $stmt = $this->pdo->prepare(
            'INSERT INTO users (name, email, created_at) VALUES (:name, :email, NOW())'
        );
        $stmt->execute(['name' => $user->name, 'email' => $user->email]);
        return $user->withId((int)$this->pdo->lastInsertId());
    }
    // ... findByEmail, delete, findAll, update
}

// Testing-এর জন্য InMemory Repository
class InMemoryUserRepository implements UserRepositoryInterface {
    private array $users = [];
    private int $nextId = 1;

    public function save(UserEntity $user): UserEntity {
        $user = $user->withId($this->nextId++);
        $this->users[$user->id] = $user;
        return $user;
    }
    // real DB ছাড়াই test চালানো যাবে!
}

// ═══ Observer Pattern ═══

interface EventInterface {
    public function getName(): string;
}

interface ListenerInterface {
    public function handle(EventInterface $event): void;
}

class EventDispatcher {
    private array $listeners = [];

    public function listen(string $event, ListenerInterface|callable $listener): void {
        $this->listeners[$event][] = $listener;
    }

    public function dispatch(EventInterface $event): void {
        $name = $event->getName();
        foreach ($this->listeners[$name] ?? [] as $listener) {
            if ($listener instanceof ListenerInterface) {
                $listener->handle($event);
            } else {
                $listener($event); // callable
            }
        }
    }
}

class UserRegistered implements EventInterface {
    public function __construct(public readonly UserEntity $user) {}
    public function getName(): string { return 'user.registered'; }
}

class SendWelcomeEmail implements ListenerInterface {
    public function handle(EventInterface $event): void {
        echo "Email পাঠানো হচ্ছে: {$event->user->email}\n";
    }
}

// Usage:
$dispatcher = new EventDispatcher();
$dispatcher->listen('user.registered', new SendWelcomeEmail());
$dispatcher->listen('user.registered', fn($e) => logActivity($e->user));
$dispatcher->dispatch(new UserRegistered($newUser));

নিজে DI Container বানানো (Auto-wiring সহ)

Laravel, Symfony সবাই DI Container ব্যবহার করে। তুমি নিজে একটা বানালে বুঝবে কীভাবে কাজ করে।

di_container.php
<?php

class Container {
    private array $bindings  = [];
    private array $instances = [];
    private array $singletons = [];

    // Interface → Implementation বাঁধো
    public function bind(string $abstract, string|callable $concrete): void {
        $this->bindings[$abstract] = $concrete;
    }

    // Singleton binding
    public function singleton(string $abstract, string|callable $concrete): void {
        $this->bind($abstract, $concrete);
        $this->singletons[$abstract] = true;
    }

    // Resolve করো (auto-wiring)
    public function make(string $abstract): mixed {
        // Singleton cache
        if (isset($this->singletons[$abstract], $this->instances[$abstract])) {
            return $this->instances[$abstract];
        }

        $concrete = $this->bindings[$abstract] ?? $abstract;

        $instance = is_callable($concrete)
            ? $concrete($this)
            : $this->build($concrete);

        if (isset($this->singletons[$abstract])) {
            $this->instances[$abstract] = $instance;
        }

        return $instance;
    }

    // Auto-wiring: Reflection দিয়ে dependency inject করো
    private function build(string $class): mixed {
        $ref = new \ReflectionClass($class);

        if (!$ref->isInstantiable()) {
            throw new \RuntimeException("Cannot instantiate: $class");
        }

        $constructor = $ref->getConstructor();
        if ($constructor === null) {
            return new $class();
        }

        $deps = array_map(
            function(\ReflectionParameter $param) {
                $type = $param->getType();
                if ($type instanceof \ReflectionNamedType && !$type->isBuiltin()) {
                    return $this->make($type->getName());
                }
                if ($param->isDefaultValueAvailable()) {
                    return $param->getDefaultValue();
                }
                throw new \RuntimeException("Cannot resolve: {$param->getName()}");
            },
            $constructor->getParameters()
        );

        return $ref->newInstanceArgs($deps);
    }
}

// ═══ Usage ═══
$container = new Container();

$container->singleton(UserRepositoryInterface::class, MySQLUserRepository::class);
$container->singleton(\PDO::class, fn() => new \PDO('mysql:host=localhost;dbname=test', 'root', ''));

// Auto-wiring: MySQLUserRepository → PDO automatically inject হবে!
$repo = $container->make(UserRepositoryInterface::class);

<?php

// ═══ Generator হলো একটি "pausable function" ═══

// ১. বড় file পড়া — memory efficient
function readLargeFile(string $path): \Generator {
    $handle = fopen($path, 'r');
    try {
        while (!feof($handle)) {
            yield fgets($handle); // একটা line, তারপর pause
        }
    } finally {
        fclose($handle);
    }
}

// শুধু কয়েক KB memory, যদিও file ১ GB হয়!
foreach (readLargeFile('/var/log/nginx/access.log') as $line) {
    if (str_contains($line, 'ERROR')) {
        echo $line;
    }
}

// ২. Pipeline দিয়ে data process
function map(\Generator $gen, callable $fn): \Generator {
    foreach ($gen as $key => $value) {
        yield $key => $fn($value);
    }
}

function filter(\Generator $gen, callable $fn): \Generator {
    foreach ($gen as $key => $value) {
        if ($fn($value)) {
            yield $key => $value;
        }
    }
}

// Pipeline: পুরো array একবারে load না করে step by step
$pipeline = filter(
    map(
        readLargeFile('data.csv'),
        fn($line) => str_getcsv($line)
    ),
    fn($row) => (int)$row[2] > 1000 // amount > 1000
);

// ৩. Bidirectional Generator (send/receive)
function accumulator(): \Generator {
    $total = 0;
    while (true) {
        $value = yield $total; // yield করে total, receive করে value
        if ($value === null) break;
        $total += $value;
    }
}

$acc = accumulator();
$acc->current(); // initialize
$acc->send(10);   // total: 10
$acc->send(25);   // total: 35
echo $acc->send(5); // 40

<?php

// ১. Trait দিয়ে Timestamp behavior যোগ করো
trait HasTimestamps {
    private ?\DateTimeImmutable $createdAt = null;
    private ?\DateTimeImmutable $updatedAt = null;

    public function touch(): void {
        $now = new \DateTimeImmutable();
        if ($this->createdAt === null) {
            $this->createdAt = $now;
        }
        $this->updatedAt = $now;
    }

    public function getCreatedAt(): ?\DateTimeImmutable { return $this->createdAt; }
    public function getUpdatedAt(): ?\DateTimeImmutable { return $this->updatedAt; }
}

// ২. Trait-এ abstract method — class কে বাধ্য করো
trait Serializable {
    abstract protected function getSerializableFields(): array;

    public function toJson(): string {
        $data = [];
        foreach ($this->getSerializableFields() as $field) {
            $data[$field] = $this->$field;
        }
        return json_encode($data, JSON_UNESCAPED_UNICODE);
    }
}

// ৩. Conflict Resolution
trait A {
    public function hello(): string { return "A"; }
    public function shared(): string { return "A::shared"; }
}

trait B {
    public function world(): string { return "B"; }
    public function shared(): string { return "B::shared"; }
}

class MyClass {
    use A, B {
        A::shared insteadof B; // A wins
        B::shared as sharedFromB; // B কে rename করো

class AttributeValidator {
    public function validate(object $model): array {
        $errors = [];
        $ref    = new \ReflectionClass($model);

        foreach ($ref->getProperties() as $property) {
            $value = $property->getValue($model);
            $name  = $property->getName();

            foreach ($property->getAttributes(Validate::class) as $attr) {
                $validate = $attr->newInstance();
                $error    = $this->check($validate->rule, $value, $validate->value);
                if ($error) $errors[$name][] = $error;
            }
        }
        return $errors;
    }

    private function check(string $rule, $value, $param): ?string {
        return match($rule) {
            'minLength' => strlen($value) < $param ? "Min length {$param}" : null,
            'maxLength' => strlen($value) > $param ? "Max length {$param}" : null,
            'email'     => filter_var($value, FILTER_VALIDATE_EMAIL) ? null : "Invalid email",
            'min'       => $value < $param ? "Min value {$param}" : null,
            'max'       => $value > $param ? "Max value {$param}" : null,
            default     => null,
        };
    }
}

class OrmMapper {
    public function toArray(object $model): array {

        foreach ($ref->getProperties() as $prop) {

            if (empty($attrs)) continue;
            $colName = $attrs[0]->newInstance()->name;
            $result[$colName] = $prop->getValue($model);
        }

    }

// ═══ Transaction Manager ═══
class TransactionManager {
    private int $depth = 0; // Nested transaction support

    public function __construct(private readonly \PDO $pdo) {}

    public function transaction(callable $callback): mixed {
        $this->begin();
        try {
            $result = $callback($this->pdo);
            $this->commit();
            return $result;
        } catch (\Throwable $e) {
            $this->rollback();
            throw $e;
        }
    }

    private function begin(): void {
        if ($this->depth === 0) {
            $this->pdo->beginTransaction();
        } else {
            // Savepoint for nested transactions
            $this->pdo->exec("SAVEPOINT sp{$this->depth}");
        }
        $this->depth++;
    }

    private function commit(): void {
        $this->depth--;
        if ($this->depth === 0) {
            $this->pdo->commit();
        } else {
            $this->pdo->exec("RELEASE SAVEPOINT sp{$this->depth}");
        }
    }

    private function rollback(): void {
        $this->depth--;
        if ($this->depth === 0) {
            $this->pdo->rollBack();
        } else {
            $this->pdo->exec("ROLLBACK TO SAVEPOINT sp{$this->depth}");
        }
    }
}

// Batch Insert — N+1 সমস্যা সমাধান
function batchInsert(\PDO $pdo, string $table, array $rows): void {
    if (empty($rows)) return;

    $cols        = array_keys($rows[0]);
    $placeholders = implode(',', array_fill(0, count($cols), '?'));
    $colStr      = implode(',', $cols);

    // Chunk করো, একবারে সব নয়
    foreach (array_chunk($rows, 500) as $chunk) {
        $rowsPlaceholder = implode(',', array_fill(0, count($chunk), "({$placeholders})"));
        $sql    = "INSERT INTO {$table} ({$colStr}) VALUES {$rowsPlaceholder}";
        $values = array_merge(...array_map('array_values', $chunk));
        $pdo->prepare($sql)->execute($values);
    }
}

// ═══ 5. Encryption (না হলে data leak) ═══
class Encryptor {
    private const ALGO      = 'aes-256-gcm';
    private const IV_LENGTH = 12; // GCM standard

    public function __construct(private readonly string $key) {
        if (strlen($key) !== 32) {
            throw new \LengthException('Key must be 32 bytes (256 bits)');
        }
    }

    public function encrypt(string $data): string {
        $iv         = random_bytes(self::IV_LENGTH);
        $ciphertext = openssl_encrypt($data, self::ALGO, $this->key,
                         OPENSSL_RAW_DATA, $iv, $tag);
        return base64_encode($iv . $tag . $ciphertext);
    }

    public function decrypt(string $encoded): string {
        $raw        = base64_decode($encoded);
        $iv         = substr($raw, 0, self::IV_LENGTH);
        $tag        = substr($raw, self::IV_LENGTH, 16);
        $ciphertext = substr($raw, self::IV_LENGTH + 16);
        return openssl_decrypt($ciphertext, self::ALGO, $this->key,
                   OPENSSL_RAW_DATA, $iv, $tag);
    }
}

class RateLimiter {
    public function __construct(
        private \Redis $redis,
        private int $maxAttempts = 5,
        private int $decaySeconds = 60
    ) {}

    public function attempt(string $key): bool {
        $hits = $this->redis->incr("rl:{$key}");
        if ($hits === 1) {
            $this->redis->expire("rl:{$key}", $this->decaySeconds);
        }
        return $hits <= $this->maxAttempts;
    }
}

// ১. Currying — function এর আংশিক প্রয়োগ
function curry(callable $fn): \Closure {
    $arity = (new \ReflectionFunction(\Closure::fromCallable($fn)))->getNumberOfParameters();

    $accumulator = function(array $args) use ($fn, $arity, &$accumulator): mixed {
        if (count($args) >= $arity) {
            return $fn(...$args);
        }
        return function() use ($args, $accumulator) {
            return $accumulator(array_merge($args, func_get_args()));
        };
    };

    return function() use ($accumulator) {
        return $accumulator(func_get_args());
    };
}

$add = curry(fn($a, $b) => $a + $b);
$add5 = $add(5); // partial application!
echo $add5(3); // 8
echo $add5(10); // 15

// ২. Memoization — expensive computation cache
function memoize(callable $fn): \Closure {
    $cache = [];
    return function() use ($fn, &$cache) {
        $key = serialize(func_get_args());
        return $cache[$key] ??= $fn(...func_get_args());
    };
}

$fibonacci = memoize(function(int $n) use (&$fibonacci): int {
    return $n <= 1 ? $n : $fibonacci($n - 1) + $fibonacci($n - 2);
});

echo $fibonacci(40); // instant! (memoized)

// ৩. Pipeline — data এর মধ্য দিয়ে function chain
class Pipeline {
    private array $stages = [];

    public function pipe(callable $stage): self {
        $new           = clone $this;
        $new->stages[] = $stage;
        return $new;
    }

    public function process(mixed $payload): mixed {
        return array_reduce(
            $this->stages,
            fn($carry, $stage) => $stage($carry),
            $payload
        );
    }
}

$result = (new Pipeline())
    ->pipe(fn($x) => $x * 2)
    ->pipe(fn($x) => $x + 10)
    ->pipe(fn($x) => "Result: {$x}")
    ->process(5);
// "Result: 20"

// ৪. Compose — functions compose করো
function compose(callable ...$fns): \Closure {
    return function($value) use ($fns) {
        return array_reduce(
            array_reverse($fns),
            fn($v, $f) => $f($v),
            $value
        );
    };
}

$transform = compose(
    fn($s) => strtoupper($s),
    fn($s) => trim($s),
    fn($s) => str_replace('-', ' ', $s)
);
echo $transform('  hello-world  '); // "HELLO WORLD"
