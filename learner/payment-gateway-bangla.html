HTTPS এবং SSL সার্টিফিকেট

🌐 HTTPS বাধ্যতামূলক কেন?

সব পেমেন্ট গেটওয়ে (bKash, Nagad, SSLCommerz) Production-এ HTTPS URL ছাড়া কাজ করে না। HTTP ব্যবহার করলে পেমেন্ট ডেটা Man-in-the-Middle Attack-এর শিকার হতে পারে।

💡
Let's Encrypt থেকে বিনামূল্যে SSL সার্টিফিকেট নিন। অথবা Hosting Panel (cPanel) থেকে সহজেই সেটআপ করুন।

🔧 Laravel-এ HTTPS Force করুন

// AppServiceProvider.php boot() মেথডে
if (app()->environment('production')) {
    \URL::forceScheme('https');
}

// .env ফাইলে
APP_URL=https://yoursite.com

📄 Laravel — VerifyCsrfToken.php
protected $except = [
    'payment/bkash/callback',
    'payment/sslcommerz/ipn',
    'payment/nagad/callback',
];
🔐 Webhook Signature যাচাই — SSLCommerz
public function sslIpn(Request $request)
{
    // SSLCommerz Hash Verification
    $postData       = $request->all();
    $receivedHash   = $postData['verify_sign'];
    $storePassword  = config('payment.sslcommerz.store_password');

    // hash তৈরি করুন
    unset($postData['verify_sign'], $postData['verify_key']);
    ksort($postData);

    $hashString = md5($storePassword);
    foreach ($postData as $key => $value) {
        $hashString .= $key . md5($value);
    }
    $generatedHash = md5($hashString);

    if ($generatedHash !== $receivedHash) {
        return response()->json(['error' => 'Invalid signature'], 403);
    }

    // এখন পেমেন্ট প্রসেস করুন
}
🚨
সবচেয়ে বড় ভুল: Callback পেলেই Order complete করা। সবসময় Gateway-এর API-তে সরাসরি Query/Verify করে পেমেন্ট নিশ্চিত করুন।

Schema::create('payment_transactions', function (Blueprint $table) {
    $table->id();
    $table->string('order_id')->unique();
    $table->string('gateway');             // bkash, nagad, sslcommerz
    $table->string('gateway_trx_id')->nullable();

    $table->string('currency')->default('BDT');
    $table->enum('status', ['pending', 'success', 'failed', 'cancelled']);

    $table->ipAddress('ip_address')->nullable();
    $table->timestamp('paid_at')->nullable();
    $table->timestamps();
});

Laravel — bKash ইন্টিগ্রেশন উদাহরণ

📦 Step 1: Token নিন

📄 BkashService.php
class BkashService
{
    private $baseUrl;
    private $token;

    public function __construct()
    {
        $this->baseUrl = config('payment.bkash.base_url');
    }

    public function getToken(): string
    {
        // Cache করুন — বারবার token নেওয়া API limit শেষ করে
        return Cache::remember('bkash_token', 3500, function () {
            $response = Http::withHeaders([
                'username'   => config('payment.bkash.username'),
                'password'   => config('payment.bkash.password'),
            ])->post($this->baseUrl . '/tokenized/checkout/token/grant', [
                'app_key'    => config('payment.bkash.app_key'),
                'app_secret' => config('payment.bkash.app_secret'),
            ]);

            return $response->json('id_token');
        });
    }

    public function createPayment(array $data): array
    {
        $token = $this->getToken();

        $response = Http::withHeaders([
            'Authorization' => $token,
            'X-APP-Key'     => config('payment.bkash.app_key'),
        ])->post($this->baseUrl . '/tokenized/checkout/create', [
            'mode'           => '0011',
            'payerReference' => $data['user_id'],
            'callbackURL'    => route('bkash.callback'),
            'amount'         => $data['amount'],
            'currency'       => 'BDT',
            'intent'         => 'sale',
            'merchantInvoiceNumber' => $data['order_id'],
        ]);

        return $response->json();
    }

    public function executePayment(string $paymentId): array
    {
        $token = $this->getToken();

        $response = Http::withHeaders([
            'Authorization' => $token,
            'X-APP-Key'     => config('payment.bkash.app_key'),
        ])->post($this->baseUrl . '/tokenized/checkout/execute', [
            'paymentID' => $paymentId,
        ]);

        return $response->json();
    }
}

📩 Step 2: Callback হ্যান্ডেল করুন (যাচাইসহ)

📄 PaymentController.php
public function bkashCallback(Request $request, BkashService $bkash)
{
    $paymentId = $request->get('paymentID');
    $status    = $request->get('status');

    if ($status !== 'success') {
        return redirect()->route('payment.failed');
    }

    // ✅ সবসময় Execute করে যাচাই করুন
    $result = $bkash->executePayment($paymentId);

    if ($result['statusCode'] !== '0000') {
        Log::error('bKash payment failed', $result);
        return redirect()->route('payment.failed');
    }

    // Double payment চেক করুন
    $existing = PaymentTransaction::where('gateway_trx_id', $result['trxID'])->first();
    if ($existing) {
        return redirect()->route('payment.success');
    }

    // DB Transaction ব্যবহার করুন
    DB::transaction(function () use ($result) {
        PaymentTransaction::where('order_id', $result['merchantInvoiceNumber'])
            ->update([
                'status'           => 'success',
                'gateway_trx_id'   => $result['trxID'],
                'gateway_response' => json_encode($result),
                'paid_at'          => now(),
            ]);
        // এখন order fulfill করুন
    });

    return redirect()->route('payment.success');
}
