Controller — সঠিক পদ্ধতি

CI3-তে Controller-ই সবচেয়ে বেশি ভুল হয়। Controller শুধু Request ও Response পরিচালনা করবে — business logic এখানে থাকবে না।

Base Controller — MY_Controller

সব controller-এ দরকারি shared logic MY_Controller-এ রাখুন। এটি CI3-এর সবচেয়ে গুরুত্বপূর্ণ pattern।

✅ Base Controllerapplication/core/MY_Controller.php
<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class MY_Controller extends CI_Controller
{
    protected $data = [];  // view-এ pass করার data

    public function __construct()
    {
        parent::__construct();
        // সব controller-এ দরকারি setup
        $this->load->model('User_model');
        $this->data['site_name'] = $this->config->item('site_name');
    }

    // ── View Loader Helper ─────────────────────
    protected function render($view, $extra_data = [])
    {
        $data = array_merge($this->data, $extra_data);
        $this->load->view('templates/header', $data);
        $this->load->view($view, $data);
        $this->load->view('templates/footer', $data);
    }

    // ── JSON Response Helper ───────────────────
    protected function json_response($data, $status_code = 200)
    {
        $this->output
            ->set_status_header($status_code)
            ->set_content_type('application/json')
            ->set_output(json_encode($data));
    }

    // ── Auth Check ────────────────────────────
    protected function require_login()
    {
        if (!$this->session->userdata('user_id')) {
            redirect('auth/login');
        }
    }
}

// ── Admin Base Controller ──────────────────────────
class Admin_Controller extends MY_Controller
{
    public function __construct()
    {
        parent::__construct();
        $this->require_login();
        $this->require_role('admin');
    }

    protected function require_role($role)
    {
        if ($this->session->userdata('role') !== $role) {
            show_error('Access Denied', 403);
        }
    }
}

✅ Slim Controller — সঠিক পদ্ধতিcontrollers/frontend/Users.php
<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class Users extends MY_Controller
{
    public function __construct()
    {
        parent::__construct();
        $this->load->model('User_model');
        $this->load->library(['form_validation', 'email']);
        $this->load->helper('security');
    }

    public function index()
    {
        $users = $this->User_model->get_all_paginated(15);
        $this->render('users/index', ['users' => $users]);
    }

    public function register()
    {
        if ($this->input->server('REQUEST_METHOD') !== 'POST') {
            $this->render('users/register');
            return;
        }

        if (!$this->_validate_registration()) {
            $this->render('users/register', [
                'errors' => $this->form_validation->error_array(),
            ]);
            return;
        }

        $user_id = $this->User_model->register([
            'name'     => $this->input->post('name', TRUE),
            'email'    => $this->input->post('email', TRUE),
            'password' => $this->input->post('password'),
        ]);

        $this->session->set_flashdata('success', 'নিবন্ধন সফল!');
        redirect('users/login');
    }

    // ── Private Helpers ───────────────────────
    private function _validate_registration()
    {
        $this->form_validation->set_rules('name',     'নাম',       'required|min_length[2]|max_length[100]|trim');
        $this->form_validation->set_rules('email',    'ইমেইল',     'required|valid_email|is_unique[users.email]');
        $this->form_validation->set_rules('phone',    'ফোন',       'required|regex_match[/^01[3-9]\d{8}$/]');
        $this->form_validation->set_rules('password', 'পাসওয়ার্ড', 'required|min_length[8]|matches[confirm_password]');
        return $this->form_validation->run();
    }
}

API Controller — JSON Response

✅ CI3 API Controllercontrollers/api/v1/Users.php
<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class Users extends MY_Controller
{
    public function __construct()
    {
        parent::__construct();
        $this->load->model('User_model');
        // API middleware: token check
        $this->_verify_api_token();
    }

    // GET /api/v1/users
    public function index()
    {
        $users = $this->User_model->get_all_users();
        $this->json_response([
            'status'  => 'success',
            'data'    => $users,
            'count'   => count($users),
        ]);
    }

    // GET /api/v1/users/1
    public function show($id)
    {
        $user = $this->User_model->get_user_by_id((int)$id);

        if (!$user) {
            $this->json_response(['status' => 'error', 'message' => 'User not found'], 404);
            return;
        }

        $this->json_response(['status' => 'success', 'data' => $user]);
    }

    // POST /api/v1/users
    public function create()
    {
        $input = json_decode($this->input->raw_input_stream, true);

        if (!isset($input['name'], $input['email'], $input['password'])) {
            $this->json_response(['status' => 'error', 'message' => 'Missing required fields'], 422);
            return;
        }

        try {
            $user_id = $this->User_model->create_user($input);
            $user    = $this->User_model->get_user_by_id($user_id);
            $this->json_response(['status' => 'success', 'data' => $user], 201);
        } catch (Exception $e) {
            $this->json_response(['status' => 'error', 'message' => $e->getMessage()], 500);
        }
    }

    private function _verify_api_token()
    {
        $token = $this->input->get_request_header('Authorization');
        if (!$token || !$this->_is_valid_token($token)) {
            $this->json_response(['status' => 'error', 'message' => 'Unauthorized'], 401);
            exit;
        }
    }

    private function _is_valid_token($token)
    {
        // token validation logic
        return $this->User_model->verify_api_token($token);
    }
}

✅ ভালো Model — সম্পূর্ণ পদ্ধতিapplication/models/User_model.php
<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class User_model extends CI_Model
{
    private $table = 'users';

    // ── Read Operations ────────────────────────

    public function get_all_users()
    {
        return $this->db
            ->select('id, name, email, phone, role, created_at')
            ->where('deleted_at', NULL)
            ->order_by('created_at', 'DESC')
            ->get($this->table)
            ->result_array();
    }

    public function get_all_paginated($per_page = 15, $offset = 0)
    {
        return $this->db
            ->select('id, name, email, role, created_at')
            ->where('deleted_at', NULL)
            ->order_by('created_at', 'DESC')
            ->limit($per_page, $offset)
            ->get($this->table)
            ->result_array();
    }

    public function get_user_by_id($id)
    {
        return $this->db
            ->where('id', (int)$id)
            ->where('deleted_at', NULL)
            ->get($this->table)
            ->row_array();
    }

    public function get_user_by_email($email)
    {
        return $this->db
            ->where('email', $email)
            ->where('deleted_at', NULL)
            ->get($this->table)
            ->row_array();
    }

    public function count_all()
    {
        return $this->db
            ->where('deleted_at', NULL)
            ->count_all_results($this->table);
    }

    // ── Write Operations ───────────────────────

    public function create_user($data)
    {
        $insert_data = [
            'name'       => $this->security->xss_clean($data['name']),
            'email'      => strtolower(trim($data['email'])),
            'phone'      => $data['phone'] ?? NULL,
            'password'   => password_hash($data['password'], PASSWORD_BCRYPT),
            'role'       => $data['role'] ?? 'user',
            'created_at' => date('Y-m-d H:i:s'),
        ];

        $this->db->insert($this->table, $insert_data);
        return $this->db->insert_id();
    }

    public function update_user($id, $data)
    {
        // শুধু allowed fields update করুন
        $allowed_fields = ['name', 'phone', 'email', 'role'];
        $update_data = array_intersect_key($data, array_flip($allowed_fields));
        $update_data['updated_at'] = date('Y-m-d H:i:s');

        $this->db->where('id', (int)$id)->update($this->table, $update_data);
        return $this->db->affected_rows();
    }

    public function soft_delete($id)
    {
        // সত্যিকারের delete না — soft delete
        $this->db->where('id', (int)$id)->update($this->table, [
            'deleted_at' => date('Y-m-d H:i:s'),
        ]);
        return $this->db->affected_rows() > 0;
    }

    // ── Auth ───────────────────────────────────

    public function verify_login($email, $password)
    {
        $user = $this->get_user_by_email($email);

        if (!$user) {
            return false;
        }

        // password_verify() — সঠিক পদ্ধতি
        if (!password_verify($password, $user['password'])) {
            return false;
        }

        // Password return করবেন না!
        unset($user['password']);
        return $user;
    }
}

<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class Sms_sender
{
    private $ci;
    private $api_url;
    private $api_key;
    private $sender_id;

    public function __construct()
    {
        $this->ci =& get_instance(); // CI instance
        $this->ci->config->load('sms');
        $this->api_url   = $this->ci->config->item('sms_api_url');
        $this->api_key   = $this->ci->config->item('sms_api_key');
        $this->sender_id = $this->ci->config->item('sms_sender_id');
    }

    public function send($phone, $message)
    {
        if (!$this->is_valid_phone($phone)) {
            log_message('error', "Invalid phone: {$phone}");
            return false;
        }

        $payload = [
            'api_key'   => $this->api_key,
            'sender_id' => $this->sender_id,
            'to'        => $phone,
            'message'   => $message,
        ];

        return $this->make_request($payload);
    }

    public function send_otp($phone)
    {
        $otp = sprintf('%06d', random_int(100000, 999999));
        $message = "আপনার OTP কোড: {$otp} (৫ মিনিট বৈধ)";

        if ($this->send($phone, $message)) {
            // OTP session-এ রাখুন
            $this->ci->session->set_userdata([
                'otp_code'   => password_hash($otp, PASSWORD_BCRYPT),
                'otp_expiry' => time() + (5 * 60),
            ]);
            return true;
        }
        return false;
    }

    private function is_valid_phone($phone)
    {
        return (bool)preg_match('/^01[3-9]\d{8}$/', $phone);
    }

    private function make_request($payload)
    {
        $ch = curl_init($this->api_url);
        curl_setopt_array($ch, [
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_POST           => true,
            CURLOPT_POSTFIELDS     => json_encode($payload),
            CURLOPT_HTTPHEADER     => ['Content-Type: application/json'],
            CURLOPT_TIMEOUT        => 30,
        ]);
        $response = curl_exec($ch);
        $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);
        return $http_code === 200;
    }
}

✅ Secure Session Managementcontrollers/Auth.php
class Auth extends MY_Controller
{
    private $max_attempts = 5;
    private $lockout_time = 900; // 15 minutes

    public function login()
    {
        if ($this->session->userdata('user_id')) {
            redirect('dashboard');
        }

        if ($this->input->server('REQUEST_METHOD') !== 'POST') {
            $this->render('auth/login');
            return;
        }

        // Brute force check
        if ($this->is_rate_limited()) {
            $this->session->set_flashdata('error', 'অনেকবার চেষ্টা করা হয়েছে। ১৫ মিনিট পরে আবার চেষ্টা করুন।');
            $this->render('auth/login');
            return;
        }

        $email    = $this->input->post('email', TRUE);
        $password = $this->input->post('password');

        $user = $this->User_model->verify_login($email, $password);

        if (!$user) {
            $this->increment_login_attempts();
            $this->session->set_flashdata('error', 'ইমেইল বা পাসওয়ার্ড সঠিক নয়');
            $this->render('auth/login');
            return;
        }

        // Session regenerate — session fixation প্রতিরোধ
        $this->session->sess_regenerate(true);
        $this->reset_login_attempts();

        $this->session->set_userdata([
            'user_id'    => $user['id'],
            'user_name'  => $user['name'],
            'user_email' => $user['email'],
            'user_role'  => $user['role'],
            'logged_in'  => TRUE,
        ]);

        redirect('dashboard');
    }

    public function logout()
    {
        $this->session->sess_destroy();
        redirect('auth/login');
    }

    private function is_rate_limited()
    {
        $attempts = $this->session->userdata('login_attempts') ?? 0;
        $last_attempt = $this->session->userdata('last_attempt_time') ?? 0;

        if ($attempts >= $this->max_attempts
            && (time() - $last_attempt) < $this->lockout_time) {
            return true;
        }
        return false;
    }
}

ভুল ১ — MD5/SHA1 দিয়ে Password Hash

❌ কখনো এটা করবেন না!

// MD5 — ২০০৩ সালেই ভাঙা!
$pass = md5($password);

// SHA1 — ও insecure
$pass = sha1($password);

// MD5 + salt — এখনো insecure!
$pass = md5($salt . $password);

// Plain text store!
$pass = $password; // সর্বনাশ!

✅ password_hash() ব্যবহার করুন

// ✅ সঠিক — bcrypt
$hashed = password_hash(
    $password,
    PASSWORD_BCRYPT
);

// ✅ Verify করার সময়
if (password_verify($input_pass, $db_hash)) {
    // login success
}

// PHP 7.2+ উত্তম পদ্ধতি
$hashed = password_hash(
    $password,
    PASSWORD_ARGON2ID

🚀 Performance চেকলিস্ট

🔵 Production Deployment-এর আগে
  • $config['cache_query_string'] = TRUE — Query string caching
  • Database-এ frequently searched column-এ Index যোগ করুন
  • SELECT * এড়িয়ে প্রয়োজনীয় column শুধু নিন
  • CI3-এর built-in caching ব্যবহার করুন: $this->output->cache(5)
  • Static assets (CSS/JS/Image) CDN বা browser cache ব্যবহার করুন
  • PHP OpCache চালু রাখুন (server config)
