Singleton

একক সত্তা প্যাটার্ন
💡 সহজ উপমা
বিয়ের কাজী উপমা: সারা গ্রামে মাত্র একজন কাজী আছেন। যতই বিয়ে হোক না কেন, সেই একজন কাজীকেই ডাকা হয়। নতুন কাজী তৈরি হয় না — সবসময় সেই একই কাজী।

একইভাবে Singleton নিশ্চিত করে যে একটি class-এর সারা application জুড়ে মাত্র একটিই instance থাকবে।
🤔 কেন দরকার?
মনে করুন আপনি প্রতিবার Database connection দরকার হলে নতুন connection তৈরি করছেন। ১০০ request এলে ১০০টি connection! এটি সার্ভারকে slow করবে, resource নষ্ট হবে। Singleton দিয়ে একটিই connection বারবার ব্যবহার করা যায়।

✔ Logger (সব জায়গা থেকে একই log file-এ লেখা)
✔ Cache Manager
✔ Session Manager
⚠️ কখন ব্যবহার করবেন না

// Pure PHP Singleton
class DatabaseConnection
{
    private static ?DatabaseConnection $instance = null;
    private \PDO $pdo;

    // constructor private — বাইরে থেকে new করা যাবে না
    private function __construct()
    {
        $this->pdo = new \PDO('mysql:host=localhost;dbname=mydb', 'root', '');
    }

    // clone ও unserialize বন্ধ
    private function __clone() {}
    public function __wakeup() { throw new \Exception("Cannot unserialize"); }

    public static function getInstance(): DatabaseConnection
    {
        if (static::$instance === null) {
            static::$instance = new static();
        }
        return static::$instance;
    }

    public function query(string $sql): array
    {
        return $this->pdo->query($sql)->fetchAll();
    }
}

// ব্যবহার
$db1 = DatabaseConnection::getInstance();
$db2 = DatabaseConnection::getInstance();

var_dump($db1 === $db2); // bool(true) — একই object!

// Interface — সকল Notification-এর চুক্তি
interface NotificationInterface
{
    public function send(string $to, string $message): bool;
}

// Concrete Products
class EmailNotification implements NotificationInterface
{
    public function send(string $to, string $msg): bool
    {
        mail($to, 'Notification', $msg);
        return true;
    }
}

class SMSNotification implements NotificationInterface
{
    public function send(string $to, string $msg): bool
    {
        // SMS API call
        return true;
    }
}

// Factory
class NotificationFactory
{
    public static function create(string $type): NotificationInterface
    {
        return match($type) {
            'email' => new EmailNotification(),
            'sms'   => new SMSNotification(),
            default => throw new \InvalidArgumentException("Unknown: {$type}"),
        };
    }
}

class PaymentGatewayFactory
{

    {
        return match($gateway) {

class QueryBuilder
{
    private string $table = '';
    private string $fields = '*';
    private array $conditions = [];
    private ?int $limit = null;
    private ?string $orderBy = null;

    public function table(string $t): static
    { $this->table = $t; return $this; }

    public function select(string $f): static
    { $this->fields = $f; return $this; }

    public function where(string $cond): static
    { $this->conditions[] = $cond; return $this; }

    public function limit(int $n): static
    { $this->limit = $n; return $this; }

    public function orderBy(string $col): static
    { $this->orderBy = $col; return $this; }

    public function build(): string
    {
        $sql = "SELECT {$this->fields} FROM {$this->table}";
        if ($this->conditions)
            $sql .= ' WHERE ' . implode(' AND ', $this->conditions);
        if ($this->orderBy)
            $sql .= " ORDER BY {$this->orderBy}";
        if ($this->limit)
            $sql .= " LIMIT {$this->limit}";
        return $sql;
    }
}

// ব্যবহার — Method Chaining

class OrderService implements \SplSubject
{
    private \SplObjectStorage $observers;
    private ?array $lastOrder = null;

    public function __construct()
    { $this->observers = new \SplObjectStorage(); }

    public function attach(\SplObserver $o): void
    { $this->observers->attach($o); }

    public function detach(\SplObserver $o): void
    { $this->observers->detach($o); }

    public function notify(): void
    { foreach ($this->observers as $o) $o->update($this); }

    public function placeOrder(array $order): void
    {
        $this->lastOrder = $order;
        $this->notify(); // সকল observer-কে জানাও
    }

    public function getLastOrder(): ?array { return $this->lastOrder; }
}

class CacheMiddleware
{
    public function handle(Request $request, Closure $next): Response
    {
        $key = 'response_' . md5($request->fullUrl());

        if (Cache::has($key)) {

        $response = $next($request); // মূল controller চালাও
        Cache::put($key, $response->getContent(), 3600);

        return $response;
    }
}

abstract class ApprovalHandler
{
    private ?ApprovalHandler $next = null;

    public function setNext(ApprovalHandler $h): ApprovalHandler
    { $this->next = $h; return $h; }

    protected function passToNext(int $amount): string
    { return $this->next ? $this->next->approve($amount) : 'অনুমোদন দেওয়া যাচ্ছে না'; }

    abstract public function approve(int $amount): string;
}

enum OrderStatus: string
{
    case PENDING    = 'pending';
    case PROCESSING = 'processing';
    case SHIPPED    = 'shipped';
    case DELIVERED  = 'delivered';
    case CANCELLED  = 'cancelled';

    public function canTransitionTo(self $new): bool
    {
        return match($this) {
            self::PENDING    => in_array($new, [self::PROCESSING, self::CANCELLED]),
            self::PROCESSING => in_array($new, [self::SHIPPED,    self::CANCELLED]),
            self::SHIPPED    => $new === self::DELIVERED,
            default         => false,
        };
    }
}

    {

    }

interface Expression
{ public function interpret(array $ctx): mixed; }

class NumberExpr implements Expression
{
    public function __construct(private float $val) {}
    public function interpret(array $ctx): float { return $this->val; }
}

class VariableExpr implements Expression
{
    public function __construct(private string $name) {}
    public function interpret(array $ctx): float
    { return $ctx[$this->name] ?? 0; }
}

class AddExpr implements Expression
{
    public function __construct(
        private Expression $l, private Expression $r) {}
    public function interpret(array $ctx): float
    { return $this->l->interpret($ctx) + $this->r->interpret($ctx); }
}

    new VariableExpr('price'),
    new VariableExpr('tax')
);

class SearchQueryInterpreter
{
    public function interpret(string $query): Builder
    {
        $builder = Product::query();
        $tokens  = explode(' ', $query);

        foreach ($tokens as $token) {
            if (str_starts_with($token, 'category:')) {
                $builder->where('category', substr($token, 9));
            } elseif (str_starts_with($token, 'price>')) {
                $builder->where('price', '>', (int)substr($token, 6));
            } else {
                $builder->where('name', 'LIKE', "%{$token}%");
            }
        }

        return $builder;
    }
}
