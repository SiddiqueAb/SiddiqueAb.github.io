02
Class ও Object
Blueprint এবং তার থেকে তৈরি বস্তু

Class হলো একটি blueprint বা ছাঁচ। Object হলো সেই ছাঁচ থেকে তৈরি বাস্তব জিনিস।

উদাহরণ: গাড়ি হলো Class, আর Toyota Corolla বা Honda Civic হলো Object।

Car.php PHP 
// Class তৈরি করা — এটি হলো ছাঁচ/blueprint
class Car {
    // Properties (গুণাবলী)
    public $brand;
    public $color;
    public $speed = 0;

    // Method (কাজ)
    public function accelerate($amount) {
        $this->speed += $amount;
        echo "গতি বেড়ে হলো: {$this->speed} km/h\n";
    }

    public function brake() {
        $this->speed = 0;
        echo "গাড়ি থেমে গেছে!\n";
    }
}

// Object তৈরি করা — new keyword দিয়ে
$toyota = new Car();
$toyota->brand = "Toyota";
$toyota->color = "লাল";
$toyota->accelerate(60);  // গতি বেড়ে হলো: 60 km/h
$toyota->accelerate(30);  // গতি বেড়ে হলো: 90 km/h
$toyota->brake();          // গাড়ি থেমে গেছে!

// একই class থেকে আরেকটি object
$honda = new Car();
$honda->brand = "Honda";
// $toyota এবং $honda সম্পূর্ণ আলাদা object!
📌 $this কী?
$this হলো বর্তমান object-কে বোঝানোর উপায়। যখন $toyota->accelerate() কল হয়, তখন method-এর ভেতরে $this মানে হলো $toyota নিজেই।

Properties (বৈশিষ্ট্য)

Properties হলো class-এর ভেতরে declare করা variables। এগুলো object-এর ডেটা ধারণ করে।

class Student {
    // Properties বিভিন্ন ধরনের হতে পারে
    public $name;              // সবাই দেখতে পাবে
    protected $grade;          // শুধু এই class ও child class
    private $password;         // শুধু এই class

    // Default value সহ
    public $rollNumber = 0;
    public $subjects = [];

    // PHP 8.0+ — Typed Properties
    public string $email;
    public int $age;
    public float $gpa = 0.0;
    public bool $isActive = true;

    // Method
    public function introduce() {
        return "আমি {$this->name}, বয়স {$this->age}";

Constructor ও Destructor
Object তৈরি ও ধ্বংসের সময় স্বয়ংক্রিয় কাজ

Constructor হলো একটি special method যা object তৈরির সাথে সাথে স্বয়ংক্রিয়ভাবে চলে। PHP-তে এর নাম সবসময় __construct()।

BankAccount.php PHP 
class BankAccount {
    private $owner;
    private $balance;
    private $accountNo;

    // Constructor — object তৈরির সময় চলে
    public function __construct(string $owner, float $initialBalance = 0) {
        $this->owner     = $owner;
        $this->balance   = $initialBalance;
        $this->accountNo = rand(100000, 999999);

        echo "✅ {$owner}-এর অ্যাকাউন্ট #{$this->accountNo} তৈরি হয়েছে\n";
    }

    public function deposit(float $amount): void {
        $this->balance += $amount;
        echo "জমা: ৳{$amount} | ব্যালেন্স: ৳{$this->balance}\n";
    }

    public function withdraw(float $amount): bool {
        if ($amount > $this->balance) {
            echo "❌ পর্যাপ্ত ব্যালেন্স নেই!\n";
            return false;
        }
        $this->balance -= $amount;
        echo "উত্তোলন: ৳{$amount} | ব্যালেন্স: ৳{$this->balance}\n";
        return true;
    }

    // Destructor — object মুছে যাওয়ার সময় চলে
    public function __destruct() {
        echo "🗑️ {$this->owner}-এর সেশন শেষ হয়েছে\n";
    }
}

// Object তৈরির সাথে সাথেই constructor চলে
$acc = new BankAccount("করিম", 5000);
$acc->deposit(2000);
$acc->withdraw(1500);

Access Modifiers
public, protected, private — কখন কোনটা?
Modifier নিজ Class Child Class বাইরে থেকে ব্যবহার
public ✅ ✅ ✅ যা সবাই দেখতে ও ব্যবহার করতে পারবে
protected ✅ ✅ ❌ নিজ class ও inherit করা class-এর জন্য
private ✅ ❌ ❌ শুধুমাত্র নিজ class-এর ভেতরে
AccessDemo.php PHP 
class Person {
    public    $name     = "রহিম";     // সবাই দেখতে পাবে
    protected $salary   = 50000;    // child class দেখতে পাবে
    private   $nidNumber = "1234567";// শুধু এই class

    public function getName(): string {
        return $this->name;
    }

    public function getNid(): string {
        // private property নিজের method থেকে দেখা যায়
        return "NID: ****" . substr($this->nidNumber, -3);
    }
}

class Employee extends Person {
    public function showSalary() {
        // protected salary দেখতে পাচ্ছে child class
        echo "বেতন: ৳{$this->salary}\n";

        // এটা ERROR দেবে — private দেখা যায় না!
        // echo $this->nidNumber;
    }
}

$emp = new Employee();
echo $emp->name;       // ✅ public — দেখা যায়
// echo $emp->salary;   ❌ protected — বাইরে দেখা যায় না
// echo $emp->nidNumber;❌ private — বাইরে দেখা যায় না

Encapsulation — ডেটা সুরক্ষা
Getter ও Setter দিয়ে ডেটা নিয়ন্ত্রণ

Encapsulation মানে হলো ডেটাকে বাইরে থেকে সরাসরি পরিবর্তন করতে না দেওয়া। এর পরিবর্তে Getter ও Setter method ব্যবহার করি।

Temperature.php PHP 
class Temperature {
    private float $celsius;

    public function __construct(float $celsius) {
        $this->setCelsius($celsius);
    }

    // Setter — validation সহ
    public function setCelsius(float $celsius): void {
        if ($celsius < -273.15) {
            throw new InvalidArgumentException(
                "তাপমাত্রা পরম শূন্যের নিচে হতে পারে না!"
            );
        }
        $this->celsius = $celsius;
    }

    // Getter
    public function getCelsius(): float {
        return $this->celsius;
    }

    public function getFahrenheit(): float {
        return ($this->celsius * 9/5) + 32;
    }

    public function getKelvin(): float {
        return $this->celsius + 273.15;
    }
}

$temp = new Temperature(100);
echo $temp->getCelsius();    // 100
echo $temp->getFahrenheit(); // 212
echo $temp->getKelvin();    // 373.15

// এটা error দেবে:
// $temp->setCelsius(-300); // Exception!
✅ সুবিধা
Setter-এ validation রাখা যায় ফলে ভুল ডেটা কখনো ঢুকতে পারবে না। যেমন বয়স -5 দিলেই error, তাপমাত্রা অসম্ভব হলেই error।

Inheritance — উত্তরাধিকার
extends keyword দিয়ে parent class-এর সব পাওয়া

Inheritance মানে একটি class আরেকটি class-এর সব properties ও methods পাবে। যে পাচ্ছে সে Child/Sub Class, যার থেকে পাচ্ছে সে Parent/Super Class।

Inheritance.php PHP 
// Parent Class
class Animal {
    protected string $name;
    protected int    $age;

    public function __construct(string $name, int $age) {
        $this->name = $name;
        $this->age  = $age;
    }

    public function breathe(): string {
        return "{$this->name} শ্বাস নিচ্ছে...";
    }

    public function introduce(): string {
        return "আমি {$this->name}, বয়স {$this->age} বছর";
    }

    // Child class override করতে পারবে
    public function makeSound(): string {
        return "...";
    }
}

// Child Class — extends দিয়ে inherit করা
class Dog extends Animal {
    private string $breed;

    public function __construct(string $name, int $age, string $breed) {
        // parent::__construct() দিয়ে parent-এর constructor ডাকি
        parent::__construct($name, $age);
        $this->breed = $breed;
    }

    // Method Override — parent-এর makeSound() বদলে দেওয়া
    public function makeSound(): string {
        return "{$this->name} বলছে: ঘেউ ঘেউ! 🐕";
    }

    public function fetch(): string {
        return "{$this->name} বল নিয়ে এসেছে!";
    }
}

class Cat extends Animal {
    public function makeSound(): string {
        return "{$this->name} বলছে: মিয়াউ! 🐈";
    }
}

$dog = new Dog("ভুলু", 3, "Labrador");
echo $dog->introduce();   // ✅ Parent থেকে পেয়েছে
echo $dog->breathe();    // ✅ Parent থেকে পেয়েছে
echo $dog->makeSound();  // ✅ Override হয়েছে
echo $dog->fetch();      // ✅ Dog-এর নিজের method

$cat = new Cat("মিনি", 2);
echo $cat->makeSound();  // মিনি বলছে: মিয়াউ!

Polymorphism — বহুরূপী
একই interface, ভিন্ন ভিন্ন আচরণ

Polymorphism মানে "অনেক রূপ"। একটাই function call করলে object-এর ধরন অনুযায়ী ভিন্ন কাজ হবে।

Polymorphism.php PHP 
abstract class Shape {
    abstract public function area(): float;
    abstract public function perimeter(): float;

    public function describe(): string {
        return sprintf(
            "%s | ক্ষেত্রফল: %.2f | পরিধি: %.2f",
            get_class($this),
            $this->area(),
            $this->perimeter()
        );
    }
}

class Circle extends Shape {
    public function __construct(private float $radius) {}

    public function area(): float {
        return M_PI * $this->radius ** 2;
    }
    public function perimeter(): float {
        return 2 * M_PI * $this->radius;
    }
}

class Rectangle extends Shape {
    public function __construct(
        private float $width,
        private float $height
    ) {}

    public function area(): float      { return $this->width * $this->height; }
    public function perimeter(): float { return 2 * ($this->width + $this->height); }
}

class Triangle extends Shape {
    public function __construct(
        private float $a, private float $b, private float $c
    ) {}

    public function area(): float {
        $s = ($this->a + $this->b + $this->c) / 2;
        return sqrt($s * ($s-$this->a) * ($s-$this->b) * ($s-$this->c));
    }
    public function perimeter(): float { return $this->a + $this->b + $this->c; }
}

// Polymorphism — একই loop-এ সব shapes কাজ করছে!
$shapes = [
    new Circle(7),
    new Rectangle(4, 6),
    new Triangle(3, 4, 5),
];

foreach ($shapes as $shape) {
    echo $shape->describe() . "\n";
}

Abstraction — বিমূর্ততা
Abstract Class দিয়ে নিয়ম বেঁধে দেওয়া

Abstract Class হলো এমন class যার নিজস্ব object তৈরি করা যায় না। এটি শুধু blueprint হিসেবে কাজ করে। Abstract method declare করা হয় কিন্তু body নেই — child class-কে সেটি implement করতেই হবে।

Payment.php PHP 
// Abstract class — object তৈরি করা যাবে না
abstract class Payment {
    protected $amount;
    protected $currency = 'BDT';

    public function __construct(float $amount) {
        $this->amount = $amount;
    }

    // Abstract methods — implement করতেই হবে
    abstract public function processPayment(): bool;
    abstract public function getGatewayName(): string;

    // Regular method — সবাই পাবে
    public function pay(): string {
        if ($this->processPayment()) {
            return "✅ {$this->getGatewayName()} দিয়ে ৳{$this->amount} পেমেন্ট সফল!";
        }
        return "❌ পেমেন্ট ব্যর্থ!";
    }
}

class BkashPayment extends Payment {
    private $phone;

    public function __construct(float $amount, string $phone) {
        parent::__construct($amount);
        $this->phone = $phone;
    }

    public function processPayment(): bool {
        // bKash API call হবে এখানে
        echo "📱 {$this->phone} নম্বরে PIN চাওয়া হচ্ছে...\n";
        return true;
    }

    public function getGatewayName(): string {
        return "bKash";
    }
}

class NagadPayment extends Payment {
    public function processPayment(): bool {
        echo "💰 Nagad processing...\n";
        return true;
    }
    public function getGatewayName(): string { return "Nagad"; }
}

$bkash = new BkashPayment(500, "01712345678");
echo $bkash->pay();

$nagad = new NagadPayment(200);
echo $nagad->pay();

Interface ও Trait

Interface

Interface হলো একটি চুক্তি (contract)। যে class এটি implement করবে সে সব method তৈরি করতে বাধ্য। Interface-এ কোনো code থাকে না, শুধু signature।

Interfaces.php PHP 
// Interface — চুক্তি
interface Printable {
    public function print(): string;
}

interface Saveable {
    public function save(): bool;
    public function load(int $id): static;
}

// একটা class একাধিক interface implement করতে পারে!
class Invoice implements Printable, Saveable {
    public function __construct(
        private int    $id,
        private string $customer,
        private float  $total
    ) {}

    public function print(): string {
        return "ইনভয়েস #{$this->id} | {$this->customer} | ৳{$this->total}";
    }

    public function save(): bool {
        // DB-তে save করার code
        return true;
    }

    public function load(int $id): static {
        // DB থেকে load করার code
        return new static($id, "Test", 0);
    }
}

Trait

Trait হলো code sharing-এর একটি পদ্ধতি। PHP-তে একটি class শুধু একটি parent class থেকে extend করতে পারে। কিন্তু একাধিক Trait use করতে পারে।

Traits.php PHP 
// Trait — পুনর্ব্যবহারযোগ্য code block
trait Timestamps {
    private $createdAt;
    private $updatedAt;

    public function setCreatedAt(): void {
        $this->createdAt = new DateTime();
    }
    public function getCreatedAt(): string {
        return $this->createdAt?->format('Y-m-d H:i:s') ?? 'N/A';
    }
}

trait SoftDelete {
    private $deletedAt = null;

    public function softDelete(): void {
        $this->deletedAt = new DateTime();
    }
    public function isDeleted(): bool {
        return $this->deletedAt !== null;
    }
}

// Custom Exception class
class ValidationException extends RuntimeException {
    private $errors;

    public function __construct(array $errors) {
        $this->errors = $errors;
        parent::__construct("Validation failed");
    }

    public function getErrors(): array {
        return $this->errors;
    }
}

class NotFoundException extends RuntimeException {
    public function __construct(string $resource, int $id) {
        parent::__construct("{$resource} #{$id} পাওয়া যায়নি", 404);
    }
}

// Exception throw করা ও ধরা
function createUser(array $data): array {
    $errors = [];

    if (empty($data['name']))  $errors[] = "নাম প্রয়োজন";
    if (!filter_var($data['email'] ?? '', FILTER_VALIDATE_EMAIL))
        $errors[] = "ইমেইল সঠিক নয়";

    if (!empty($errors)) {
        throw new ValidationException($errors);
    }

    return ['success' => true, 'data' => $data];
}

// try-catch-finally
try {
    $result = createUser(['name' => '', 'email' => 'invalid']);
} catch (ValidationException $e) {
    echo "Validation Error: " . implode(", ", $e->getErrors());
} catch (NotFoundException $e) {
    echo "404: " . $e->getMessage();
} catch (Exception $e) {
    // যেকোনো অন্য error
    echo "Error: " . $e->getMessage();
} finally {
    // সবসময় চলবে — error হোক বা না হোক
    echo "প্রক্রিয়া শেষ।";
}

Step 3 — Router Class

core/Router.php PHP 
<?php
namespace Core;

class Router {
    private array $routes = [];

    // GET route register করা
    public function get(string $path, array|callable $handler): static {
        $this->addRoute('GET', $path, $handler);
        return $this;
    }

    // POST route
    public function post(string $path, array|callable $handler): static {
        $this->addRoute('POST', $path, $handler);
        return $this;
    }

    private function addRoute(string $method, string $path, $handler): void {
        // Dynamic segments: /user/{id} → regex
        $pattern = preg_replace('/\{([^}]+)\}/', '([^/]+)', $path);
        $pattern = '#^' . $pattern . '$#';

        // Parameter names বের করা
        preg_match_all('/\{([^}]+)\}/', $path, $params);

        $this->routes[] = [
            'method'  => $method,
            'pattern' => $pattern,
            'params'  => $params[1],
            'handler' => $handler,
        ];
    }

    // Request dispatch করা
    public function dispatch(Request $request): void {
        $uri    = parse_url($request->getUri(), PHP_URL_PATH);
        $method = $request->getMethod();

        foreach ($this->routes as $route) {
            if ($route['method'] !== $method) continue;

            if (preg_match($route['pattern'], $uri, $matches)) {
                array_shift($matches);

                // URL parameters collect করা
                $paramValues = array_combine($route['params'], $matches) ?: [];
                $request->setParams($paramValues);

                $handler = $route['handler'];

                if (is_callable($handler)) {
                    // Closure handler
                    ($handler)($request);
                } elseif (is_array($handler)) {
                    // [Controller::class, 'method'] format
                    [$controllerClass, $method] = $handler;
                    $controller = new $controllerClass();
                    $controller->$method($request);
                }
                return;
            }
        }

        // 404 Not Found
        http_response_code(404);
        echo "<h1>404 — পৃষ্ঠা পাওয়া যায়নি</h1>";
    }
}

<?php
namespace Core;

abstract class Controller {
    protected function view(string $viewPath, array $data = []): void {
        // Data extract করা — array key = variable নাম
        extract($data);

        $file = ROOT_PATH . '/app/Views/'
               . str_replace('.', '/', $viewPath)
               . '.php';

        if (!file_exists($file)) {
            throw new \RuntimeException("View '{$viewPath}' পাওয়া যায়নি");
        }

        require $file;
    }

    protected function json(mixed $data, int $status = 200): void {
        http_response_code($status);
        header('Content-Type: application/json; charset=utf-8');
        echo json_encode($data, JSON_UNESCAPED_UNICODE);
    }

    protected function redirect(string $url): void {
        header("Location: {$url}");
        exit;
    }
}

core/Request.php PHP 
<?php
namespace Core;

class Request {
    private array $params = [];

    public function getUri(): string {
        return $_SERVER['REQUEST_URI'] ?? '/';
    }

    public function getMethod(): string {
        return strtoupper($_SERVER['REQUEST_METHOD'] ?? 'GET');
    }

    public function getBody(): array {
        if ($this->getMethod() === 'POST') {
            // JSON body check করা
            $input = file_get_contents('php://input');
            $json  = json_decode($input, true);

            if (json_last_error() === JSON_ERROR_NONE) return $json;
            return $_POST;
        }
        return $_GET;
    }

    public function setParams(array $params): void {
        $this->params = $params;
    }

    public function getParam(string $key, mixed $default = null): mixed {
        return $this->params[$key] ?? $default;
    }
}
core/Database.php — PDO Wrapper PHP 
<?php
namespace Core;

class Database {
    private static ?self $instance = null;
    private \PDO $pdo;

    private function __construct() {
        $config = require ROOT_PATH . '/config/database.php';

        $dsn = "mysql:host={$config['host']};dbname={$config['name']};charset=utf8mb4";

        $this->pdo = new \PDO($dsn, $config['user'], $config['pass'], [
            \PDO::ATTR_ERRMODE            => \PDO::ERRMODE_EXCEPTION,
            \PDO::ATTR_DEFAULT_FETCH_MODE => \PDO::FETCH_ASSOC,
        ]);
    }

    public static function getInstance(): static {
        if (self::$instance === null) {
            self::$instance = new static();
        }
        return self::$instance;
    }

    public function query(string $sql, array $params = []): \PDOStatement {
        $stmt = $this->pdo->prepare($sql);
        $stmt->execute($params);
        return $stmt;
    }

    public function lastInsertId(): int {
        return (int) $this->pdo->lastInsertId();
    }
}
