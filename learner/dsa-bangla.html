PHP
// Node Class তৈরি
class Node {
    public $data;
    public $next;

    public function __construct($data) {
        $this->data = $data;
        $this->next = null;
    }
}

// LinkedList Class
class LinkedList {
    private $head = null;

    // শেষে যোগ করা — O(n)
    public function append($data) {
        $new = new Node($data);
        if (!$this->head) {
            $this->head = $new;
            return;
        }
        $curr = $this->head;
        while ($curr->next) {
            $curr = $curr->next;
        }
        $curr->next = $new;
    }

    // Print করা
    public function print() {
        $curr = $this->head;
        while ($curr) {
            echo $curr->data . " → ";
            $curr = $curr->next;
        }
        echo "NULL";
    }
}

$list = new LinkedList();
$list->append(10);
$list->append(25);
$list->append(37);
$list->print(); // 10 → 25 → 37 → NULL

PHP — Real Example: Bracket Check
function isBalanced($str): bool {
    $stack = [];
    $map = [')' => '(', ']' => '[', '}' => '{'];

    foreach (str_split($str) as $ch) {
        if (in_array($ch, ['(', '[', '{'])) {
            array_push($stack, $ch); // push opening
        } elseif (isset($map[$ch])) {
            if (empty($stack) || array_pop($stack) !== $map[$ch])
                return false;
        }
    }
    return empty($stack);
}

echo isBalanced("{[()]}") ? "✅ Balanced" : "❌ Not Balanced";
// Output: ✅ Balanced

class TreeNode {
    public $val, $left, $right;
    function __construct($v) {
        $this->val = $v;
        $this->left = $this->right = null;
    }
}

class BST {
    public $root = null;

    public function insert($val) {
        $this->root = $this->_insert(
            $this->root, $val
        );
    }

    private function _insert($node, $val) {
        if (!$node) return new TreeNode($val);
        if ($val < $node->val)
            $node->left = $this->_insert(
                $node->left, $val
            );
        else
            $node->right = $this->_insert(
                $node->right, $val
            );
        return $node;
    }
}

$graph = [
    'A' => ['B', 'E'],

    'C' => ['B', 'D'],
    'D' => ['B', 'C', 'E'],
    'E' => ['A', 'D'],

// BFS — Breadth First Search
function bfs($graph, $start) {
    $visited = [$start => true];
    $queue   = [$start];

    while (!empty($queue)) {
        $node = array_shift($queue);
        echo $node . " ";

        foreach ($graph[$node] as $neighbor) {
            if (!isset($visited[$neighbor])) {
                $visited[$neighbor] = true;
                $queue[] = $neighbor;
            }
        }
    }
}

bfs($graph, 'A');
// Output: A B E C D

PHP
function bubbleSort($arr) {
    $n = count($arr);
    for ($i = 0; $i < $n - 1; $i++) {
        for ($j = 0; $j < $n - $i - 1; $j++) {
            if ($arr[$j] > $arr[$j+1]) {
                // Swap
                [$arr[$j], $arr[$j+1]] = [$arr[$j+1], $arr[$j]];
            }
        }
    }
    return $arr;
}

$arr = [64, 34, 25, 12, 22, 11, 90];
print_r(bubbleSort($arr));
// [11, 12, 22, 25, 34, 64, 90]

🔀 Merge Sort — মার্জ সর্ট O(n log n) — দ্রুত

ভাগ করো → সাজাও → মিলাও। Divide & Conquer কৌশল।

১
ভাগ করো: Array কে মাঝ থেকে দুই ভাগ করো
২

function mergeSort($arr) {
    $n = count($arr);
    if ($n <= 1) return $arr;

    $mid   = (int) ($n / 2);
    $left  = mergeSort(array_slice($arr, 0, $mid));
    $right = mergeSort(array_slice($arr, $mid));

    return merge($left, $right);
}

function merge($left, $right) {
    $result = [];
    $i = $j = 0;

    while ($i < count($left) && $j < count($right)) {
        if ($left[$i] <= $right[$j]) {
            $result[] = $left[$i++];
        } else {
            $result[] = $right[$j++];
        }
    }

    return array_merge($result,
        array_slice($left, $i),
        array_slice($right, $j)
    );
}

print_r(mergeSort([38, 27, 43, 3, 9, 82, 10]));
// [3, 9, 10, 27, 38, 43, 82]

function linearSearch($arr, $target) {
    foreach ($arr as $i => $val) {
        if ($val === $target) return $i;
    }
    return -1;
}

echo linearSearch([10,25,37,42], 37);

function binarySearch($arr, $target) {
    $low = 0; $high = count($arr) - 1;

    while ($low <= $high) {
        $mid = (int)(($low + $high) / 2);

        if ($arr[$mid] === $target)
            return $mid;    // পাওয়া গেছে!
        elseif ($arr[$mid] < $target)
            $low = $mid + 1; // ডানে খোঁজো
        else
            $high = $mid - 1;// বামে খোঁজো
    }
    return -1;
}

সারসংক্ষেপ — কোনটা কখন? Quick Reference

📦 Array
Product list, User list, Config data, JSON response। Index দিয়ে দ্রুত access দরকার হলে।
🔗 Linked List
Playlist, Undo list, Browser history। মাঝে মাঝে insert/delete বেশি হলে।
🥞 Stack
Undo/Redo, Back button, Function calls, Bracket matching। LIFO দরকার হলে।
🚶 Queue
Laravel Jobs, Email queue, Print queue, Customer service। FIFO দরকার হলে।
🌳 Tree
Category tree, Comment threads, Route system, File directory। Hierarchy দরকার হলে।
🕸️ Graph
Social network, Google Maps, Recommendation system, Dependency resolution। Complex relationships-এ।
🔀 Merge Sort
বড় data sort করতে। O(n log n) stable। Laravel-এ orderBy() ব্যবহার করো।
⚡ Binary Search
Sorted data-তে দ্রুত খোঁজা। O(log n)। DB index-এর পেছনে এই idea।
🗂️ Hash Map
PHP associative array! O(1) lookup। Cache, Session, Config-এ সব জায়গায়।
