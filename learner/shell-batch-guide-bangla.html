📌 স্ক্রিপ্ট মানে কী?

স্ক্রিপ্ট হলো একটি টেক্সট ফাইল যেখানে একের পর এক কমান্ড লেখা থাকে। কম্পিউটার সেই কমান্ডগুলো উপর থেকে নিচে পড়ে এবং স্বয়ংক্রিয়ভাবে কাজ করে।

সহজ ভাষায়: আপনি যদি প্রতিদিন ১০টি কমান্ড টাইপ করেন, সেগুলো একটি ফাইলে লিখে রাখুন — একবার চালালেই সব হয়ে যাবে!

💡
উদাহরণ: ব্যাকআপ নেওয়া, ফোল্ডার তৈরি করা, ফাইল মুভ করা, সফটওয়্যার ইনস্টল করা — এসব কাজ স্ক্রিপ্ট দিয়ে অটোমেট করা যায়।

⚠️
মনে রাখুন: = এর দুই পাশে কোনো space দেওয়া যাবে না! name = "Rahim" ভুল, name="Rahim" সঠিক।

#!/bin/bash

# ── Variable তৈরি ──────────────────────────
name="রহিম"
age=25
city="ঢাকা"

# ── Variable ব্যবহার ($চিহ্ন দিয়ে) ────────

echo "আপনি: $whoami_result"

# ── Read-only (পরিবর্তন করা যাবে না) ───────
readonly PI=3.14159
echo "PI = $PI"

If-Else — শর্তভিত্তিক কাজ

conditions.sh
#!/bin/bash

read -p "সংখ্যা লিখুন: " num

# ── সংখ্যা তুলনা ──────────────────────────
if [ $num -gt 100 ]; then
    echo "১০০ এর বেশি"
elif [ $num -eq 100 ]; then
    echo "ঠিক ১০০"
else
    echo "১০০ এর কম"
fi

# ── তুলনা অপারেটর ─────────────────────────
# -eq  সমান (equal)
# -ne  সমান না (not equal)
# -gt  বড় (greater than)
# -lt  ছোট (less than)
# -ge  বড় বা সমান
# -le  ছোট বা সমান

# ── ফাইল চেক ──────────────────────────────
if [ -f "/etc/passwd" ]; then
    echo "ফাইলটি আছে"
fi

if [ -d "/home" ]; then
    echo "ফোল্ডারটি আছে"
fi

# ── String তুলনা ──────────────────────────
read -p "হ্যাঁ বা না? " ans
if [ "$ans" = "হ্যাঁ" ]; then
    echo "আপনি হ্যাঁ বললেন!"
fi

Loop — বারবার কাজ করা

loops.sh
#!/bin/bash

# ── For Loop ──────────────────────────────
echo "=== ফলের তালিকা ==="
for fruit in আম কলা লিচু আনারস; do
    echo "  ফল: $fruit"
done

# ── সংখ্যার range ─────────────────────────
for i in {1..5}; do
    echo "  সংখ্যা: $i"
done

# ── While Loop ────────────────────────────
count=1
while [ $count -le 3 ]; do
    echo "  গণনা: $count"
    count=$((count + 1))
done

# ── ফাইল পড়া loop দিয়ে ───────────────────
for file in *.txt; do
    echo "  পাওয়া গেছে: $file"
done

Function — পুনর্ব্যবহারযোগ্য কোড

functions.sh
#!/bin/bash

# ── Function তৈরি ─────────────────────────
greet() {
    echo "হ্যালো, $1! আপনাকে স্বাগতম।"
}

# ── Function call ─────────────────────────
greet "রহিম"
greet "করিম"

# ── যোগফল বের করার function ───────────────
add_numbers() {
    local result=$(($1 + $2))
    echo "$1 + $2 = $result"
}

add_numbers 15 27
add_numbers 100 200

# ── Return value ──────────────────────────
check_age() {
    if [ $1 -ge 18 ]; then
        echo "প্রাপ্তবয়স্ক"
    else
        echo "অপ্রাপ্তবয়স্ক"
    fi
}
status=$(check_age 20)
echo "বয়স যাচাই: $status"

IF-ELSE — শর্তভিত্তিক কাজ

conditions.bat
@echo off

SET /P num=একটি সংখ্যা লিখুন: 

REM ── সংখ্যা তুলনা ─────────────────────
IF %num% GTR 100 (
    echo ১০০ এর বেশি
) ELSE IF %num% EQU 100 (
    echo ঠিক ১০০
) ELSE (
    echo ১০০ এর কম
)

REM ── তুলনা অপারেটর ────────────────────
REM EQU = সমান,  NEQ = সমান না
REM GTR = বড়,   LSS = ছোট
REM GEQ = >=,    LEQ = <=

REM ── ফাইল চেক ────────────────────────
IF EXIST "C:\Windows\notepad.exe" (
    echo Notepad পাওয়া গেছে!
)

REM ── ফোল্ডার চেক ─────────────────────
IF EXIST "C:\Users" (
    echo Users ফোল্ডার আছে
)

REM ── String তুলনা ────────────────────
SET /P ans=চালিয়ে যাবেন? (হ্যাঁ/না): 
IF /I "%ans%"=="হ্যাঁ" echo ঠিক আছে, চলুন!

pause

FOR Loop — বারবার কাজ করা

loops.bat
@echo off

REM ── তালিকার উপর loop ─────────────────
echo === ফলের তালিকা ===
FOR %%f IN (আম কলা লিচু আনারস) DO (
    echo   ফল: %%f
)

REM ── সংখ্যার range: 1 থেকে 5 ───────────
REM FOR /L %%var IN (শুরু, বৃদ্ধি, শেষ)
FOR /L %%i IN (1,1,5) DO (
    echo   সংখ্যা: %%i
)

REM ── .txt ফাইল খোঁজা ─────────────────
echo === টেক্সট ফাইল ===
FOR %%f IN (*.txt) DO (
    echo   পাওয়া গেছে: %%f
)

REM ── GOTO দিয়ে loop (while এর মতো) ─────
SET count=1
:LOOP_START
IF %count% GTR 3 GOTO LOOP_END
echo   গণনা: %count%
SET /A count=%count%+1
GOTO LOOP_START
:LOOP_END

pause

GOTO END

:SHOW_DATE

GOTO END

:EXIT
echo বিদায়!
EXIT

:END
pause

Backup Script

নিচে একই কাজ (backup folder তৈরি ও ফাইল কপি) Shell ও Batch দুটি ভাষায় দেখানো হয়েছে:

🐧 Shell Script
backup.sh
#!/bin/bash
src="/home/user/docs"
dst="/backup"
date=$(date +%Y%m%d)

mkdir -p "$dst/$date"
cp -r "$src" "$dst/$date"
echo "Backup সম্পন্ন!"
🪟 Batch Script
backup.bat
@echo off
SET src=C:\Users\docs
SET dst=C:\Backup
SET dt=%DATE:~10,4%%DATE:~4,2%%DATE:~7,2%

MKDIR "%dst%\%dt%"
XCOPY "%src%" "%dst%\%dt%" /E /I
echo Backup সম্পন্ন!
pause

Automatic Folder Cleaner — Windows

cleaner.bat
@echo off
title ফোল্ডার ক্লিনার
color 0A

echo ╔═══════════════════════════════╗
echo ║   Downloads ফোল্ডার ক্লিনার   ║
echo ╚═══════════════════════════════╝
echo.

REM ── ফোল্ডার তৈরি ─────────────────────
SET base=C:\Users\%USERNAME%\Downloads

MKDIR "%base%\Images"    2>nul
MKDIR "%base%\Videos"    2>nul
MKDIR "%base%\Documents"  2>nul
MKDIR "%base%\Archives"  2>nul

echo ফোল্ডার তৈরি হয়েছে।

REM ── ফাইল সরানো ────────────────────────
FOR %%f IN ("%base%\*.jpg" "%base%\*.png" "%base%\*.gif") DO
    IF EXIST "%%f" MOVE "%%f" "%base%\Images\"

FOR %%f IN ("%base%\*.mp4" "%base%\*.mkv") DO
    IF EXIST "%%f" MOVE "%%f" "%base%\Videos\"

FOR %%f IN ("%base%\*.zip" "%base%\*.rar") DO
    IF EXIST "%%f" MOVE "%%f" "%base%\Archives\"

echo.
