01 React JS কি এবং কেন শিখবো?

1

03 JSX — JavaScript এর মধ্যে HTML

JSX মানে JavaScript XML। এটি JavaScript এর মধ্যে HTML লেখার একটি বিশেষ সিনট্যাক্স। Browser সরাসরি JSX বোঝে না, Babel এটাকে JavaScript এ convert করে।

JSX ছাড়া (Babel এর পরে এটাই হয়)
// এভাবে লিখতে হতো
const element = React.createElement(
  'h1',
  { className: 'title' },
  'হ্যালো বাংলাদেশ'
);

// Nested element
const card = React.createElement(
  'div',
  { className: 'card' },
  React.createElement('h2', null, 'Title'),
  React.createElement('p', null, 'Content')
);
JSX দিয়ে (React তে যা লিখি)
// JSX — পড়তে ও লিখতে সহজ!
const element = (
  <h1 className="title">
    হ্যালো বাংলাদেশ
  </h1>
);

// Nested — পড়া সহজ
const card = (
  <div className="card">
    <h2>Title</h2>
    <p>Content</p>
  </div>
);

⚠️ JSX এর গুরুত্বপূর্ণ নিয়ম

⚠️
১. class → className
HTML: <div class="box">
JSX: <div className="box">
⚠️
২. একটি Root Element থাকতে হবে
ভুল: <h1>...</h1><p>...</p>
সঠিক: <div><h1>...</h1><p>...</p></div> অথবা <>...</> (Fragment)
⚠️
৩. Self-closing tags বন্ধ করতে হবে
HTML: <br>  JSX: <br />

    </div>
  );
};
04 Components — React এর মূল ভিত্তি

Component হলো React এর একটি স্বাধীন, পুনর্ব্যবহারযোগ্য UI অংশ। এটি একটি JavaScript Function যা JSX return করে। Component এর নাম সবসময় Capital Letter দিয়ে শুরু হয়।

Class Component (পুরনো উপায়)
import React, { Component } from 'react';

// Class Based Component
class Welcome extends Component {
  constructor(props) {
    super(props);
    this.state = { count: 0 };
  }

  render() {
    return (
      <div>
        <h1>হ্যালো {this.props.name}</h1>
        <p>Count: {this.state.count}</p>
      </div>
    );
  }
}
export default Welcome;
Functional Component (আধুনিক, এটাই শেখো)
import { useState } from 'react';

// Functional Component — সহজ ও আধুনিক
const Welcome = ({ name }) => {
  const [count, setCount] = useState(0);

  return (
    <div>
      <h1>হ্যালো {name}</h1>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>
        বাড়াও
      </button>
    </div>
  );
};
export default Welcome;

import { useState } from 'react';

const Counter = () => {

const UserForm = () => {

  const handleNameChange = (e) => {
    setUser({ ...user, name: e.target.value }); // Spread!
  };

  // Array তে নতুন item যোগ

  const removeItem = (index) => {

      <input value={user.name} onChange={handleNameChange} />
      <p>নাম: {user.name}</p>
    </div>

07 useEffect — Side Effects পরিচালনা

useEffect Hook দিয়ে আমরা Component এর Side Effects পরিচালনা করি। যেমন: API call, Timer, DOM manipulation, Subscription। এটি Class Component এর lifecycle methods (componentDidMount, componentDidUpdate, componentWillUnmount) এর বিকল্প।

Mount (প্রথমবার)
componentDidMount
useEffect(() => {}, [])
Update (পরিবর্তনে)
componentDidUpdate
useEffect(() => {}, [dep])
Unmount (সরানোর আগে)
componentWillUnmount
useEffect(() => { return () => {} })
useEffect — ৪ ধরনের ব্যবহার
import { useState, useEffect } from 'react';

const DataComponent = ({ userId }) => {
  const [user, setUser] = useState(null);
  const [count, setCount] = useState(0);

  // ১. প্রতিটি render এর পরে চলে (dependency নেই)
  useEffect(() => {
    console.log("Component render হলো");
  });

  // ২. শুধু প্রথমবার চলে (empty array [])
  useEffect(() => {
    console.log("Component প্রথমবার mount হলো");
    // API call এখানে করো
    fetch('https://api.example.com/data')
      .then(res => res.json())
      .then(data => setUser(data));
  }, []); // ← empty dependency array

  // ৩. userId পরিবর্তন হলে চলে
  useEffect(() => {
    console.log(`User ${userId} এর data লোড করো`);
    async function fetchUser() {
      const res = await fetch(`/api/users/${userId}`);
      const data = await res.json();
      setUser(data);
    }
    fetchUser();
  }, [userId]); // ← userId dependency

  // ৪. Cleanup (timer, subscription বন্ধ করা)
  useEffect(() => {
    const timer = setInterval(() => {
      setCount(c => c + 1);
    }, 1000);

    return () => clearInterval(timer); // ← Cleanup!
  }, []);

  if (!user) return <p>লোড হচ্ছে...</p>;
  return <div><h2>{user.name}</h2></div>;
};

const ConditionalDemo = ({ isLoggedIn, role, score, items }) => {
  // ১. if-else (জটিল logic এর জন্য)
  if (!isLoggedIn) {
    return <p>অনুগ্রহ করে লগইন করুন</p>;
  }

  return (
    <div>
      {/* ২. Ternary Operator — সবচেয়ে বেশি ব্যবহার */}
      {isLoggedIn
        ? <p>✅ স্বাগতম!</p>
        : <p>❌ লগইন করুন</p>
      }

      {/* ৩. && Operator — শুধু show/hide */}
      {role === "admin" && <button>Admin Panel</button>}
      {items.length > 0 && <p>{items.length}টি item</p>}

      {/* ৪. Switch-like (অনেক condition) */}
      {score >= 90 ? <span>🥇 A+</span>
        : score >= 80 ? <span>🥈 A</span>
        : score >= 70 ? <span>🥉 B</span>
        : <span>📚 আরো পড়ো</span>
      }
    </div>
  );
};
11 Forms — Controlled Components

React এ Form input এর value State এ রাখা হয় — এটাকে বলে Controlled Component। React State সবসময় "single source of truth" হিসেবে কাজ করে।

Complete Form Example
import { useState } from 'react';

const RegistrationForm = () => {
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    password: "",
    gender: "male",
    agree: false
  });
  const [errors, setErrors] = useState({});

  // একটি handler দিয়ে সব input handle
  const handleChange = (e) => {
    const { name, value, type, checked } = e.target;
    setFormData({
      ...formData,
      [name]: type === "checkbox" ? checked : value
    });
  };

  const validate = () => {
    const newErrors = {};
    if (!formData.name) newErrors.name = "নাম দিন";
    if (!formData.email) newErrors.email = "ইমেইল দিন";
    if (formData.password.length < 6)
      newErrors.password = "পাসওয়ার্ড ৬ অক্ষরের বেশি";
    return newErrors;
  };

  const handleSubmit = (e) => {
    e.preventDefault(); // page reload বন্ধ!
    const newErrors = validate();
    if (Object.keys(newErrors).length > 0) {
      setErrors(newErrors);
    } else {
      console.log("Form Data:", formData);
      // API তে পাঠাও
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        name="name"
        value={formData.name}
        onChange={handleChange}
        placeholder="আপনার নাম"
      />
      {errors.name && <span>{errors.name}</span>}

      <select name="gender" value={formData.gender} onChange={handleChange}>
        <option value="male">পুরুষ</option>
        <option value="female">মহিলা</option>
      </select>

      <input type="checkbox" name="agree" checked={formData.agree} onChange={handleChange} />
      <button type="submit">রেজিস্ট্রেশন করুন</button>
    </form>
  );
};

// Custom Hook — useFetch
const useFetch = (url) => {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch(url)
      .then(res => res.json())
      .then(d => { setData(d); setLoading(false); })
      .catch(err => setError(err));
  }, [url]);

  return { data, loading, error };
};

  return <p>{data?.length} জন User</p>;
