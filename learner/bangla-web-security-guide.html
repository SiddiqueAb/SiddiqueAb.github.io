Classic SQL Injection
ক্রিটিকাল
লগইন ফর্মে ' OR '1'='1 টাইপ করলেই পাসওয়ার্ড ছাড়া লগইন।
❌ ভুল পদ্ধতি (SQL Injection সম্ভব)
// ইউজার ইনপুট সরাসরি কোয়েরিতে
$username = $_POST['username'];
$password = $_POST['password'];

$query = "SELECT * FROM users 
  WHERE username='$username' 
  AND password='$password'";

// আক্রমণ: username = ' OR '1'='1
// ফলে কোয়েরি হয়: WHERE username=''
// OR '1'='1' -- সব ইউজার রিটার্ন!
✅ সঠিক পদ্ধতি (Prepared Statement)
// Prepared Statement ব্যবহার করুন
$stmt = $pdo->prepare(
  "SELECT * FROM users 
   WHERE username = ? 
   AND password = ?"
);

$stmt->execute([
  $_POST['username'],
  hash('sha256', $_POST['password'])
]);

$user = $stmt->fetch();
// ✅ SQL কোড কখনো ইনজেক্ট হবে না!
❌ ভুল পদ্ধতি
import sqlite3

def login(username, password):
  conn = sqlite3.connect('db.sqlite')
  query = f"SELECT * FROM users 
    WHERE user='{username}'"
  # f-string = বিপদ!
  return conn.execute(query)
✅ সঠিক পদ্ধতি
import sqlite3

def login(username, password):
  conn = sqlite3.connect('db.sqlite')
  query = "SELECT * FROM users 
    WHERE user=? AND pass=?"
  # ? placeholder ব্যবহার করুন
  return conn.execute(
    query, (username, password)
  )

❌ ভুল পদ্ধতি
// এগুলো কখনো করবেন না:

✅ সঠিক পদ্ধতি (bcrypt)
// bcrypt ব্যবহার করুন:

  $pass, 
  PASSWORD_BCRYPT,
  ['cost' => 12]
);

// যাচাই করা:
if (password_verify($pass, $hash)) {
  // লগইন সফল
}
❌ ভুল পদ্ধতি

  authMiddleware, // প্রথমে auth
  async (req, res) => {

      // ইউজারের অর্ডার কিনা চেক!

      );
    }

  }

  email: Joi.string()
    .email()
    .lowercase()
    .required(),
  age: Joi.number()
    .integer().min(18).max(120)
    .required(),
  phone: Joi.string()
    .pattern(/^01[3-9]\d{8}$/) // BD নম্বর
    .required()
});

// Middleware:
const { error } = registerSchema.validate(req.body);
if (error) {
  return res.status(400).json({
    error: error.details[0].message
  });
}

const multer = require('multer');
const path = require('path');
const crypto = require('crypto');

const upload = multer({
  limits: { fileSize: 5 * 1024 * 1024 }, // 5MB
  fileFilter: (req, file, cb) => {
    const allowed = ['.jpg', '.png', '.pdf'];
    const ext = path.extname(file.originalname)
      .toLowerCase();
    
    // MIME type চেক:
    const okMime = ['image/jpeg', 'image/png'];
    if (!allowed.includes(ext) || 
        !okMime.includes(file.mimetype)) {
      return cb(new Error('অননুমোদিত ফাইল!'));
    }
    cb(null, true);
  },
  storage: multer.diskStorage({
    destination: './uploads', // public এর বাইরে!
    filename: (req, file, cb) => {
      // র‍্যান্ডম নাম:
      const name = crypto.randomUUID() + 
        path.extname(file.originalname);
      cb(null, name);
    }
  })
});
