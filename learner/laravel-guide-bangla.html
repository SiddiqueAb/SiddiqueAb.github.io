Dynamic Table Name — ডায়নামিক টেবিল ব্যবস্থাপনা (ETKA Style)

ETKA-র মতো ডাটাবেজে টেবিলের নাম নির্দিষ্ট থাকে না — যেমন K001, K002, PR_456 ইত্যাদি নম্বার-ভিত্তিক নাম। Laravel-এ এটা হ্যান্ডেল করার জন্য Dynamic Model এবং Trait ব্যবহার করা হয়।

Dynamic Model — টেবিলের নাম রানটাইমে সেট করা

app/Models/DynamicModel.php 
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;

class DynamicModel extends Model
{
    // টেবিলের নাম ডায়নামিকভাবে সেট করার জন্য
    protected $table;

    /**
     * যেকোনো টেবিলের নাম দিয়ে Model তৈরি করুন
     * ব্যবহার: DynamicModel::forTable('K001')->get();
     */
    public static function forTable(string $tableName): static
    {
        $instance = new static();
        $instance->table = $tableName;
        return $instance;
    }

    // টেবিল নাম রিটার্ন করে
    public function getTable(): string
    {
        return $this->table ?? parent::getTable();
    }
}

ব্যবহার — Controller এ কিভাবে লিখবেন

app/Http/Controllers/EtkaController.php 
use App\Models\DynamicModel;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Schema;

class EtkaController extends Controller
{
    // ডায়নামিক টেবিল থেকে ডাটা আনা
    public function getPartsFromTable(string $tableCode)
    {
        // যেমন: K001, PR_123 ইত্যাদি
        $tableName = 'K' . str_pad($tableCode, 3, '0', STR_PAD_LEFT);

        // টেবিল আছে কিনা চেক করুন
        if (!Schema::hasTable($tableName)) {
            return back()->with('error', 'টেবিল পাওয়া যায়নি');
        }

        // DynamicModel দিয়ে যেকোনো টেবিল query করুন
        $parts = DynamicModel::forTable($tableName)
            ->newQuery()
            ->select(['PR_NR', 'MCODE', 'PRICE'])
            ->paginate(20);

        return inertia('Etka/Parts', [
            'parts' => $parts,
            'tableName' => $tableName,
        ]);
    }

    // সব ডায়নামিক টেবিলের তালিকা বের করা
    public function getAllDynamicTables()
    {
        $allTables = DB::select('SHOW TABLES');

        // K দিয়ে শুরু হওয়া ডায়নামিক টেবিল ফিল্টার করুন
        $dynamicTables = collect($allTables)
            ->map(fn($t) => array_values((array)$t)[0])
            ->filter(fn($name) => str_starts_with($name, 'K'))
            ->values();

        return inertia('Etka/Tables', ['tables' => $dynamicTables]);
    }
}
টেবিলের নাম সবসময় Schema::hasTable() দিয়ে আগে চেক করুন — না হলে error আসবে।

Meta Table — কোন টেবিলে কী আছে সেটা ট্র্যাক রাখুন

database/migrations/create_table_registry.php 
Schema::create('table_registry', function (Blueprint $table) {
    $table->id();
    $table->string('table_name');      // যেমন: K001
    $table->string('catalog_type');    // যেমন: VW, AUDI
    $table->string('description')->nullable();
    $table->boolean('is_active')->default(true);
    $table->timestamps();
});

Model — সঠিকভাবে Model লেখা

Model হলো ডাটাবেজের প্রতিনিধি। এতে শুধু ডাটার structure ও relationship থাকবে — business logic থাকবে না।
app/Models/Part.php 
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Factories\HasFactory;
use App\Traits\LogActivityTrait;

class Part extends Model
{
    use HasFactory, LogActivityTrait;

    // Mass assignment থেকে রক্ষা করার জন্য
    protected $fillable = [
        'part_no',
        'description',
        'price',
        'catalog_type',
        'mcode',
    ];

    // JSON response এ লুকানো field
    protected $hidden = ['deleted_at'];

    // Type casting
    protected function casts(): array
    {
        return [
            'price' => 'decimal:2',
            'is_active' => 'boolean',
        ];
    }

    // Relationship — একটি Part একটি Catalog এ থাকে
    public function catalog()
    {
        return $this->belongsTo(Catalog::class);
    }

    // Scope — শুধু Active parts আনার জন্য
    public function scopeActive($query)
    {
        return $query->where('is_active', true);
    }
}
Model-এ কখনো DB::select() কাঁচা query লিখবেন না — সেটা Repository-তে লিখুন।

Repository Pattern — Interface ও Implementation

Repository Pattern মানে হলো ডাটাবেজ কাজ আলাদা করা। Controller সরাসরি Model ব্যবহার করবে না — Repository এর মাধ্যমে করবে। এতে পরে MySQL থেকে MongoDB-তে যাওয়া সহজ হয়।

Base Repository — সবার জন্য Common কাজ

app/Repositories/BaseRepository.php 
<?php

namespace App\Repositories;

use Illuminate\Database\Eloquent\Model;

abstract class BaseRepository
{
    protected Model $model;

    public function getAll(int $perPage = 20)
    {
        return $this->model->latest()->paginate($perPage);
    }

    public function findById(int $id)
    {
        return $this->model->findOrFail($id);
    }

    public function create(array $data)
    {
        return $this->model->create($data);
    }

    public function update(int $id, array $data)
    {
        $record = $this->findById($id);
        $record->update($data);
        return $record->fresh();
    }

    public function delete(int $id): bool
    {
        return $this->model->destroy($id) > 0;
    }
}

PartRepository — নির্দিষ্ট কাজ

app/Repositories/PartRepository.php 
class PartRepository extends BaseRepository implements PartRepositoryInterface
{
    public function __construct(Part $model)
    {
        $this->model = $model;
    }

    // Part নম্বার দিয়ে খোঁজা
    public function findByPartNo(string $partNo)
    {
        return $this->model
            ->where('part_no', 'like', "%{$partNo}%")
            ->active()   // Model scope ব্যবহার
            ->paginate(20);
    }

    // Catalog অনুযায়ী Parts আনা
    public function getByCatalog(string $catalogType)
    {
        return $this->model
            ->where('catalog_type', $catalogType)
            ->with('catalog')
            ->paginate(20);
    }
}

ApiResponseTrait

app/Traits/ApiResponseTrait.php 
<?php

namespace App\Traits;

use Inertia\Inertia;

trait ApiResponseTrait
{
    // Inertia Page রেন্ডার করুন
    protected function render(string $page, array $props = [])
    {
        return Inertia::render($page, $props);
    }

    // সফল হলে Flash message সহ ফিরে যান
    protected function success(string $message, string $redirectTo = null)
    {
        $redirect = $redirectTo
            ? redirect()->route($redirectTo)
            : back();

        return $redirect->with('success', $message);
    }

    // Error হলে
    protected function error(string $message)
    {
        return back()->with('error', $message);
    }

    // JSON API response (যদি API দরকার হয়)
    protected function jsonSuccess($data, string $message = 'সফল')
    {
        return response()->json([
            'success' => true,
            'message' => $message,
            'data'    => $data,
        ]);
    }
}

LogActivityTrait — Model এ ব্যবহার

app/Traits/LogActivityTrait.php 
trait LogActivityTrait
{
    // Model save হওয়ার পর automatically চলবে
    public static function bootLogActivityTrait()
    {
        static::created(function ($model) {
            logger()->info($model::class . ' তৈরি হয়েছে', [
                'id'   => $model->id,
                'user' => auth()->id(),
            ]);
        });

        static::deleted(function ($model) {
            logger()->warning($model::class . ' মুছে ফেলা হয়েছে', [
                'id' => $model->id,
            ]);
        });
    }
}

Service Provider — Dependency Injection রেজিস্ট্রেশন

Service Provider হলো Laravel এর বুটস্ট্র্যাপ সিস্টেম। অ্যাপ চালু হওয়ার সময় Provider গুলো চলে। এখানে Interface কে কোন Class এ bind করবেন তা বলে দেওয়া হয়।
Controller চায় Interface
→
Provider বলে দেয় কোন Class দিতে হবে
→
Laravel Repository inject করে
app/Providers/RepositoryServiceProvider.php 
<?php

namespace App\Providers;

use Illuminate\Support\ServiceProvider;

// Interfaces
use App\Repositories\Interfaces\PartRepositoryInterface;

// Implementations
use App\Repositories\PartRepository;

class RepositoryServiceProvider extends ServiceProvider
{
    public function register(): void
    {
        // Interface → Implementation bind করুন
        $this->app->bind(
            PartRepositoryInterface::class,
            PartRepository::class
        );

        // নতুন Repository যোগ করতে শুধু এখানে bind করুন
        // $this->app->bind(CatalogRepositoryInterface::class, CatalogRepository::class);
    }

    public function boot(): void
    {
        // Inertia Shared Data — সব page এ পাঠানো data
        inertia()->share([
            'auth.user' => fn() => auth()->user()
                ?->only(['id', 'name', 'email', 'role']),
            'flash' => fn() => [
                'success' => session('success'),
                'error'   => session('error'),
            ],
        ]);
    }
}

Middleware হলো Request এবং Response এর মাঝখানের ফিল্টার। Controller পৌঁছানোর আগে Middleware চেক করে — User লগইন আছে কিনা, Role আছে কিনা ইত্যাদি।

Role Middleware — Admin/User আলাদা করা

app/Http/Middleware/RoleMiddleware.php 
<?php

namespace App\Http\Middleware;

use Closure;
use Illuminate\Http\Request;

class RoleMiddleware
{
    public function handle(Request $request, Closure $next, string ...$roles)
    {
        // লগইন আছে কিনা চেক
        if (!auth()->check()) {
            return redirect()->route('login');
        }

        $user = auth()->user();

        // Role চেক করুন
        if (!in_array($user->role, $roles)) {
            return inertia()->render('Errors/403')
                ->toResponse($request)
                ->setStatusCode(403);
        }

        return $next($request);
    }
}

Controller — Inertia সহ ক্লিন Controller

Controller শুধু Request নেবে, Repository কে বলবে, Response পাঠাবে — এর বেশি কিছু করবে না। Business Logic Service Class এ রাখুন।
app/Http/Controllers/PartController.php 
<?php

namespace App\Http\Controllers;

use App\Http\Requests\StorePartRequest;
use App\Repositories\Interfaces\PartRepositoryInterface;
use App\Traits\ApiResponseTrait;

class PartController extends Controller
{
    use ApiResponseTrait;

    // Constructor Injection — Provider এটা ঠিক করে দেবে
    public function __construct(
        private readonly PartRepositoryInterface $partRepo
    ) {}

    public function index()
    {
        $parts = $this->partRepo->getAll();
        return $this->render('Parts/Index', compact('parts'));
    }

    public function store(StorePartRequest $request)
    {
        // Request validation আলাদা class এ (StorePartRequest)
        $this->partRepo->create($request->validated());
        return $this->success('পার্ট সফলভাবে যোগ হয়েছে', 'parts.index');
    }

    public function update(StorePartRequest $request, int $id)
    {
        $this->partRepo->update($id, $request->validated());
        return $this->success('পার্ট আপডেট হয়েছে');
    }

    public function destroy(int $id)
    {
        $this->partRepo->delete($id);
        return $this->success('পার্ট মুছে ফেলা হয়েছে');
    }
}

class StorePartRequest extends FormRequest
{
    public function rules(): array
    {
        return [
            'part_no'      => 'required|string|max:50|unique:parts',
            'description'  => 'required|string|max:255',
            'price'        => 'required|numeric|min:0',
            'catalog_type' => 'required|in:VW,AUDI,SKODA,SEAT',
        ];
    }

    public function messages(): array
    {
        return [
            'part_no.required' => 'পার্ট নম্বার দিতে হবে',
            'price.numeric'    => 'মূল্য সংখ্যায় দিতে হবে',
        ];
    }
}

Cache ব্যবহার

Repository তে Cache ব্যবহার 
use Illuminate\Support\Facades\Cache;

public function getAllCatalogTypes()
{
    // ৬০ মিনিট cache রাখবে — বারবার DB query হবে না
    return Cache::remember('catalog_types', 60 * 60, function () {
        return $this->model->distinct()->pluck('catalog_type');
    });
}

// Data update হলে cache clear করুন
public function create(array $data)
{
    Cache::forget('catalog_types');
    return $this->model->create($data);
}

Lazy Loading Vue Components

resources/js/app.js — Inertia Entry Point 
import { createApp, h } from 'vue';
import { createInertiaApp } from '@inertiajs/vue3';
import { resolvePageComponent } from 'laravel-vite-plugin/inertia-helpers';

createInertiaApp({
  // Page গুলো Lazy Load হবে — সব একসাথে load হবে না
  resolve: (name) =>
    resolvePageComponent(
      `./Pages/${name}.vue`,
      import.meta.glob('./Pages/**/*.vue')
    ),
  setup({ el, App, props, plugin }) {
    createApp({ render: () => h(App, props) })
      .use(plugin)
      .mount(el);
  },
});
